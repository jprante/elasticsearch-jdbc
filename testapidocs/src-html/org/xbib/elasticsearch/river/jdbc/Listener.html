002package org.xbib.elasticsearch.river.jdbc;
003
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.testng.ITestContext;
007import org.testng.ITestListener;
008import org.testng.ITestResult;
009
010public class Listener implements ITestListener {
011
012    private final ESLogger logger = ESLoggerFactory.getLogger(Listener.class.getSimpleName());
013
014    @Override
015    public void onTestStart(ITestResult result) {
016        logger.info("------------------------------------------------------");
017        logger.info("starting test method {}", result.getName());
018        logger.info("------------------------------------------------------");
019    }
020
021    @Override
022    public void onTestSuccess(ITestResult result) {
023        logger.info("------------------------------------------------------");
024        logger.info("success of test method {}", result.getName());
025        logger.info("------------------------------------------------------");
026    }
027
028    @Override
029    public void onTestFailure(ITestResult result) {
030        logger.info("------------------------------------------------------");
031        logger.info("failure of test method {}", result.getName());
032        logger.info("------------------------------------------------------");
033    }
034
035    @Override
036    public void onTestSkipped(ITestResult result) {
037        logger.info("skipped test {}", result.getMethod().getMethodName());
038        result.setStatus(ITestResult.FAILURE);
039    }
040
041    @Override
042    public void onTestFailedButWithinSuccessPercentage(ITestResult result) {
043    }
044
045    @Override
046    public void onStart(ITestContext context) {
047        logger.info("------------------------------------------------------");
048        logger.info("starting test {}", context.getName());
049        logger.info("------------------------------------------------------");
050    }
051
052    @Override
053    public void onFinish(ITestContext context) {
054        logger.info("------------------------------------------------------");
055        logger.info("finished test {}", context.getName());
056        logger.info("------------------------------------------------------");
057    }
058
059}
