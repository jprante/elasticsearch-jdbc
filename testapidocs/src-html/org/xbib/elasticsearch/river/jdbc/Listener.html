001package org.xbib.elasticsearch.river.jdbc;
002
003import org.elasticsearch.common.logging.ESLogger;
004import org.elasticsearch.common.logging.ESLoggerFactory;
005import org.testng.ITestContext;
006import org.testng.ITestListener;
007import org.testng.ITestResult;
008
009public class Listener implements ITestListener {
010
011    private final ESLogger logger = ESLoggerFactory.getLogger(Listener.class.getSimpleName());
012    
013    @Override
014    public void onTestStart(ITestResult result) {
015        logger.info("------------------------------------------------------");
016        logger.info("Starting test method: {}", result.getName());
017    }
018
019    @Override
020    public void onTestSuccess(ITestResult result) {
021    }
022
023    @Override
024    public void onTestFailure(ITestResult result) {
025    }
026
027    @Override
028    public void onTestSkipped(ITestResult result) {
029        logger.info("Skipped test: {}", result.getMethod().getMethodName());
030        result.setStatus(ITestResult.FAILURE);
031    }
032
033    @Override
034    public void onTestFailedButWithinSuccessPercentage(ITestResult result) {
035    }
036
037    @Override
038    public void onStart(ITestContext context) {
039        logger.info("------------------------------------------------------");
040        logger.info("Starting test: {}", context.getName());    
041    }
042
043    @Override
044    public void onFinish(ITestContext context) {
045    }
046
047}
