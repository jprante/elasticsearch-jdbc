058    @Test
059    @Parameters({"sql1"})
060    public void testBill(String sql) throws Exception {
061        try {
062            List<Object> params = new ArrayList();
063            RiverMouth target = new MockRiverMouth() {
064                @Override
065                public void index(StructuredObject object) throws IOException {
066                    logger.debug("sql1 object={}", object);
067                }
068            };
069            PreparedStatement statement = source.prepareQuery(sql);
070            source.bind(statement, params);
071            ResultSet results = source.executeQuery(statement);
072            SimpleValueListener listener = new SimpleValueListener().target(target);
073            long rows = 0L;
074            source.beforeFirstRow(results, listener);
075            while (source.nextRow(results, listener)) {
076                rows++;
077            }
078            listener.reset();
079            assertEquals(rows, 5);
080            source.close(results);
081            source.close(statement);
082        } catch (Exception e) {
083            logger.error(e.getMessage(), e);
084        }
085    }
