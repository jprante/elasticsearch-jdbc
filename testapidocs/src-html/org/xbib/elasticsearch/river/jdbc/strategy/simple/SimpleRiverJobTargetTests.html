019package org.xbib.elasticsearch.river.jdbc.strategy.simple;
020
021import org.elasticsearch.client.Client;
022import org.elasticsearch.river.RiverName;
023import org.elasticsearch.river.RiverSettings;
024import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
025import org.xbib.elasticsearch.river.jdbc.RiverSource;
026import org.xbib.elasticsearch.river.jdbc.support.AbstractRiverNodeTest;
027import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
028import org.testng.annotations.Parameters;
029import org.testng.annotations.Test;
030
031import java.io.IOException;
032import java.sql.Connection;
033import java.sql.PreparedStatement;
034import java.sql.ResultSet;
035import java.sql.SQLException;
036import java.util.ArrayList;
037import java.util.List;
038import java.util.UUID;
039
040public class SimpleRiverJobTargetTests extends AbstractRiverNodeTest {
041
042    private Client client;
043
044    @Override
045    public RiverSource getRiverSource() {
046        return new SimpleRiverSource();
047    }
048
049    @Override
050    public RiverContext getRiverContext() {
051        RiverContext context = new RiverContext();
052        context.digesting(true);
053        return context;
054    }
055
056    @Test
057    @Parameters({"river1", "sql1", "sql2"})
058    public void testSimpleRiverJob(String riverResource, String sql1, String sql2) throws IOException, InterruptedException, SQLException {
059        Connection connection = source.connectionForWriting();
060        createRandomProductsJob(connection, sql2, 100);
061        source.closeWriting();
062
063        connection = source.connectionForReading();
064        ResultSet results = connection.createStatement().executeQuery(sql1);
065        if (!connection.getAutoCommit()) {
066            connection.commit();
067        }
068        int count = results.next() ? results.getInt(1) : -1;
069        source.close(results);
070        source.closeReading();
071        assertEquals(count, 100);
072
073        client = client("1");
074        assertEquals(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits(), 0);
075
076        RiverSettings settings = riverSettings(riverResource);
077        JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, "_river", client);
078        river.once();
079
080        Thread.sleep(3000L); // let some good things happen
081
082        assertEquals(client.prepareSearch(INDEX).setTypes(TYPE).execute().actionGet().getHits().getTotalHits(), 100);
083
084        connection = source.connectionForReading();
085        results = connection.createStatement().executeQuery(sql1);
086        count = results.next() ? results.getInt(1) : -1;
087        results.close();
088        assertEquals(count, 0);
089
090        river.close();
091    }
092
093    private void createRandomProductsJob(Connection connection, String sql, int size)
094            throws SQLException {
095        for (int i = 0; i < size; i++) {
096            long amount = Math.round(Math.random() * 1000);
097            double price = (Math.random() * 10000) / 100.00;
098            add(connection, sql, "1", UUID.randomUUID().toString().substring(0, 32), amount, price);
099        }
100        if (!connection.getAutoCommit()) {
101            connection.commit();
102        }
103    }
104
105    private void add(Connection connection, String sql, final String job, final String name, final long amount, final double price)
106            throws SQLException {
107        PreparedStatement stmt = connection.prepareStatement(sql);
108        List<Object> params = new ArrayList() {
109            {
110                add(job);
111                add(name);
112                add(amount);
113                add(price);
114            }
115        };
116        source.bind(stmt, params);
117        stmt.execute();
118    }
119}
