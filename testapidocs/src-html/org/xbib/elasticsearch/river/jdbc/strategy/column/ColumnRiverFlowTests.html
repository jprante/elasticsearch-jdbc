002package org.xbib.elasticsearch.river.jdbc.strategy.column;
003
004import java.io.IOException;
005import java.util.Map;
006
007import org.elasticsearch.action.get.GetResponse;
008import org.elasticsearch.client.Client;
009import org.elasticsearch.client.Requests;
010import org.elasticsearch.common.unit.TimeValue;
011import org.elasticsearch.common.xcontent.XContentBuilder;
012import org.elasticsearch.common.xcontent.XContentFactory;
013import org.elasticsearch.indices.IndexMissingException;
014import org.elasticsearch.river.RiverName;
015import org.elasticsearch.river.RiverSettings;
016
017import org.testng.annotations.Parameters;
018import org.testng.annotations.Test;
019
020import org.xbib.elasticsearch.river.jdbc.RiverFlow;
021import org.xbib.elasticsearch.river.jdbc.RiverSource;
022import org.xbib.elasticsearch.river.jdbc.strategy.mock.MockRiverMouth;
023import org.xbib.elasticsearch.river.jdbc.strategy.mock.MockRiverSource;
024import org.xbib.elasticsearch.river.jdbc.support.AbstractRiverNodeTest;
025import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
026
027import static org.xbib.elasticsearch.river.jdbc.RiverFlow.ID_INFO_RIVER_INDEX;
028import static org.xbib.elasticsearch.river.jdbc.strategy.column.ColumnRiverFlow.LAST_RUN_TIME;
029
030public class ColumnRiverFlowTests extends AbstractRiverNodeTest {
031
032    @Override
033    public RiverSource getRiverSource() {
034        return new ColumnRiverSource();
035    }
036
037    @Override
038    public RiverContext getRiverContext() {
039        return new RiverContext();
040    }
041    
042    @Test()
043    @Parameters({"river-existedWhereClause"})
044    public void testWriteLastRiverRunTimeToCustomRiverInfo(String riverResource) throws IOException {
045
046        final Client client = client("1");
047        setupContext(new MockRiverSource(){
048            @Override
049            public void fetch() {
050            }
051        }, riverResource, client);
052        
053        RiverFlow flow = new ColumnRiverFlow();
054        flow.riverContext(context);
055        
056        flow.move();
057        
058        assertLastRiverRunTimeExists(client);
059    }
060    
061    @Test()
062    @Parameters({"river-existedWhereClause"})
063    public void testReadLastRiverRunTimeFromCustomRiverInfo(String riverResource) throws IOException {
064        final Client client = client("1");
065        
066        final TimeValue lastRunAt = new TimeValue(600);
067        
068        setupContext(new MockRiverSource(){
069            @Override
070            public void fetch() {
071                assertLastRunDateEquals(lastRunAt);
072            }
073        }, riverResource, client);
074        
075        
076        writeLastRunDateToCustomRiverInfo(client, lastRunAt);
077
078        
079        RiverFlow flow = new ColumnRiverFlow();
080        flow.riverContext(context);
081        
082        flow.move();
083    }
084    
085    private void setupContext(RiverSource riverSource, String riverResource, final Client client) throws IOException {
086        RiverSettings riverSettings = riverSettings(riverResource);      
087
088        context.riverMouth(new MockRiverMouth(){
089            @Override
090            public Client client() {
091                return client;
092            }            
093        });
094        context.riverSource(riverSource);
095        context.riverName(new RiverName(INDEX, TYPE).getName());
096        context.riverSettings(riverSettings.settings());
097        context.columnEscape(true);
098    }
099    
100    private void assertLastRiverRunTimeExists(Client client) {
101        try {
102            GetResponse get = client.prepareGet("_river", context.riverName(), ID_INFO_RIVER_INDEX).execute().actionGet();
103            Map map = (Map) get.getSourceAsMap().get("jdbc");
104
105            assertNotNull(map, "jdbc key not exists in custom info");        
106            assertNotNull(map.get(ColumnRiverFlow.LAST_RUN_TIME), "last run time should not be null");
107        } catch(IndexMissingException e) {
108            fail("custom info not found");
109        }
110    }
111    
112    private void assertLastRunDateEquals(TimeValue expectedLastRunAt) {
113        Map map = (Map) context.riverSettings().get("jdbc");
114        assertNotNull(map);
115        assertEquals(map.get(ColumnRiverFlow.LAST_RUN_TIME), expectedLastRunAt);        
116    }
117
118    private void writeLastRunDateToCustomRiverInfo(Client client, TimeValue lastRunAt) throws IOException {
119        XContentBuilder builder = XContentFactory.jsonBuilder();
120
121        builder
122                .startObject()
123                .startObject("jdbc")
124                .field(LAST_RUN_TIME, lastRunAt.millis())
125                .endObject()
126                .endObject();
127
128        client.prepareBulk()
129                .add(Requests.indexRequest("_river")
130                    .type(context.riverName())
131                    .id(ID_INFO_RIVER_INDEX)
132                    .source(builder)
133                )
134                .execute()
135                .actionGet();
136    }
137}
