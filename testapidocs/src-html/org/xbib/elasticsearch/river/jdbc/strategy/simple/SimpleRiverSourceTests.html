042    @Test
043    public void testConnectionClose() throws Exception {
044        Connection connection = source.connectionForReading();
045        assertFalse(connection.isClosed());
046        source.closeReading();
047        assertTrue(connection.isClosed());
048        source.connectionForReading();
049    }
050
051    @Test
052    @Parameters({"sql1"})
053    public void testSQL(String sql) throws Exception {
054        PreparedStatement statement = source.prepareQuery(sql);
055        ResultSet results = source.executeQuery(statement);
056        for (int i = 0; i < 5; i++) {
057            assertTrue(results.next());
058        }
059        source.close(results);
060        source.close(statement);
061    }
062
063    @Test
064    @Parameters({"sql2", "n"})
065    public void testStarQuery(String sql, @Optional Integer n) throws Exception {
066        List<? extends Object> params = newLinkedList();
067        RiverMouth output = new MockRiverMouth() {
068            @Override
069            public void index(StructuredObject object, boolean create) throws IOException {
070                logger.debug("object={}", object);
071            }
072        };
073        PreparedStatement statement = source.prepareQuery(sql);
074        source.bind(statement, params);
075        ResultSet results = source.executeQuery(statement);
076        KeyValueStreamListener listener = new StructuredObjectKeyValueStreamListener()
077                .output(output);
078        long rows = 0L;
079        source.beforeRows(results, listener);
080        while (source.nextRow(results, listener)) {
081            rows++;
082        }
083        source.afterRows(results, listener);
084        assertEquals(rows, n == null ? 5 : n);
085        source.close(results);
086        source.close(statement);
087    }
088
089    @Test
090    @Parameters({"sql3"})
091    public void testNullInteger(String sql) throws Exception {
092        List<? extends Object> params = newLinkedList();
093        RiverMouth mouth = new MockRiverMouth() {
094            @Override
095            public void index(StructuredObject object, boolean create) throws IOException {
096                if (object == null || object.source() == null) {
097                    throw new IllegalArgumentException("object missing");
098                }
099                Values o = (Values)object.source().get("amount");
100                if (o == null) {
101                    o = (Values)object.source().get("AMOUNT"); // hsqldb is uppercase
102                }
103                if (!o.isNull()) {
104                    throw new IllegalArgumentException("amount not null??? " + o.getClass().getName() );
105                }
106            }
107        };
108        PreparedStatement statement = source.prepareQuery(sql);
109        source.bind(statement, params);
110        ResultSet results = source.executeQuery(statement);
111        KeyValueStreamListener listener = new StructuredObjectKeyValueStreamListener()
112                .output(mouth);
113        long rows = 0L;
114        source.beforeRows(results, listener);
115        if (source.nextRow(results, listener)) {
116            // only one row
117            rows++;
118        }
119        source.afterRows(results, listener);
120        assertEquals(rows, 1);
121        source.close(results);
122        source.close(statement);
123    }
