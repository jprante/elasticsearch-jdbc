008    import org.testng.annotations.Test;
009    import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
010    import org.xbib.elasticsearch.river.jdbc.RiverSource;
011    import org.xbib.elasticsearch.river.jdbc.support.AbstractRiverNodeTest;
012    import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
013    
014    import java.sql.Connection;
015    import java.sql.PreparedStatement;
016    import java.sql.SQLException;
017    import java.util.ArrayList;
018    import java.util.List;
019    import java.util.UUID;
020    
021    import static org.hamcrest.MatcherAssert.assertThat;
022    import static org.hamcrest.Matchers.greaterThan;
023    
024    public class SimpleRiverScheduleTests extends AbstractRiverNodeTest {
025    
026        @Override
027        public RiverSource getRiverSource() {
028            return new SimpleRiverSource();
029        }
030    
031        @Override
032        public RiverContext getRiverContext() {
033            return new RiverContext();
034        }
035    
036        /**
037         * Product table star select, scheduled for more than one run
038         */
039        @Test
040        @Parameters({"river6", "sql1"})
041        public void testSchedule(String riverResource, String sql) throws Exception {
042            Connection connection = source.connectionForWriting();
043            createRandomProducts(connection, sql, 100);
044            source.closeWriting();
045            Client client = client("1");
046            RiverSettings settings = riverSettings(riverResource);
047            JDBCRiver river = new JDBCRiver(new RiverName(INDEX, TYPE), settings, client);
048            river.start();
049            Thread.sleep(12500L); // let the river run at least twice
050            assertThat(client.prepareSearch(INDEX).execute().actionGet().getHits().getTotalHits(),
051                greaterThan(104L));
052            river.close();
053        }
054    
055        private void createRandomProducts(Connection connection, String sql, int size)
056                throws SQLException {
057            for (int i = 0; i < size; i++) {
058                long amount = Math.round(Math.random() * 1000);
059                double price = (Math.random() * 10000) / 100.00;
060                addData(connection, sql, UUID.randomUUID().toString().substring(0, 32), amount, price);
061            }
062            if (!connection.getAutoCommit()) {
063                connection.commit();
064            }
065        }
066    
067        private void addData(Connection connection, String sql, final String name, final long amount, final double price)
068                throws SQLException {
069            PreparedStatement stmt = connection.prepareStatement(sql);
070            List<Object> params = new ArrayList<Object>() {{
071                add(name);
072                add(amount);
073                add(price);
074            }};
075            source.bind(stmt, params);
076            stmt.execute();
077        }

079    }
