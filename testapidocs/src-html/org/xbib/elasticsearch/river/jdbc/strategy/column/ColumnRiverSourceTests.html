088        final int updatedRecordsInTimeRange = 4;

112    public void testColumnRiver_removeObjects(String riverResource, String insertSql, String updateSql) throws SQLException, IOException, InterruptedException {

115
116        boolean[] shouldProductsBeDeleted = new boolean[] { true, true, false };
117        
118        ProductFixtures productFixtures = createFixturesAndPopulateMouth(shouldProductsBeDeleted, riverMouth);
119        
120        testColumnRiver(riverMouth, riverResource, insertSql, productFixtures.fixtures, productFixtures.expectedCount);
121    }

123    private ProductFixtures createFixturesAndPopulateMouth(boolean[] shouldProductsBeDeleted, MockRiverMouth riverMouth) throws IOException {
124        ProductFixture[] fixtures = new ProductFixture[shouldProductsBeDeleted.length];
125        int expectedExistsCountAfterRiverRun = 0;
126        for(int i=0; i<shouldProductsBeDeleted.length; i++) {           
127            riverMouth.index(new PlainStructuredObject() {}
128                    .id(Integer.toString(i))
129                    .source(createStructuredObjectSource(i))
130                    .optype(Operations.OP_DELETE));
131            
132            Timestamp deletedAt;
133            
134            if(shouldProductsBeDeleted[i]) {
135                deletedAt = okTimestamp();
136            } else {
137                deletedAt = oldTimestamp();
138                expectedExistsCountAfterRiverRun++;
139            }
140            
141            fixtures[i] = ProductFixture.one()
142                    .setId(i)
143                    .createdAt(oldTimestamp())
144                    .updatedAt(oldTimestamp())
145                    .deletedAt(deletedAt);
146        }
147        ProductFixtures productFixtures = new ProductFixtures(fixtures, expectedExistsCountAfterRiverRun);
148        return productFixtures;
149    }

151    private Map<String, ? super Object> createStructuredObjectSource(int id) {
152        Map<String, ? super Object> map = new HashMap<String, Object>();
153        map.put("id", id);
154        map.put("name", null);
155        return map;
156    }

158    private void testColumnRiver(MockRiverMouth riverMouth, String riverResource, String sql, ProductFixture[] fixtures, int expectedHits) throws SQLException, IOException, InterruptedException {
159        createData(sql, fixtures);
160
161        context.riverMouth(riverMouth);        
162        
163        setContextSettings(riverResource);
164        
165        source.fetch();
166
167        assertEquals(riverMouth.data().size(), expectedHits);
168    }
169    
170    private void setContextSettings(String riverResource) throws IOException {
171        RiverSettings riverSettings = riverSettings(riverResource);
172        
173        Map<String, Object> settings = (Map<String, Object>) riverSettings.settings().get("jdbc");
174        logger.info("settings: "+settings);
175        
176        settings.put(ColumnRiverFlow.LAST_RUN_TIME, LAST_RUN_TIME);
177        settings.put(ColumnRiverFlow.CURRENT_RUN_STARTED_TIME, CURRENT_RUN_START_TIME);
178        
179        context.pollStatement(XContentMapValues.nodeStringValue(settings.get("sql"), null));
180        context.columnCreatedAt(XContentMapValues.nodeStringValue(settings.get("column_created_at"), null));
181        context.columnUpdatedAt(XContentMapValues.nodeStringValue(settings.get("column_updated_at"), null));
182        context.columnDeletedAt(XContentMapValues.nodeStringValue(settings.get("column_deleted_at"), null));
183        context.riverSettings(riverSettings.settings());
184        context.pollStatementParams(XContentMapValues.extractRawValues("sqlparams", settings));
185        context.columnEscape(true);
186    }

188    private Timestamp okTimestamp() {
189        return new Timestamp(LAST_RUN_TIME.getMillis() + 1000);
190    }
191    
192    private Timestamp oldTimestamp() {
193        return new Timestamp(LAST_RUN_TIME.getMillis() - 1000);
194    }
195    
196    private void createData(String sql, ProductFixture[] fixtures) throws SQLException {
197        Connection conn = source.connectionForWriting();
198        
199        for(ProductFixture fixture : fixtures) {
200            createData(conn, sql, fixture);
201        }
202        
203        source.closeWriting();
204    }
205    
206    private void createData(Connection connection, String sql, ProductFixture fixture) throws SQLException {
207        PreparedStatement stmt = connection.prepareStatement(sql);
208        
209        logger.info("timestamps: ["+fixture.createdAt+", "+fixture.updatedAt+", "+fixture.deletedAt+"]");
210        
211        for(int i=0; i<fixture.size; i++) {
212            stmt.setInt(1, fixture.id >= 0 ? fixture.id : random.nextInt());
213            stmt.setNull(2, Types.VARCHAR);
214            stmt.setInt(3, 1);
215            stmt.setDouble(4, 1.1);
216            
217            if(fixture.createdAt != null) {
218                stmt.setTimestamp(5, fixture.createdAt);
219            } else {
220                stmt.setNull(5, Types.TIMESTAMP);
221            }
222            
223            if(fixture.updatedAt != null) {
224                stmt.setTimestamp(6, fixture.updatedAt);
225            } else {
226                stmt.setNull(6, Types.TIMESTAMP);
227            }
228            
229            if(fixture.deletedAt != null) {
230                stmt.setTimestamp(7, fixture.deletedAt);
231            } else {
232                stmt.setNull(7, Types.TIMESTAMP);
233            }
234
235            stmt.execute();
236        }
237    }
