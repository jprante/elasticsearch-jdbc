116    private ProductFixtures createFixturesAndPopulateMouth(boolean[] shouldProductsBeDeleted, MockRiverMouth riverMouth)
117            throws IOException {
118        ProductFixture[] fixtures = new ProductFixture[shouldProductsBeDeleted.length];
119        int expectedExistsCountAfterRiverRun = 0;
120        for(int i=0; i<shouldProductsBeDeleted.length; i++) {           
121            riverMouth.index(new PlainIndexableObject() {}
122                    .id(Integer.toString(i))
123                    .source(createSource(i))
124                    .optype("delete"), false);
125            
126            Timestamp deletedAt;
127            
128            if(shouldProductsBeDeleted[i]) {
129                deletedAt = okTimestamp();
130            } else {
131                deletedAt = oldTimestamp();
132                expectedExistsCountAfterRiverRun++;
133            }
134            
135            fixtures[i] = ProductFixture.one()
136                    .setId(i)
137                    .createdAt(oldTimestamp())
138                    .updatedAt(oldTimestamp())
139                    .deletedAt(deletedAt);
140        }
141        return new ProductFixtures(fixtures, expectedExistsCountAfterRiverRun);
142    }
143    
144    private Map<String,Object> createSource(int id) {
145        Map<String, Object> map = new HashMap<String, Object>();
146        map.put("id", id);
147        map.put("name", null);
148        return map;
149    }

151    private void testColumnRiver(MockRiverMouth riverMouth, String riverResource, String sql, ProductFixture[] fixtures, int expectedHits)
152            throws SQLException, IOException, InterruptedException {
153        createData(sql, fixtures);
154        context.riverMouth(riverMouth);
155        setContextSettings(riverResource);
156        source.fetch();
157        assertEquals(riverMouth.data().size(), expectedHits);
158    }
159    
160    private void setContextSettings(String riverResource) throws IOException {
161        RiverSettings riverSettings = riverSettings(riverResource);
162        
163        Map<String, Object> settings = (Map<String, Object>) riverSettings.settings().get("jdbc");
164        logger.info("settings: "+settings);
165        
166        settings.put(ColumnRiverFlow.LAST_RUN_TIME, LAST_RUN_TIME);
167        settings.put(ColumnRiverFlow.CURRENT_RUN_STARTED_TIME, CURRENT_RUN_START_TIME);
168        
169        context.setStatements(SQLCommand.parse(settings));
170        context.columnCreatedAt(XContentMapValues.nodeStringValue(settings.get("column_created_at"), null));
171        context.columnUpdatedAt(XContentMapValues.nodeStringValue(settings.get("column_updated_at"), null));
172        context.columnDeletedAt(XContentMapValues.nodeStringValue(settings.get("column_deleted_at"), null));
173        context.riverSettings(riverSettings.settings());
174        context.columnEscape(true);
175    }
176    
177    private Timestamp okTimestamp() {
178        return new Timestamp(LAST_RUN_TIME.getMillis() + 1000);
179    }
180    
181    private Timestamp oldTimestamp() {
182        return new Timestamp(LAST_RUN_TIME.getMillis() - 1000);
183    }
184    
185    private void createData(String sql, ProductFixture[] fixtures) throws SQLException {
186        Connection conn = source.connectionForWriting();
187        
188        for(ProductFixture fixture : fixtures) {
189            createData(conn, sql, fixture);
190        }
191        
192        source.closeWriting();
193    }
194    
195    private void createData(Connection connection, String sql, ProductFixture fixture) throws SQLException {
196        PreparedStatement stmt = connection.prepareStatement(sql);
197        
198        logger.info("timestamps: ["+fixture.createdAt+", "+fixture.updatedAt+", "+fixture.deletedAt+"]");
199        
200        for(int i=0; i<fixture.size; i++) {
201            stmt.setInt(1, fixture.id >= 0 ? fixture.id : random.nextInt());
202            stmt.setNull(2, Types.VARCHAR);
203            stmt.setInt(3, 1);
204            stmt.setDouble(4, 1.1);
205            
206            if(fixture.createdAt != null) {
207                stmt.setTimestamp(5, fixture.createdAt);
208            } else {
209                stmt.setNull(5, Types.TIMESTAMP);
210            }
211            
212            if(fixture.updatedAt != null) {
213                stmt.setTimestamp(6, fixture.updatedAt);
214            } else {
215                stmt.setNull(6, Types.TIMESTAMP);
216            }
217            
218            if(fixture.deletedAt != null) {
219                stmt.setTimestamp(7, fixture.deletedAt);
220            } else {
221                stmt.setNull(7, Types.TIMESTAMP);
222            }
223
224            stmt.execute();
225        }
226    }
