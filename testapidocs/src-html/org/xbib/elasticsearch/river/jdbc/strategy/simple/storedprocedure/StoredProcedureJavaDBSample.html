001package org.xbib.elasticsearch.river.jdbc.strategy.simple.storedprocedure;
002
003import java.sql.Connection;
004import java.sql.DriverManager;
005import java.sql.PreparedStatement;
006import java.sql.ResultSet;
007import java.sql.SQLException;
008import java.sql.Statement;
009
010public class StoredProcedureJavaDBSample {
011
012    public static void showSuppliers(ResultSet[] rs)
013            throws SQLException {
014
015        Connection con = DriverManager.getConnection("jdbc:default:connection");
016        Statement stmt = null;
017
018        String query =
019                "select SUPPLIERS.SUP_NAME, " +
020                        "COFFEES.COF_NAME " +
021                        "from SUPPLIERS, COFFEES " +
022                        "where SUPPLIERS.SUP_ID = " +
023                        "COFFEES.SUP_ID " +
024                        "order by SUP_NAME";
025
026        stmt = con.createStatement();
027        rs[0] = stmt.executeQuery(query);
028    }
029
030    public static void getSupplierOfCoffee(String coffeeName, String[] supplierName)
031            throws SQLException {
032
033        Connection con = DriverManager.getConnection("jdbc:default:connection");
034        PreparedStatement pstmt = null;
035        ResultSet rs = null;
036
037        String query =
038                "select SUPPLIERS.SUP_NAME " +
039                        "from SUPPLIERS, COFFEES " +
040                        "where " +
041                        "SUPPLIERS.SUP_ID = COFFEES.SUP_ID " +
042                        "and ? = COFFEES.COF_NAME";
043
044        pstmt = con.prepareStatement(query);
045        pstmt.setString(1, coffeeName);
046        rs = pstmt.executeQuery();
047
048        if (rs.next()) {
049            supplierName[0] = rs.getString(1);
050        } else {
051            supplierName[0] = null;
052        }
053    }
054
055}
