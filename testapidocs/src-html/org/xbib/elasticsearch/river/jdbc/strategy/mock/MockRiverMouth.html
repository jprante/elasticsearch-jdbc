031public class MockRiverMouth implements RiverMouth {

033    private Map<String, String> data;

036
037    @Override
038    public String strategy() {
039        return "mock";
040    }
041
042    public MockRiverMouth() {
043        data = new TreeMap(); // sort order for stability in assertions
044        counter = 0L;
045    }
046
047    @Override
048    public void create(StructuredObject object) throws IOException {
049        data.put(object.toString(), object.build());
050        //logger.debug("got data for creation: {}", data);
051        counter++;
052    }
053
054    @Override
055    public void index(StructuredObject object) throws IOException {
056        data.put(object.toString(), object.build());
057        //logger.debug("got data for index: {}", data);
058        counter++;
059    }
060
061    @Override
062    public void delete(StructuredObject object) throws IOException {
063        data.remove(object.toString());
064        //logger.debug("got data for delete: {}", data);
065        counter--;
066    }
067

069        return data;
070    }
071
