001package org.xbib.elasticsearch.river.jdbc.strategy.mock;
002
003import org.elasticsearch.common.logging.ESLogger;
004import org.elasticsearch.common.logging.ESLoggerFactory;
005import org.xbib.elasticsearch.plugin.jdbc.IndexableObject;
006import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
007import org.xbib.elasticsearch.river.jdbc.RiverMouth;
008import org.xbib.elasticsearch.support.client.Ingest;
009
010import java.io.IOException;
011import java.util.Map;
012import java.util.TreeMap;
013
014public class MockRiverMouth implements RiverMouth {
015
016    private final static ESLogger logger = ESLoggerFactory.getLogger(MockRiverMouth.class.getName());
017
018    private Map<IndexableObject, String> data;
019
020    private long counter;
021
022    @Override
023    public String strategy() {
024        return "mock";
025    }
026
027    public MockRiverMouth() {
028        data = new TreeMap<IndexableObject, String>();
029        counter = 0L;
030    }
031
032    @Override
033    public void index(IndexableObject object, boolean create) throws IOException {
034        logger.info("index {} = {}", object.toString(), object.build());
035        data.put(object, object.build());
036        counter++;
037        logger.info("size after insert {}", data.size());
038    }
039
040    @Override
041    public void delete(IndexableObject object) throws IOException {
042        logger.info("delete {}", object.toString());
043        data.remove(object);
044        counter--;
045        logger.info("size after delete {}", data.size());
046    }
047
048    public Map<IndexableObject, String> data() {
049        return data;
050    }
051
052    @Override
053    public RiverMouth setRiverContext(RiverContext context) {
054        return this;
055    }
056
057    @Override
058    public RiverMouth setIngest(Ingest ingester) {
059        return this;
060    }
061
062    @Override
063    public RiverMouth setIndex(String index) {
064        return this;
065    }
066    @Override
067    public RiverMouth setType(String type) {
068        return this;
069    }
070    @Override
071    public RiverMouth setId(String id) {
072        return this;
073    }
074
075    @Override
076    public String getId() {
077        return null;
078    }
079
080    @Override
081    public void flush() throws IOException {
082    }
083
084    @Override
085    public void close() {
086    }
087
088    public long getCounter() {
089        return counter;
090    }
091
092}
