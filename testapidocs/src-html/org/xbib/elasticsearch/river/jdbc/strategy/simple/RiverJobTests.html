001package org.xbib.elasticsearch.river.jdbc.strategy.simple;
002
003import org.testng.annotations.Parameters;
004import org.testng.annotations.Test;
005import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
006import org.xbib.elasticsearch.river.jdbc.RiverSource;
007import org.xbib.elasticsearch.support.helper.AbstractRiverNodeTest;
008
009import java.io.IOException;
010import java.sql.Connection;
011import java.sql.ResultSet;
012import java.sql.SQLException;
013
014public class RiverJobTests extends AbstractRiverNodeTest {
015
016    @Override
017    public RiverSource getRiverSource() {
018        return new SimpleRiverSource();
019    }
020
021    @Override
022    public RiverContext getRiverContext() {
023        return new RiverContext();
024    }
025
026    @Test
027    @Parameters({"river1", "sql1", "sql2"})
028    public void testSimpleRiverJob(String riverResource, String sql1, String sql2)
029            throws IOException, InterruptedException, SQLException {
030        createRandomProductsJob(sql2, 100);
031        Connection connection = source.getConnectionForReading();
032        ResultSet results = connection.createStatement().executeQuery(sql1);
033        if (!connection.getAutoCommit()) {
034            connection.commit();
035        }
036        int count = results.next() ? results.getInt(1) : -1;
037        source.close(results);
038        source.closeReading();
039        assertEquals(count, 100);
040        //assertHits("1", 0);
041        createRiver(riverResource);
042        waitForRiverEnabled();
043        waitForInactiveRiver();
044        assertHits("1", 100);
045        connection = source.getConnectionForReading();
046        // sql1 = select count(*)
047        results = connection.createStatement().executeQuery(sql1);
048        if (!connection.getAutoCommit()) {
049            connection.commit();
050        }
051        count = results.next() ? results.getInt(1) : -1;
052        results.close();
053        assertEquals(count, 0);
054    }
055
056}
