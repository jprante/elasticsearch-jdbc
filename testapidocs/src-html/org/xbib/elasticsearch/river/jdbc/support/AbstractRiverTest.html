019package org.xbib.elasticsearch.river.jdbc.support;
020
021import org.elasticsearch.common.io.Streams;
022import org.elasticsearch.common.logging.ESLogger;
023import org.elasticsearch.common.logging.Loggers;
024import org.elasticsearch.common.settings.ImmutableSettings;
025import org.elasticsearch.common.unit.TimeValue;
026import org.elasticsearch.common.xcontent.XContentHelper;
027import org.elasticsearch.river.RiverSettings;
028import org.xbib.elasticsearch.river.jdbc.RiverSource;
029import org.testng.Assert;
030import org.testng.annotations.AfterMethod;
031import org.testng.annotations.BeforeMethod;
032import org.testng.annotations.Optional;
033import org.testng.annotations.Parameters;
034
035import java.io.BufferedReader;
036import java.io.IOException;
037import java.io.InputStream;
038import java.io.InputStreamReader;
039import java.sql.Connection;
040import java.sql.SQLException;
041import java.sql.Statement;
042
043public abstract class AbstractRiverTest extends Assert {
044
045    private static final ESLogger logger = Loggers.getLogger(AbstractRiverTest.class);
046
047    protected static RiverSource source;
048
049    protected static RiverContext context;
050
051    public abstract RiverSource getRiverSource();
052
053    public abstract RiverContext getRiverContext();
054
055    @BeforeMethod
056    @Parameters({"driver", "starturl", "user", "password"})
057    public void createClient(String driver, String starturl, String user, String password)
058            throws Exception {
059        source = getRiverSource()
060                .driver(driver)
061                .url(starturl)
062                .user(user)
063                .password(password);
064        context = getRiverContext()
065                .riverSource(source)
066                .retries(1)
067                .maxRetryWait(TimeValue.timeValueSeconds(5))
068                .locale("en");
069        context.contextualize();
070    }
071
072    @BeforeMethod(dependsOnMethods = {"createClient"})
073    @Parameters({"create"})
074    public void createTable(@Optional String resourceName) throws Exception {
075        if (resourceName == null || "".equals(resourceName)) {
076            return;
077        }
078        Connection connection = source.connectionForWriting();
079        if (connection == null) {
080            throw new IOException("no connection");
081        }
082        sqlScript(connection, resourceName);
083        source.closeWriting();
084    }
085
086    @AfterMethod
087    @Parameters({"delete"})
088    public void removeTable(@Optional String resourceName) throws Exception {
089        if (resourceName == null || "".equals(resourceName)) {
090            return;
091        }
092        // before dropping tables, open read connection must be closed to avoid hangs in mysql/postgresql
093        logger.debug("closing reads...");
094        source.closeReading();
095
096        logger.debug("connecting for close...");
097        Connection connection = source.connectionForWriting();
098        if (connection == null) {
099            throw new IOException("no connection");
100        }
101        logger.debug("cleaning...");
102        // clean up tables
103        sqlScript(connection, resourceName);
104        logger.debug("closing writes...");
105        source.closeWriting();
106    }
107
108    @AfterMethod(dependsOnMethods = {"removeTable"})
109    @Parameters({"driver", "starturl", "stopurl", "user", "password"})
110    public void removeClient(String driver, String starturl, String stopurl, String user, String password)
111            throws Exception {
112        // some driver can drop database by magic URL
113        source = getRiverSource()
114                .driver(driver)
115                .url(stopurl)
116                .user(user)
117                .password(password);
118        try {
119            logger.debug("connecting to stop URL...");
120            // activate stop URL
121            source.connectionForWriting();
122        } catch (Exception e) {
123            // exception is expected, ignore
124        }
125        // close open write connection
126        source.closeWriting();
127        logger.debug("stopped");
128    }
129
130    protected RiverSettings riverSettings(String resource)
131            throws IOException {
132        InputStream in = getClass().getResourceAsStream(resource);
133        return new RiverSettings(ImmutableSettings.settingsBuilder()
134                .build(), XContentHelper.convertToMap(Streams.copyToByteArray(in), false).v2());
135    }
136
137    private void sqlScript(Connection connection, String resourceName) throws Exception {
138        InputStream in = getClass().getResourceAsStream(resourceName);
139        BufferedReader br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
140        String sql;
141        while ((sql = br.readLine()) != null) {
142            try {
143                logger.debug("executing {}", sql);
144                Statement p = connection.createStatement();
145                p.execute(sql);
146                p.close();
147            } catch (SQLException e) {
148                // ignore
149                logger.error(sql + " failed. Reason: " + e.getMessage());
150            }
151        }
152        br.close();
153    }
154
155}
