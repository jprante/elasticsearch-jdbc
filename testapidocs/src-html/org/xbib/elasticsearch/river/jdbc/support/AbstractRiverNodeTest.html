019package org.xbib.elasticsearch.river.jdbc.support;
020
021import org.elasticsearch.action.admin.indices.create.CreateIndexRequest;
022import org.elasticsearch.action.admin.indices.delete.DeleteIndexRequest;
023import org.elasticsearch.client.Client;
024import org.elasticsearch.common.logging.ESLogger;
025import org.elasticsearch.common.logging.Loggers;
026import org.elasticsearch.common.network.NetworkUtils;
027import org.elasticsearch.common.settings.ImmutableSettings;
028import org.elasticsearch.common.settings.Settings;
029import org.elasticsearch.indices.IndexMissingException;
030import org.elasticsearch.node.Node;
031import org.testng.annotations.AfterMethod;
032import org.testng.annotations.BeforeMethod;
033
034import java.util.Map;
035import java.util.concurrent.atomic.AtomicLong;
036
037import static org.elasticsearch.common.collect.Maps.newHashMap;
038import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
039import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
040
041public abstract class AbstractRiverNodeTest extends AbstractRiverTest {
042
043    private static final ESLogger logger = Loggers.getLogger(AbstractRiverNodeTest.class);
044
045    public final String INDEX = "my_jdbc_river";
046
047    public final String TYPE = "my_jdbc_river";
048
049    private final static AtomicLong counter = new AtomicLong();
050
051    private Map<String, Node> nodes = newHashMap();
052
053    private Map<String, Client> clients = newHashMap();
054
055    private Settings defaultSettings() {
056        return ImmutableSettings
057            .settingsBuilder()
058            .put("cluster.name", "testing-jdbc-river-on-" + NetworkUtils.getLocalAddress().getHostName() + "-" + counter.incrementAndGet())
059            .build();
060    }
061
062    @BeforeMethod
063    public void createIndices() throws Exception {
064        startNode("1").client();
065        client("1").admin().indices().create(new CreateIndexRequest(INDEX)).actionGet();
066    }
067
068    @AfterMethod
069    public void deleteIndices() {
070        try {
071            // clear test index
072            client("1").admin().indices()
073                    .delete(new DeleteIndexRequest().indices(INDEX))
074                    .actionGet();
075        } catch (IndexMissingException e) {
076            logger.error(e.getMessage());
077        }
078        try {
079            // clear rivers
080            client("1").admin().indices()
081                    .delete(new DeleteIndexRequest().indices("_river"))
082                    .actionGet();
083        } catch (IndexMissingException e) {
084            logger.error(e.getMessage());
085        }
086        closeNode("1");
087        closeAllNodes();
088    }
089
090
091    public Node startNode(String id) {
092        return buildNode(id).start();
093    }
094
095    public Node buildNode(String id) {
096        return buildNode(id, defaultSettings());
097    }
098
099    public Node buildNode(String id, Settings settings) {
100        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
101        Settings finalSettings = settingsBuilder()
102                .loadFromClasspath(settingsSource)
103                .put(settings)
104                .put("name", id)
105                .build();
106
107        if (finalSettings.get("gateway.type") == null) {
108            // default to non gateway
109            finalSettings = settingsBuilder().put(finalSettings).put("gateway.type", "none").build();
110        }
111
112        Node node = nodeBuilder()
113                .settings(finalSettings)
114                .build();
115        nodes.put(id, node);
116        clients.put(id, node.client());
117        return node;
118    }
119
120    public void closeNode(String id) {
121        Client client = clients.remove(id);
122        if (client != null) {
123            client.close();
124        }
125        Node node = nodes.remove(id);
126        if (node != null) {
127            node.close();
128        }
129    }
130
131    public Client client(String id) {
132        return clients.get(id);
133    }
134
135    public void closeAllNodes() {
136        for (Client client : clients.values()) {
137            client.close();
138        }
139        clients.clear();
140        for (Node node : nodes.values()) {
141            node.close();
142        }
143        nodes.clear();
144    }
145}
