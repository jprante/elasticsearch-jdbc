050    @BeforeMethod
051    @Parameters({"starturl", "user", "password", "create"})
052    public void beforeMethod(String starturl, String user, String password, @Optional String resourceName)
053            throws Exception {
054        startNodes();
055
056        logger.info("nodes started");
057
058        waitForYellow("1");
059        try {
060            // create river index
061            client("1").admin().indices().create(new CreateIndexRequest("_river")).actionGet();
062            logger.info("river index created");
063        } catch (IndexAlreadyExistsException e) {
064            logger.warn(e.getMessage());
065        }
066        source = getRiverSource()
067                .setUrl(starturl)
068                .setUser(user)
069                .setPassword(password);
070        context = getRiverContext()
071                .setRiverSource(source)
072                .setRetries(1)
073                .setMaxRetryWait(TimeValue.timeValueSeconds(5))
074                .setLocale("en");
075        context.contextualize();
076        logger.info("create table {}", resourceName);
077        if (resourceName == null || "".equals(resourceName)) {
078            return;
079        }
080        Connection connection = source.getConnectionForWriting();
081        if (connection == null) {
082            throw new IOException("no connection");
083        }
084        sqlScript(connection, resourceName);
085        source.closeWriting();
086    }
087
088    @AfterMethod
089    @Parameters({"stopurl", "user", "password", "delete"})
090    public void afterMethod(String stopurl, String user, String password, @Optional String resourceName)
091            throws Exception {
092
093        logger.info("remove table {}", resourceName);
094        if (resourceName == null || "".equals(resourceName)) {
095            return;
096        }
097        // before dropping tables, open read connection must be closed to avoid hangs in mysql/postgresql
098        logger.debug("closing reads...");
099        source.closeReading();
100
101        logger.debug("connecting for close...");
102        Connection connection = source.getConnectionForWriting();
103        if (connection == null) {
104            throw new IOException("no connection");
105        }
106        logger.debug("cleaning...");
107        // clean up tables
108        sqlScript(connection, resourceName);
109        logger.debug("closing writes...");
110        source.closeWriting();
111
112        // some driver can drop database by a magic 'stop' URL
113        source = getRiverSource()
114                .setUrl(stopurl)
115                .setUser(user)
116                .setPassword(password);
117        try {
118            logger.info("connecting to stop URL...");
119            // activate stop URL
120            source.getConnectionForWriting();
121        } catch (Exception e) {
122            // exception is expected, ignore
123        }
124        // close open write connection
125        source.closeWriting();
126        logger.info("stopped");
127
128        // delete test index
129        try {
130            client("1").admin().indices().delete(new DeleteIndexRequest(index)).actionGet();
131            logger.info("index {} deleted", index);
132        } catch (IndexMissingException e) {
133            logger.warn(e.getMessage());
134        }
135        try {
136            client("1").admin().indices().deleteMapping(new DeleteMappingRequest()
137                    .indices(new String[]{"_river"}).types("my_jdbc_river")).actionGet();
138            logger.info("river my_jdbc_river deleted");
139        } catch (Exception e) {
140            logger.warn(e.getMessage());
141        }
142        stopNodes();
143    }

145    protected void createRiver(String resource) {
146        try {
147            waitForYellow("1");
148            Map<String, Object> map = XContentHelper.convertToMap(Streams.copyToByteArray(getClass().getResourceAsStream(resource)), false).v2();
149            XContentBuilder builder = jsonBuilder()
150                    .startObject()
151                    .field("type", "jdbc")
152                    .field("jdbc", map.get("jdbc"))
153                    .endObject();
154            logger.debug("river spec = {}", builder.string());
155            IndexRequest indexRequest = Requests.indexRequest("_river").type("my_jdbc_river").id("_meta")
156                    .source(builder.string());
157            client("1").index(indexRequest).actionGet();
158            client("1").admin().indices().prepareRefresh("_river").execute().actionGet();
159            logger.info("river is created");
160            RiverHelper.waitForRiverEnabled(client("1").admin().cluster(), "my_jdbc_river", "jdbc", 15);
161            logger.info("river is enabled");
162            RiverHelper.waitForActiveRiver(client("1").admin().cluster(), "my_jdbc_river", "jdbc",  15);
163            logger.info("river is active");
164        } catch (InterruptedException e) {
165            logger.error(e.getMessage());
166        } catch (IOException e) {
167            logger.error(e.getMessage());
168        }
169    }

190    protected void createRandomProducts(String sql, int size)
191            throws SQLException {
192        Connection connection = source.getConnectionForWriting();
193        for (int i = 0; i < size; i++) {
194            long amount = Math.round(Math.random() * 1000);
195            double price = (Math.random() * 10000) / 100.00;
196            add(connection, sql, UUID.randomUUID().toString().substring(0, 32), amount, price);
197        }
198        if (!connection.getAutoCommit()) {
199            connection.commit();
200        }
201        source.closeWriting();
202    }

204    private void add(Connection connection, String sql, final String name, final long amount, final double price)
205            throws SQLException {
206        PreparedStatement stmt = connection.prepareStatement(sql);
207        List<Object> params = new ArrayList<Object>() {{
208            add(name);
209            add(amount);
210            add(price);
211        }};
212        source.bind(stmt, params);
213        stmt.execute();
214    }

245    private void sqlScript(Connection connection, String resourceName) throws Exception {
246        InputStream in = getClass().getResourceAsStream(resourceName);
247        BufferedReader br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
248        String sql;
249        while ((sql = br.readLine()) != null) {
250
251            try {
252                logger.trace("executing {}", sql);
253                Statement p = connection.createStatement();
254                p.execute(sql);
255                p.close();
256            } catch (SQLException e) {
257                // ignore
258                logger.error(sql + " failed. Reason: " + e.getMessage());
259            } finally {
260                connection.commit();
261            }
262        }
263        br.close();
264    }
