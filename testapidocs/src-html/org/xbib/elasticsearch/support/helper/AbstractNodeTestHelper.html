001package org.xbib.elasticsearch.support.helper;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
004import org.elasticsearch.action.admin.cluster.node.info.NodesInfoRequest;
005import org.elasticsearch.action.admin.cluster.node.info.NodesInfoResponse;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.common.network.NetworkUtils;
008import org.elasticsearch.common.settings.ImmutableSettings;
009import org.elasticsearch.common.settings.Settings;
010import org.elasticsearch.common.transport.InetSocketTransportAddress;
011import org.elasticsearch.common.transport.LocalTransportAddress;
012import org.elasticsearch.node.Node;
013import org.testng.Assert;
014import org.testng.annotations.AfterMethod;
015import org.testng.annotations.BeforeMethod;
016import org.xbib.logging.Logger;
017import org.xbib.logging.LoggerFactory;
018
019import java.io.IOException;
020import java.util.Map;
021import java.util.concurrent.atomic.AtomicInteger;
022
023import static org.elasticsearch.common.collect.Maps.newHashMap;
024import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
025import static org.elasticsearch.node.NodeBuilder.nodeBuilder;
026
027public abstract class AbstractNodeTestHelper extends Assert {
028
029    protected final static Logger logger = LoggerFactory.getLogger("test");
030
031    private final AtomicInteger clusterCount = new AtomicInteger();
032
033    // note, this must be same name as in json river specs
034    protected final String INDEX = "my_jdbc_river_index";
035
036    public final String TYPE = "my_jdbc_river_type";
037
038    private Map<String, Node> nodes = newHashMap();
039
040    private Map<String, Client> clients = newHashMap();
041
042    protected String getClusterName() {
043        return "test-jdbc-cluster-" + NetworkUtils.getLocalAddress().getHostName() + "-" + clusterCount.incrementAndGet();
044    }
045
046    protected Settings getNodeSettings() {
047        return ImmutableSettings
048                .settingsBuilder()
049                .put("cluster.name", getClusterName())
050                .put("index.number_of_shards", 1)
051                .put("index.number_of_replica", 0)
052                .put("cluster.routing.schedule", "50ms")
053                .put("gateway.type", "none")
054                .put("index.store.type", "ram")
055                .put("http.enabled", false)
056                .put("discovery.zen.multicast.enabled", false)
057                .build();
058    }
059
060    @BeforeMethod
061    public void startNodes() throws Exception {
062        startNode("1");
063        // find node address
064        NodesInfoRequest nodesInfoRequest = new NodesInfoRequest().transport(true);
065        NodesInfoResponse response = client("1").admin().cluster().nodesInfo(nodesInfoRequest).actionGet();
066        Object obj = response.iterator().next().getTransport().getAddress().publishAddress();
067        if (obj instanceof InetSocketTransportAddress) {
068            InetSocketTransportAddress address = (InetSocketTransportAddress) obj;
069            // ... do we need our transport address?
070        }
071        if (obj instanceof LocalTransportAddress) {
072            LocalTransportAddress address = (LocalTransportAddress) obj;
073            // .... do we need local transport?
074        }
075    }
076
077    public Node startNode(String id) {
078        return buildNode(id).start();
079    }
080
081    public Node buildNode(String id) {
082        return buildNode(id, getNodeSettings());
083    }
084
085    public Node buildNode(String id, Settings settings) {
086        String settingsSource = getClass().getName().replace('.', '/') + ".yml";
087        Settings finalSettings = settingsBuilder()
088                .loadFromClasspath(settingsSource)
089                .put(getNodeSettings())
090                .put(settings)
091                .put("name", id)
092                .build();
093        Node node = nodeBuilder().local(true).settings(finalSettings).build();
094        Client client = node.client();
095        nodes.put(id, node);
096        clients.put(id, client);
097        return node;
098    }
099
100    public void waitForYellow(String id) throws IOException {
101        logger.info("wait for healthy cluster...");
102        ClusterHealthResponse clusterHealthResponse = client(id).admin().cluster().prepareHealth().setWaitForYellowStatus().execute().actionGet();
103        if (clusterHealthResponse.isTimedOut()) {
104            throw new IOException("error, cluster health is " + clusterHealthResponse.getStatus().name());
105        }
106        logger.info("cluster health is {}", clusterHealthResponse.getStatus().name());
107    }
108
109    public void assertHits(String id, int expectedHits) {
110        client(id).admin().indices().prepareRefresh(INDEX).execute().actionGet();
111        logger.debug("{} refreshed", INDEX);
112        long hitsFound = client(id).prepareSearch(INDEX).setTypes(TYPE).execute().actionGet().getHits().getTotalHits();
113        logger.debug("{}/{} = {} hits", INDEX, TYPE, hitsFound);
114        assertEquals(hitsFound, expectedHits);
115    }
116
117    public Client client(String id) {
118        Client client = clients.get(id);
119        if (client == null) {
120            client = nodes.get(id).client();
121            clients.put(id, client);
122        }
123        return client;
124    }
125
126    @AfterMethod
127    public void stopNodes() {
128        for (Client client : clients.values()) {
129            client.close();
130        }
131        clients.clear();
132        for (Node node : nodes.values()) {
133            node.stop();
134            node.close();
135        }
136        nodes.clear();
137    }
138
139}
