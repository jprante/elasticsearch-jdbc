021import java.io.IOException;
022import org.elasticsearch.common.xcontent.XContentBuilder;
023import org.testng.annotations.Test;
024
025public class MergerTest {
026
027    @Test
028    public void test() throws Exception {
029        
030        String columns[] = new String[] {"_id","person.salary","person.name", "person.position.name", "person.position.since"};
031        String row1[] = new String[]{"1","$1000","Joe Doe", "Worker", "2012-06-12"};
032        String row2[] = new String[]{"2","$2000","Bill Smith", "Boss", "2012-06-13"};
033        
034        Action listener =  new DefaultAction() {
035
036            @Override
037            public void index(String index, String type, String id, long version, XContentBuilder builder) throws IOException {
038               System.err.println("index="+index + " type="+type + " id="+id+ " builder="+builder.string());
039            }
040            
041        };
042        Merger merger = new Merger(listener, 1L);
043        merger.row(columns, row1);
044        merger.row(columns, row2);
045        merger.close();
046    }
047}
