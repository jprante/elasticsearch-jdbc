032    @Test
033    public void testStarQuery() {
034        try {
035            String driverClassName = "com.mysql.jdbc.Driver";
036            String url = "jdbc:mysql://localhost:3306/test";
037            String username = "";
038            String password = "";
039            String sql = "select * from orders";
040            List<Object> params = new ArrayList();
041            int fetchsize = 0;
042            Action listener = new DefaultAction() {
043
044                @Override
045                public void index(String index, String type, String id, long version, XContentBuilder builder) throws IOException {
046                    System.err.println("index=" + index + " type=" + type + " id=" + id + " builder=" + builder.string());
047                }
048            };
049            SQLService service = new SQLService();
050            Connection connection = service.getConnection(driverClassName, url, username, password, true);
051            PreparedStatement statement = service.prepareStatement(connection, sql);
052            service.bind(statement, params);
053            ResultSet results = service.execute(statement, fetchsize);
054            Merger merger = new Merger(listener, 1L);
055            long rows = 0L;
056            while (service.nextRow(results, merger)) {
057                rows++;
058            }
059            merger.close();
060            System.err.println("rows = " + rows);
061            service.close(results);
062            service.close(statement);
063            service.close(connection);
064        } catch (Exception e) {
065            e.printStackTrace();
066        }
067    }
