050public class JDBCRiver extends AbstractRiverComponent implements River {
051
052    private final Client client;
053    private final String riverIndexName;
054    private final String indexName;
055    private final String typeName;
056    private final SQLService service;
057    private final BulkOperation operation;
058    private final int bulkSize;
059    private final int maxBulkRequests;
060    private final TimeValue bulkTimeout;
061    private final TimeValue poll;
062    private final TimeValue interval;
063    private final String url;
064    private final String driver;
065    private final String user;
066    private final String password;
067    private final String sql;
068    private final int fetchsize;
069    private final List<Object> params;
070    private final boolean rivertable;
071    private final boolean versioning;
072    private final String rounding;
073    private final int scale;
074    private volatile Thread thread;
075    private volatile boolean closed;
076    private Date creationDate;

078    @Inject
079    public JDBCRiver(RiverName riverName, RiverSettings settings,
080                     @RiverIndexName String riverIndexName, Client client) {
081        super(riverName, settings);
082        this.riverIndexName = riverIndexName;
083        this.client = client;
084        if (settings.settings().containsKey("jdbc")) {
085            Map<String, Object> jdbcSettings = (Map<String, Object>) settings.settings().get("jdbc");
086            poll = XContentMapValues.nodeTimeValue(jdbcSettings.get("poll"), TimeValue.timeValueMinutes(60));
087            url = XContentMapValues.nodeStringValue(jdbcSettings.get("url"), null);
088            driver = XContentMapValues.nodeStringValue(jdbcSettings.get("driver"), null);
089            user = XContentMapValues.nodeStringValue(jdbcSettings.get("user"), null);
090            password = XContentMapValues.nodeStringValue(jdbcSettings.get("password"), null);
091            sql = XContentMapValues.nodeStringValue(jdbcSettings.get("sql"), null);
092            fetchsize = XContentMapValues.nodeIntegerValue(jdbcSettings.get("fetchsize"), 0);
093            params = XContentMapValues.extractRawValues("params", jdbcSettings);
094            rivertable = XContentMapValues.nodeBooleanValue(jdbcSettings.get("rivertable"), false);
095            interval = XContentMapValues.nodeTimeValue(jdbcSettings.get("interval"), TimeValue.timeValueMinutes(60));
096            versioning = XContentMapValues.nodeBooleanValue(jdbcSettings.get("versioning"), true);
097            rounding = XContentMapValues.nodeStringValue(jdbcSettings.get("rounding"), null);
098            scale = XContentMapValues.nodeIntegerValue(jdbcSettings.get("scale"), 0);
099        } else {
100            poll = TimeValue.timeValueMinutes(60);
101            url = null;
102            driver = null;
103            user = null;
104            password = null;
105            sql = null;
106            fetchsize = 0;
107            params = null;
108            rivertable = false;
109            interval = TimeValue.timeValueMinutes(60);
110            versioning = true;
111            rounding = null;
112            scale = 0;
113        }
114        if (settings.settings().containsKey("index")) {
115            Map<String, Object> indexSettings = (Map<String, Object>) settings.settings().get("index");
116            indexName = XContentMapValues.nodeStringValue(indexSettings.get("index"), "jdbc");
117            typeName = XContentMapValues.nodeStringValue(indexSettings.get("type"), "jdbc");
118            bulkSize = XContentMapValues.nodeIntegerValue(indexSettings.get("bulk_size"), 100);
119            maxBulkRequests = XContentMapValues.nodeIntegerValue(indexSettings.get("max_bulk_requests"), 30);
120            if (indexSettings.containsKey("bulk_timeout")) {
121                bulkTimeout = TimeValue.parseTimeValue(XContentMapValues.nodeStringValue(indexSettings.get("bulk_timeout"), "60s"), TimeValue.timeValueMillis(60000));
122            } else {
123                bulkTimeout = TimeValue.timeValueMillis(60000);
124            }
125        } else {
126            indexName = "jdbc";
127            typeName = "jdbc";
128            bulkSize = 100;
129            maxBulkRequests = 30;
130            bulkTimeout = TimeValue.timeValueMillis(60000);
131        }
132        service = new SQLService(logger).setPrecision(scale).setRounding(rounding);
133        operation = new BulkOperation(client, logger).setIndex(indexName).setType(typeName).setVersioning(versioning)
134                .setBulkSize(bulkSize).setMaxActiveRequests(maxBulkRequests)
135                .setMillisBeforeContinue(bulkTimeout.millis()).setAcknowledge(riverName.getName(), rivertable ? service : null);
136    }
137
138    @Override
139    public void start() {
140        logger.info("starting JDBC connector: URL [{}], driver [{}], sql [{}], river table [{}], indexing to [{}]/[{}], poll [{}]",
141                url, driver, sql, rivertable, indexName, typeName, poll);
142        try {
143            client.admin().indices().prepareCreate(indexName).execute().actionGet();
144            creationDate = new Date();
145        } catch (Exception e) {
146            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
147                creationDate = null;
148                // that's fine
149            } else if (ExceptionsHelper.unwrapCause(e) instanceof ClusterBlockException) {
150                // ok, not recovered yet..., lets start indexing and hope we recover by the first bulk
151            } else {
152                logger.warn("failed to create index [{}], disabling river...", e, indexName);
153                return;
154            }
155        }
156        thread = EsExecutors.daemonThreadFactory(settings.globalSettings(), "JDBC connector").newThread(rivertable ? new JDBCRiverTableConnector() : new JDBCConnector());
157        thread.start();
158    }
159
160    @Override
161    public void close() {
162        if (closed) {
163            return;
164        }
165        logger.info("closing JDBC river");
166        thread.interrupt();
167        closed = true;
168    }

170    private class JDBCConnector implements Runnable {
171
172        @Override
173        public void run() {
174            while (true) {
175                try {
176                    Number version;
177                    String digest;
178                    // read state from _custom
179                    client.admin().indices().prepareRefresh(riverIndexName).execute().actionGet();
180                    GetResponse get = client.prepareGet(riverIndexName, riverName().name(), "_custom").execute().actionGet();
181                    if (creationDate != null || !get.exists()) {
182                        version = 1L;
183                        digest = null;
184                    } else {
185                        Map<String, Object> jdbcState = (Map<String, Object>) get.sourceAsMap().get("jdbc");
186                        if (jdbcState != null) {
187                            version = (Number) jdbcState.get("version");
188                            version = version.longValue() + 1; // increase to next version
189                            digest = (String) jdbcState.get("digest");
190                        } else {
191                            throw new IOException("can't retrieve previously persisted state from " + riverIndexName + "/" + riverName().name());
192                        }
193                    }
194                    Connection connection = service.getConnection(driver, url, user, password, true);
195                    PreparedStatement statement = service.prepareStatement(connection, sql);
196                    service.bind(statement, params);
197                    ResultSet results = service.execute(statement, fetchsize);
198                    Merger merger = new Merger(operation, version.longValue());
199                    long rows = 0L;
200                    while (service.nextRow(results, merger)) {
201                        rows++;
202                    }
203                    merger.close();
204                    service.close(results);
205                    service.close(statement);
206                    service.close(connection);
207                    logger.info("got " + rows + " rows for version " + version.longValue() + ", digest = " + merger.getDigest());
208                    // this flush is required before house keeping starts
209                    operation.flush();
210                    // save state to _custom
211                    XContentBuilder builder = jsonBuilder();
212                    builder.startObject().startObject("jdbc");
213                    if (creationDate != null) {
214                        builder.field("created", creationDate);
215                    }
216                    builder.field("version", version.longValue());
217                    builder.field("digest", merger.getDigest());
218                    builder.endObject().endObject();
219                    client.prepareBulk().add(indexRequest(riverIndexName).type(riverName.name()).id("_custom").source(builder)).execute().actionGet();
220                    // house keeping if data has changed
221                    if (digest != null && !merger.getDigest().equals(digest)) {
222                        housekeeper(version.longValue());
223                        // perform outstanding housekeeper bulk requests
224                        operation.flush();
225                    }
226                    delay("next run");
227                } catch (Exception e) {
228                    logger.error(e.getMessage(), e, (Object) null);
229                    closed = true;
230                }
231                if (closed) {
232                    return;
233                }
234            }
235        }
236
237        private void housekeeper(long version) throws IOException {
238            logger.info("housekeeping for version " + version);
239            client.admin().indices().prepareRefresh(indexName).execute().actionGet();
240            SearchResponse response = client.prepareSearch().setIndices(indexName).setTypes(typeName).setSearchType(SearchType.SCAN).setScroll(TimeValue.timeValueMinutes(10)).setSize(bulkSize).setVersion(true).setQuery(matchAllQuery()).execute().actionGet();
241            if (response.timedOut()) {
242                logger.error("housekeeper scan query timeout");
243                return;
244            }
245            if (response.failedShards() > 0) {
246                logger.error("housekeeper failed shards in scan response: {0}", response.failedShards());
247                return;
248            }
249            String scrollId = response.getScrollId();
250            if (scrollId == null) {
251                logger.error("housekeeper failed, no scroll ID");
252                return;
253            }
254            boolean done = false;
255            // scroll
256            long deleted = 0L;
257            long t0 = System.currentTimeMillis();
258            do {
259                response = client.prepareSearchScroll(response.getScrollId()).setScroll(TimeValue.timeValueMinutes(10)).execute().actionGet();
260                if (response.timedOut()) {
261                    logger.error("housekeeper scroll query timeout");
262                    done = true;
263                } else if (response.failedShards() > 0) {
264                    logger.error("housekeeper failed shards in scroll response: {}", response.failedShards());
265                    done = true;
266                } else {
267                    // terminate scrolling?
268                    if (response.hits() == null) {
269                        done = true;
270                    } else {
271                        for (SearchHit hit : response.getHits().getHits()) {
272                            // delete all documents with lower version
273                            if (hit.getVersion() < version) {
274                                operation.delete(hit.getIndex(), hit.getType(), hit.getId());
275                                deleted++;
276                            }
277                        }
278                        scrollId = response.getScrollId();
279                    }
280                }
281                if (scrollId != null) {
282                    done = true;
283                }
284            } while (!done);
285            long t1 = System.currentTimeMillis();
286            logger.info("housekeeper ready, {} documents deleted, took {} ms", deleted, t1 - t0);
287        }
288    }

290    private class JDBCRiverTableConnector implements Runnable {
291
292        private String[] optypes = new String[]{"create", "index", "delete"};
293
294        @Override
295        public void run() {
296            while (true) {
297                for (String optype : optypes) {
298                    try {
299                        Connection connection = service.getConnection(driver, url, user, password, false);
300                        PreparedStatement statement = service.prepareRiverTableStatement(connection, riverName.getName(), optype, interval.millis());
301                        ResultSet results = service.execute(statement, fetchsize);
302                        Merger merger = new Merger(operation);
303                        long rows = 0L;
304                        while (service.nextRiverTableRow(results, merger)) {
305                            rows++;
306                        }
307                        merger.close();
308                        service.close(results);
309                        service.close(statement);
310                        logger.info(optype + ": got " + rows + " rows");
311                        // this flush is required before next run
312                        operation.flush();
313                        service.close(connection);
314                    } catch (Exception e) {
315                        logger.error(e.getMessage(), e, (Object) null);
316                        closed = true;
317                    }
318                    if (closed) {
319                        return;
320                    }
321                }
322                delay("next run");
323            }
324        }
325    }
326
327
328    private void delay(String reason) {
329        if (poll.millis() > 0L) {
330            logger.info("{}, waiting {}, URL [{}] driver [{}] sql [{}] river table [{}]",
331                    reason, poll, url, driver, sql, rivertable);
332            try {
333                Thread.sleep(poll.millis());
334            } catch (InterruptedException e1) {
335            }
336        }
337    }
