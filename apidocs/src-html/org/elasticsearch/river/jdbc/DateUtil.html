019package org.elasticsearch.river.jdbc;
020
021import java.text.ParseException;
022import java.text.SimpleDateFormat;
023import java.util.Calendar;
024import java.util.Date;
025import java.util.TimeZone;
026
027public class DateUtil {
028
029    private static final String ISO_FORMAT_SECONDS = "yyyy-MM-dd'T'HH:mm:ss'Z'";
030    private static final SimpleDateFormat isoFormat = new SimpleDateFormat();
031    private static final TimeZone tz = TimeZone.getTimeZone("GMT");
032    private static final Calendar cal = Calendar.getInstance();
033
034    static {
035        isoFormat.applyPattern(ISO_FORMAT_SECONDS);
036        isoFormat.setTimeZone(tz);
037        isoFormat.setLenient(true);
038    }
039    
040    public static String formatNow() {
041        return formatDateISO(new Date());
042    }
043
044    public static String formatDateISO(long millis) {
045        Date d = new Date();
046        d.setTime(millis);
047        return formatDateISO(d);
048    }
049    
050    public synchronized static String formatDateISO(Date date) {
051        if (date == null) {
052            return null;
053        }
054        isoFormat.applyPattern(ISO_FORMAT_SECONDS);
055        return isoFormat.format(date);
056    }
057
058    public synchronized static Date parseDateISO(String value) {
059        if (value == null) {
060            return null;
061        }
062        isoFormat.applyPattern(ISO_FORMAT_SECONDS);
063        try {
064            return isoFormat.parse(value);
065        } catch (ParseException pe) {
066            // skip
067        }
068        isoFormat.applyPattern("yyyy-MM-dd");
069        try {
070            return isoFormat.parse(value);
071        } catch (ParseException pe) {
072            return null;
073        }
074    }
075
076    public synchronized static int getYear(Date date) {
077        cal.setTime(date);
078        return cal.get(Calendar.YEAR);
079    }
080
081    public synchronized static Date midnight() {
082        return DateUtil.midnight(new Date());
083    }
084
085    public synchronized static Date midnight(Date date) {
086        cal.setTime(date);
087        cal.set(Calendar.HOUR_OF_DAY, 0);
088        cal.set(Calendar.MINUTE, 0);
089        cal.set(Calendar.SECOND, 0);
090        cal.set(Calendar.MILLISECOND, 0);
091        return cal.getTime();
092    }
093
094    public synchronized static Date yesterday() {
095        return DateUtil.yesterday(new Date());
096    }
097
098    public synchronized static Date yesterday(Date date) {
099        return days(date, -1);
100    }
101
102    public synchronized static Date tomorrow() {
103        return DateUtil.tomorrow(new Date());
104    }
105
106    public synchronized static Date tomorrow(Date date) {
107        return days(date, 1);
108    }
109
110    public synchronized static Date years(int years) {
111        return DateUtil.years(new Date(), years);
112    }
113
114    public synchronized static Date years(Date date, int years) {
115        cal.setTime(date);
116        cal.add(Calendar.YEAR, years);
117        return cal.getTime();
118    }
119    
120    public synchronized static Date months(int months) {
121        return DateUtil.months(new Date(), months);
122    }
123
124    public synchronized static Date months(Date date, int months) {
125        cal.setTime(date);
126        cal.add(Calendar.MONTH, months);
127        return cal.getTime();
128    }
129    
130    public static Date weeks(int weeks) {
131        return DateUtil.weeks(new Date(), weeks);
132    }
133
134    public synchronized static Date weeks(Date date, int weeks) {
135        cal.setTime(date);
136        cal.add(Calendar.WEEK_OF_YEAR, weeks);
137        return cal.getTime();
138    }
139
140    public static Date days(int days) {
141        return DateUtil.days(new Date(), days);
142    }
143
144    public synchronized static Date days(Date date, int days) {
145        cal.setTime(date);
146        cal.add(Calendar.DAY_OF_YEAR, days);
147        return cal.getTime();
148    }
149
150    public static Date hours(int hours) {
151        return DateUtil.hours(new Date(), hours);
152    }
153
154    public synchronized static Date hours(Date date, int hours) {
155        cal.setTime(date);
156        cal.add(Calendar.HOUR_OF_DAY, hours);
157        return cal.getTime();
158    }
159    
160    public static Date minutes(int minutes) {
161        return DateUtil.minutes(new Date(), minutes);
162    }
163
164    public synchronized static Date minutes(Date date, int minutes) {
165        cal.setTime(date);
166        cal.add(Calendar.MINUTE, minutes);
167        return cal.getTime();
168    }
169    
170    public static Date seconds(int seconds) {
171        return DateUtil.seconds(new Date(), seconds);
172    }
173
174    public synchronized static Date seconds(Date date, int seconds) {
175        cal.setTime(date);
176        cal.add(Calendar.MINUTE, seconds);
177        return cal.getTime();
178    }
179
180}
