019package org.elasticsearch.river.jdbc;
020
021import java.io.IOException;
022import org.elasticsearch.common.xcontent.XContentBuilder;
023
024public interface Action {
025
026    /**
027     * Create action.  Indicating that an object has been built and is ready to be indexed.
028     * The source of the document is held by the XContentBuilder.
029     * The document is only indexed if no document with that index/type/id exists.
030     * 
031     * @param index the index
032     * @param type the type
033     * @param id the id
034     * @param version the version
035     * @param builder the builder
036     * @throws IOException 
037     */
038    void create(String index, String type, String id, long version, XContentBuilder builder) 
039            throws IOException;
040
041    /**
042     * Index action. Indicating that an object has been built and is ready to be indexed.
043     * The source of the document is held by the XContentBuilder.
044     * @param index the index
045     * @param type the type 
046     * @param id the id 
047     * @param version the version
048     * @param builder the XContentBuilder
049     * @throws IOException 
050     */
051    void index(String index, String type, String id, long version, XContentBuilder builder) 
052            throws IOException;
053
054    /**
055     * Delete action. Indicating that an object should be deleted from the index.
056     * @param index
057     * @param type
058     * @param id
059     * @throws IOException 
060     */
061    void delete(String index, String type, String id)
062            throws IOException;
063
064}
