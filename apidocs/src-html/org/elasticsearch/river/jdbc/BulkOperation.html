019package org.elasticsearch.river.jdbc;

021import java.io.IOException;
022import java.util.concurrent.atomic.AtomicInteger;
023import org.elasticsearch.action.ActionListener;
024import org.elasticsearch.action.bulk.BulkRequestBuilder;
025import org.elasticsearch.action.bulk.BulkResponse;
026import org.elasticsearch.action.index.IndexRequest;
027import org.elasticsearch.client.Client;

033/**
034 * Send bulk data to Elasticsearch
035 *
036 * @author <a href="mailto:joergprante@gmail.com">J&ouml;rg Prante</a>
037 */
038public class BulkOperation implements Action {
039
040    private String index;
041    private String type;
042    private String id;
043    private Client client;
044    private ESLogger logger;
045    private int bulkSize = 100;
046    private int maxActiveRequests = 30;
047    private long millisBeforeContinue = 60000L;
048    private int totalTimeouts;
049    private static final int MAX_TOTAL_TIMEOUTS = 10;
050    private static final AtomicInteger onGoingBulks = new AtomicInteger(0);
051    private static final AtomicInteger counter = new AtomicInteger(0);
052    private ThreadLocal<BulkRequestBuilder> currentBulk = new ThreadLocal();
053    private String riverName;
054    private BulkAcknowledge ack;
055    private boolean versioning;
056
057    public BulkOperation(Client client, ESLogger logger) {
058        this.client = client;
059        this.logger = logger;
060        this.id = null;
061        this.totalTimeouts = 0;
062    }

118    @Override
119    public void create(String index, String type, String id, long version, XContentBuilder builder) {
120        if (!isNullOrEmpty(index)) {
121            setIndex(index);
122        }
123        if (!isNullOrEmpty(type)) {
124            setType(type);
125        }
126        if (!isNullOrEmpty(id)) {
127           setId(id);
128        }
129        if (id == null) {
130            return; // skip if ID is null
131        }
132        if (currentBulk.get() == null) {
133            currentBulk.set(client.prepareBulk());
134        }
135        IndexRequest request = Requests.indexRequest(getIndex()).type(getType()).id(getId()).create(true).source(builder);
136        if (versioning) {
137            request.versionType(VersionType.EXTERNAL).version(version);
138        }
139        currentBulk.get().add(request);
140        if (currentBulk.get().numberOfActions() >= bulkSize) {
141            processBulk();
142        }
143    }    
144    
145    @Override
146    public void index(String index, String type, String id, long version, XContentBuilder builder) {
147        if (!isNullOrEmpty(index)) {
148            setIndex(index);
149        }
150        if (!isNullOrEmpty(type)) {
151            setType(type);
152        }
153        if (!isNullOrEmpty(id)) {
154           setId(id);
155        }
156        if (id == null) {
157            return; // skip if ID is null
158        }
159        if (currentBulk.get() == null) {
160            currentBulk.set(client.prepareBulk());
161        }
162        IndexRequest request = Requests.indexRequest(getIndex()).type(getType()).id(getId()).source(builder);
163        if (versioning) {
164            request.versionType(VersionType.EXTERNAL).version(version);
165        }
166        currentBulk.get().add(request);
167        if (currentBulk.get().numberOfActions() >= bulkSize) {
168            processBulk();
169        }
170    }
171    
172    @Override
173    public void delete(String index, String type, String id) {
174        if (!isNullOrEmpty(index)) {
175            setIndex(index);
176        }
177        if (!isNullOrEmpty(type)) {
178            setType(type);
179        }
180        if (!isNullOrEmpty(id)) {
181           setId(id);
182        }
183        if (id == null) {
184            return; // skip
185        }
186        if (currentBulk.get() == null) {
187            currentBulk.set(client.prepareBulk());
188        }
189        currentBulk.get().add(Requests.deleteRequest(getIndex()).type(getType()).id(getId()));
190        if (currentBulk.get().numberOfActions() >= bulkSize) {
191            processBulk();
192        }        
193    }
194    
195    public void flush() throws IOException {
196        if (totalTimeouts > MAX_TOTAL_TIMEOUTS) {
197            // waiting some minutes is much too long, do not wait any longer            
198            throw new IOException("total flush() timeouts exceeded limit of + " + MAX_TOTAL_TIMEOUTS + ", aborting");
199        }
200        if (currentBulk.get().numberOfActions() > 0) {
201            processBulk();
202        }
203        // wait for all outstanding bulk requests
204        while (onGoingBulks.intValue() > 0) {
205            logger.info("waiting for {} active bulk requests", onGoingBulks);
206            synchronized (onGoingBulks) {
207                try {
208                    onGoingBulks.wait(millisBeforeContinue);
209                } catch (InterruptedException e) {
210                    logger.warn("timeout while waiting, continuing after {} ms", millisBeforeContinue);
211                    totalTimeouts++;
212                }
213            }
214        }
215    }
216
217    private void processBulk() {
218        while (onGoingBulks.intValue() >= maxActiveRequests) {
219            logger.info("waiting for {} active bulk requests", onGoingBulks);
220            synchronized (onGoingBulks) {
221                try {
222                    onGoingBulks.wait(millisBeforeContinue);
223                } catch (InterruptedException e) {
224                    logger.warn("timeout while waiting, continuing after {} ms", millisBeforeContinue);
225                    totalTimeouts++;
226                }
227            }
228        }
229        int currentOnGoingBulks = onGoingBulks.incrementAndGet();
230        final int numberOfActions = currentBulk.get().numberOfActions();
231        logger.info("submitting new bulk request ({} docs, {} requests currently active)", new Object[]{numberOfActions, currentOnGoingBulks});
232        try {
233            currentBulk.get().execute(new ActionListener<BulkResponse>() {
234
235                @Override
236                public void onResponse(BulkResponse bulkResponse) {
237                    if (ack != null) try {
238                        ack.acknowledge(riverName, bulkResponse.items());
239                    } catch (IOException ex) {
240                        logger.error("bulk acknowledge failed", ex);
241                    }
242                    if (bulkResponse.hasFailures()) {
243                        logger.error("bulk request has failures: {}", bulkResponse.buildFailureMessage());
244                    } else {
245                        final int totalActions = counter.addAndGet(numberOfActions);
246                        logger.info("bulk request success ({} millis, {} docs, total of {} docs)", new Object[]{bulkResponse.tookInMillis(), numberOfActions, totalActions});
247                    }
248                    onGoingBulks.decrementAndGet();
249                    synchronized (onGoingBulks) {
250                        onGoingBulks.notifyAll();
251                    }
252                }
253
254                @Override
255                public void onFailure(Throwable e) {
256                    logger.error("bulk request failed", e);
257                }
258            });
259        } catch (Exception e) {
260            logger.error("unhandled exception, failed to execute bulk request", e);
261        } finally {
262            currentBulk.set(client.prepareBulk());
263        }
264    }
265
266    private boolean isNullOrEmpty(String s) {
267        return s == null || s.length() == 0;
268    }
