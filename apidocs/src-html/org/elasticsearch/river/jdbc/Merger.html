019package org.elasticsearch.river.jdbc;
020
021import java.io.IOException;
022import java.security.MessageDigest;
023import java.security.NoSuchAlgorithmException;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027import org.elasticsearch.common.Base64;
028import org.elasticsearch.common.xcontent.XContentBuilder;
029import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
030
031/**
032 * The Merger class consumes SQL rows and produces Elasticsearch JSON sources.
033 */
034public class Merger implements RowListener {
035
036    private final static String OPTYPE_COLUMN = "_optype";
037    private final static String INDEX_COLUMN = "_index";
038    private final static String TYPE_COLUMN = "_type";
039    private final static String ID_COLUMN = "_id";
040    private final static String OP_CREATE = "create";
041    private final static String OP_INDEX = "index";
042    private final static String OP_DELETE = "delete";
043    
044    private final static String DIGEST_ALGORITHM = "SHA-256";
045    private final static String DIGEST_ENCODING = "UTF-8";
046    
047    private char delimiter;
048    private XContentBuilder builder;
049    private Action listener;
050    private Map<String, Object> map;
051    private String prevoptype;
052    private String previndex;
053    private String prevtype;
054    private String previd;
055    private String optype;
056    private String index;
057    private String type;
058    private String id;
059    private final long version;
060    private final MessageDigest digest;
061    private String digestString;
062    private boolean closed;
063
064    /**
065     * Constructor for a new Merger object
066     * 
067     * @param action
068     * @throws IOException
069     * @throws NoSuchAlgorithmException 
070     */
071    public Merger(Action action) throws IOException, NoSuchAlgorithmException {
072        this('.', action, -1L);
073    }
074    
075    /**
076     * Constructor for a new Merger object
077     *
078     * @param action the action
079     * @param version the version
080     * @throws IOException
081     */
082    public Merger(Action action, long version) throws IOException, NoSuchAlgorithmException {
083        this('.', action, version);
084    }
085
086    /**
087     * Constructor for a new Merger object
088     *
089     * @param delimiter the delimiter
090     * @param action the action
091     * @param version the version
092     * @throws IOException
093     */
094    public Merger(char delimiter, Action action, long version) throws IOException, NoSuchAlgorithmException {
095        this.delimiter = delimiter;
096        this.builder = jsonBuilder();
097        this.listener = action;
098        this.map = new HashMap();
099        this.version = version;
100        this.digest = MessageDigest.getInstance(DIGEST_ALGORITHM);
101        this.closed = false;
102    }
103
104    /**
105     * Merge a row given by string arrays
106     *
107     * @param columns the column names for the row elements
108     * @param row the row elements
109     * @throws IOException
110     */
111    public void row(String[] columns, Object[] row) throws IOException {
112        boolean changed = false;
113        for (int i = 0; i < columns.length; i++) {
114            if (OPTYPE_COLUMN.equals(columns[i])) {
115                if (optype != null && !optype.equals(row[i])) {
116                    changed = true;
117                }
118                prevoptype = optype;
119                optype = (String) row[i];
120            } else if (INDEX_COLUMN.equals(columns[i])) {
121                if (index != null && !index.equals(row[i])) {
122                    changed = true;
123                }
124                previndex = index;
125                index = (String) row[i];
126            } else if (TYPE_COLUMN.equals(columns[i])) {
127                if (type != null && !type.equals(row[i])) {
128                    changed = true;
129                }
130                prevtype = type;
131                type = (String) row[i];
132            } else if (ID_COLUMN.equals(columns[i])) {
133                if (id != null && !id.equals(row[i])) {
134                    changed = true;
135                }
136                previd = id;
137                id = (String) row[i];
138            }
139        }
140        if (changed) {
141            flush(prevoptype, previndex, prevtype, previd);
142        }
143        for (int i = 0; i < columns.length; i++) {
144            if (!OPTYPE_COLUMN.equals(columns[i]) &&
145                !INDEX_COLUMN.equals(columns[i]) && 
146                    !TYPE_COLUMN.equals(columns[i]) && 
147                    !ID_COLUMN.equals(columns[i])) {
148                merge(map, columns[i], row[i]);
149            }
150        }
151    }
152
153    /**
154     * Implement RowListener interface. A row is coming in and needs to be merged.
155     *
156     * @param index the index 
157     * @param type the type
158     * @param id the id
159     * @param keys the keys (column labels) of the row
160     * @param values the values of the row
161     * @throws IOException
162     */
163    @Override
164    public void row(String optype, String index, String type, String id, List<String> keys, List<Object> values) 
165            throws IOException {
166        boolean changed = false;
167        if (this.optype != null && !this.optype.equals(optype)) {
168            changed = true;
169        }
170        this.prevoptype = this.optype;
171        this.optype = optype;
172        if (this.index != null && !this.index.equals(index)) {
173            changed = true;
174        }
175        this.previndex = this.index;
176        this.index = index;
177        if (this.type != null && !this.type.equals(type)) {
178            changed = true;
179        }
180        this.prevtype = this.type;
181        this.type = type;
182        if (this.id != null && !this.id.equals(id)) {
183            changed = true;
184        }
185        this.previd = this.id;
186        this.id = id;
187        if (changed) {
188            flush(prevoptype, previndex, prevtype, previd);
189        }
190        for (int i = 0; i < keys.size(); i++) {
191            merge(map, keys.get(i), values.get(i));
192        }
193    }
194
195    public XContentBuilder getBuilder() {
196        return builder;
197    }
198
199    /**
200     * Flush and invoke appropriate operation type
201     *
202     * @param optype the operation type
203     * @param index the index 
204     * @param type the type
205     * @param id the id
206     * @throws IOException
207     */
208    public void flush(String optype, String index, String type, String id) throws IOException {
209        if (!map.isEmpty()) {
210            if (index != null) {
211                digest.update(index.getBytes(DIGEST_ENCODING));
212            }
213            if (type != null) {
214                digest.update(type.getBytes(DIGEST_ENCODING));
215            }
216            if (id != null) {
217                digest.update(id.getBytes(DIGEST_ENCODING));
218            }
219            build(map);
220            if (listener != null) {
221                if (OP_CREATE.equals(optype)) {
222                    listener.create(index, type, id, version, builder);
223                } else if (OP_INDEX.equals(optype)) {
224                    listener.index(index, type, id, version, builder);
225                } else if (OP_DELETE.equals(optype)) {
226                    listener.delete(index, type, id);
227                }
228            }
229            builder.close();
230            builder = jsonBuilder();
231            map = new HashMap();
232        }
233    }
234
235    /**
236     * Return a message digest (in base64-encoded form)
237     *
238     * @return the message digest
239     */
240    public String getDigest() {
241        return digestString;
242    }
243
244    /**
245     * Close the merger
246     *
247     * @throws IOException
248     */
249    public void close() throws IOException {
250        if (!closed) {
251            flush(optype, index, type, id);
252            this.digestString = Base64.encodeBytes(digest.digest());
253            closed = true;
254        }
255    }
256
257    /**
258     * Merge key/value pair to a map holding a JSON object. The key consists of
259     * a path pointing to the value position in the JSON object. The key,
260     * representing a path, is divided into head/tail. The recursion terminates
261     * if there is only a head and no tail. In this case, the value is added as
262     * a tuple to the map. If the head key exists, the merge process is
263     * continued by following the path represented by the key. If the path does
264     * not exist, a new map is created. A conflict arises if there is no map at
265     * a head key position. Then, the prefix given in the path is considered
266     * illegal.
267     *
268     * @param map the map for the JSON object
269     * @param key the key
270     * @param value the value
271     */
272    protected void merge(Map<String, Object> map, String key, Object value) {
273        int i = key.indexOf(delimiter);
274        if (i <= 0) {
275            map.put(key, new ValueSet(map.get(key), value));
276        } else {
277            String p = key.substring(0, i);
278            String q = key.substring(i + 1);
279            if (map.containsKey(p)) {
280                Object o = map.get(p);
281                if (o instanceof Map) {
282                    merge((Map<String, Object>) o, q, value);
283                } else {
284                    throw new IllegalArgumentException("illegal prefix: " + p);
285                }
286            } else {
287                Map<String, Object> m = new HashMap();
288                map.put(p, m);
289                merge(m, q, value);
290            }
291        }
292    }
293
294    /**
295     * Build JSON with the help of XContentBuilder.
296     *
297     * @param map the map holding the JSON object
298     * @throws IOException
299     */
300    private void build(Map<String, Object> map) throws IOException {
301        builder.startObject();
302        for (String k : map.keySet()) {
303            builder.field(k);
304            digest.update(k.getBytes(DIGEST_ENCODING));
305            Object o = map.get(k);
306            if (o instanceof ValueSet) {
307                ((ValueSet) o).build(builder, digest, DIGEST_ENCODING);
308            } else if (o instanceof Map) {
309                build((Map<String, Object>) o);
310            }
311        }
312        builder.endObject();
313    }
