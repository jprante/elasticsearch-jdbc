062    public SQLService setRounding(String rounding) {
063        if ("ceiling".equalsIgnoreCase(rounding)) {
064            this.rounding = BigDecimal.ROUND_CEILING;
065        } else if ("down".equalsIgnoreCase(rounding)) {
066            this.rounding = BigDecimal.ROUND_DOWN;
067        } else if ("floor".equalsIgnoreCase(rounding)) {
068            this.rounding = BigDecimal.ROUND_FLOOR;
069        } else if ("halfdown".equalsIgnoreCase(rounding)) {
070            this.rounding = BigDecimal.ROUND_HALF_DOWN;
071        } else if ("halfeven".equalsIgnoreCase(rounding)) {
072            this.rounding = BigDecimal.ROUND_HALF_EVEN;
073        } else if ("halfup".equalsIgnoreCase(rounding)) {
074            this.rounding = BigDecimal.ROUND_HALF_UP;
075        } else if ("unnecessary".equalsIgnoreCase(rounding)) {
076            this.rounding = BigDecimal.ROUND_UNNECESSARY;
077        } else if ("up".equalsIgnoreCase(rounding)) {
078            this.rounding = BigDecimal.ROUND_UP;
079        }
080        return this;
081    }

083    public SQLService setPrecision(int scale) {
084        this.scale = scale;
085        return this;
086    }

088    /**
089     * Get JDBC connection
090     *
091     * @param driverClassName
092     * @param jdbcURL
093     * @param user
094     * @param password
095     * @return the connection
096     * @throws ClassNotFoundException
097     * @throws SQLException
098     */
099    public Connection getConnection(final String driverClassName,
100            final String jdbcURL, final String user, final String password, boolean readOnly)
101            throws ClassNotFoundException, SQLException {
102        Class.forName(driverClassName);
103        this.connection = DriverManager.getConnection(jdbcURL, user, password);
104        connection.setReadOnly(readOnly);
105        connection.setAutoCommit(false);
106        return connection;
107    }

109    /**
110     * Prepare statement
111     *
112     * @param connection
113     * @param sql
114     * @return a prepared statement
115     * @throws SQLException
116     */
117    public PreparedStatement prepareStatement(Connection connection, String sql)
118            throws SQLException {
119        return connection.prepareStatement(sql, ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY);
120    }

122    /**
123     * Prepare statement for processing a river sync table
124     *
125     * @param connection the JDBC connection
126     * @param riverName the name of the river
127     * @param optype the operation type
128     * @param interval the interval to cover, in milliseconds (back from current
129     * time)
130     * @return a preapred river table statement
131     * @throws SQLException
132     */
133    public PreparedStatement prepareRiverTableStatement(Connection connection, String riverName, String optype, long interval)
134            throws SQLException {
135        PreparedStatement p = connection.prepareStatement("select * from " + riverName + " where source_operation = ? and target_operation = 'n/a' and source_timestamp between ? and ?",
136                ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY);
137        p.setString(1, optype);
138        java.util.Date d = new java.util.Date();
139        long now = d.getTime();
140        p.setTimestamp(2, new Timestamp(now - interval));
141        p.setTimestamp(3, new Timestamp(now));
142        return p;
143    }

145    /**
146     * Bind values to prepared statement
147     *
148     * @param pstmt
149     * @param values
150     * @throws SQLException
151     */
152    public void bind(PreparedStatement pstmt, List<Object> values) throws SQLException {
153        if (values == null) {
154            return;
155        }
156        for (int i = 1; i <= values.size(); i++) {
157            bind(pstmt, i, values.get(i - 1));
158        }
159    }

175    /**
176     * Get next row and prepare the values for processing. The labels of each
177     * columns are used for the RowListener as paths for JSON object merging.
178     *
179     * @param result the result set
180     * @param listener the listener
181     * @return true if row exists and was processed, false otherwise
182     * @throws SQLException
183     * @throws IOException
184     */
185    public boolean nextRow(ResultSet result, RowListener listener)
186            throws SQLException, IOException {
187        if (result.next()) {
188            String index = null;
189            String type = null;
190            String id = null;
191            processRow(result, listener, "index", index, type, id);
192            return true;
193        }
194        return false;
195    }

197    /**
198     * Get next row from a river table. The river table contains embedded SQL
199     * which is executed.
200     *
201     * @param result the result set
202     * @param listener the row listener
203     * @return true if there was a row being executed, false otherwise
204     * @throws SQLException
205     * @throws IOException
206     */
207    public boolean nextRiverTableRow(ResultSet result, RowListener listener)
208            throws SQLException, IOException {
209        if (result.next()) {
210            ResultSetMetaData metadata = result.getMetaData();
211            String operation = null;
212            String index = null;
213            String type = null;
214            String id = null;
215            String sql = null;
216            int columns = metadata.getColumnCount();
217            for (int i = 1; i <= columns; i++) {
218                String name = metadata.getColumnLabel(i);
219                if ("_index".equalsIgnoreCase(name)) {
220                    index = result.getString(i);
221                } else if ("_type".equalsIgnoreCase(name)) {
222                    type = result.getString(i);
223                } else if ("_id".equalsIgnoreCase(name)) {
224                    id = result.getString(i);
225                } else if ("source_operation".equalsIgnoreCase(name)) {
226                    operation = result.getString(i);
227                } else if ("source_sql".equalsIgnoreCase(name)) {
228                    sql = result.getString(i);
229                }
230            }
231            if (sql != null) {
232                // execute embedded SQL
233                PreparedStatement stmt = prepareStatement(connection, sql);
234                ResultSet rs = stmt.executeQuery();
235                long rows = 0L;
236                if (rs.next()) {
237                    processRow(rs, listener, operation, index, type, id);
238                    rows++;
239                }
240                logger.info("embedded sql gave " + rows + " rows");
241                rs.close();
242                stmt.close();
243            }
244            return true;
245        }
246        return false;
247    }

249    private void processRow(ResultSet result, RowListener listener, String operation, String index, String type, String id)
250            throws SQLException, IOException {
251        LinkedList<String> keys = new LinkedList();
252        LinkedList<Object> values = new LinkedList();
253        ResultSetMetaData metadata = result.getMetaData();
254        int columns = metadata.getColumnCount();
255        for (int i = 1; i <= columns; i++) {
256            String name = metadata.getColumnLabel(i);
257            if ("_index".equalsIgnoreCase(name)) {
258                index = result.getString(i);
259                continue;
260            } else if ("_type".equalsIgnoreCase(name)) {
261                type = result.getString(i);
262                continue;
263            } else if ("_id".equalsIgnoreCase(name)) {
264                id = result.getString(i);
265                continue;
266            }
267            keys.add(name);
268            switch (metadata.getColumnType(i)) {
269                /**
270                 * The JDBC types CHAR, VARCHAR, and LONGVARCHAR are closely
271                 * related. CHAR represents a small, fixed-length character
272                 * string, VARCHAR represents a small, variable-length character
273                 * string, and LONGVARCHAR represents a large, variable-length
274                 * character string.
275                 */
276                case Types.CHAR:
277                case Types.VARCHAR:
278                case Types.LONGVARCHAR: {
279                    String s = result.getString(i);
280                    values.add(s);
281                    break;
282                }
283                case Types.NCHAR:
284                case Types.NVARCHAR:
285                case Types.LONGNVARCHAR: {
286                    String s = result.getNString(i);
287                    values.add(s);
288                    break;
289                }
290                /**
291                 * The JDBC types BINARY, VARBINARY, and LONGVARBINARY are
292                 * closely related. BINARY represents a small, fixed-length
293                 * binary value, VARBINARY represents a small, variable-length
294                 * binary value, and LONGVARBINARY represents a large,
295                 * variable-length binary value
296                 */
297                case Types.BINARY:
298                case Types.VARBINARY:
299                case Types.LONGVARBINARY: {
300                    byte[] b = result.getBytes(name);
301                    values.add(b);
302                    break;
303                }
304                /**
305                 * The JDBC type ARRAY represents the SQL3 type ARRAY.
306                 *
307                 * An ARRAY value is mapped to an instance of the Array
308                 * interface in the Java programming language. If a driver
309                 * follows the standard implementation, an Array object
310                 * logically points to an ARRAY value on the server rather than
311                 * containing the elements of the ARRAY object, which can
312                 * greatly increase efficiency. The Array interface contains
313                 * methods for materializing the elements of the ARRAY object on
314                 * the client in the form of either an array or a ResultSet
315                 * object.
316                 */
317                case Types.ARRAY: {
318                    Array a = result.getArray(i);
319                    values.add(a != null ? a.toString() : null);
320                    break;
321                }
322                /**
323                 * The JDBC type BIGINT represents a 64-bit signed integer value
324                 * between -9223372036854775808 and 9223372036854775807.
325                 *
326                 * The corresponding SQL type BIGINT is a nonstandard extension
327                 * to SQL. In practice the SQL BIGINT type is not yet currently
328                 * implemented by any of the major databases, and we recommend
329                 * that its use be avoided in code that is intended to be
330                 * portable.
331                 *
332                 * The recommended Java mapping for the BIGINT type is as a Java
333                 * long.
334                 */
335                case Types.BIGINT: {
336                    long l = result.getLong(i);
337                    values.add(l);
338                    break;
339                }
340                /**
341                 * The JDBC type BIT represents a single bit value that can be
342                 * zero or one.
343                 *
344                 * SQL-92 defines an SQL BIT type. However, unlike the JDBC BIT
345                 * type, this SQL-92 BIT type can be used as a parameterized
346                 * type to define a fixed-length binary string. Fortunately,
347                 * SQL-92 also permits the use of the simple non-parameterized
348                 * BIT type to represent a single binary digit, and this usage
349                 * corresponds to the JDBC BIT type. Unfortunately, the SQL-92
350                 * BIT type is only required in "full" SQL-92 and is currently
351                 * supported by only a subset of the major databases. Portable
352                 * code may therefore prefer to use the JDBC SMALLINT type,
353                 * which is widely supported.
354                 */
355                case Types.BIT: {
356                    int n = result.getInt(i);
357                    values.add(n);
358                    break;
359                }
360                /**
361                 * The JDBC type BOOLEAN, which is new in the JDBC 3.0 API, maps
362                 * to a boolean in the Java programming language. It provides a
363                 * representation of true and false, and therefore is a better
364                 * match than the JDBC type BIT, which is either 1 or 0.
365                 */
366                case Types.BOOLEAN: {
367                    boolean b = result.getBoolean(i);
368                    values.add(b);
369                    break;
370                }
371                /**
372                 * The JDBC type BLOB represents an SQL3 BLOB (Binary Large
373                 * Object).
374                 *
375                 * A JDBC BLOB value is mapped to an instance of the Blob
376                 * interface in the Java programming language. If a driver
377                 * follows the standard implementation, a Blob object logically
378                 * points to the BLOB value on the server rather than containing
379                 * its binary data, greatly improving efficiency. The Blob
380                 * interface provides methods for materializing the BLOB data on
381                 * the client when that is desired.
382                 */
383                case Types.BLOB: {
384                    Blob blob = result.getBlob(i);
385                    if (blob != null) {
386                        long n = blob.length();
387                        if (n > Integer.MAX_VALUE) {
388                            throw new IOException("can't process blob larger than Integer.MAX_VALUE");
389                        }
390                        values.add(blob.getBytes(1, (int) n));
391                        blob.free();
392                    }
393                    break;
394                }
395                /**
396                 * The JDBC type CLOB represents the SQL3 type CLOB (Character
397                 * Large Object).
398                 *
399                 * A JDBC CLOB value is mapped to an instance of the Clob
400                 * interface in the Java programming language. If a driver
401                 * follows the standard implementation, a Clob object logically
402                 * points to the CLOB value on the server rather than containing
403                 * its character data, greatly improving efficiency. Two of the
404                 * methods on the Clob interface materialize the data of a CLOB
405                 * object on the client.
406                 */
407                case Types.CLOB: {
408                    Clob clob = result.getClob(i);
409                    if (clob != null) {
410                        long n = clob.length();
411                        if (n > Integer.MAX_VALUE) {
412                            throw new IOException("can't process clob larger than Integer.MAX_VALUE");
413                        }
414                        values.add(clob.getSubString(1, (int) n));
415                        clob.free();
416                    }
417                    break;
418                }
419                case Types.NCLOB: {
420                    NClob nclob = result.getNClob(i);
421                    if (nclob != null) {
422                        long n = nclob.length();
423                        if (n > Integer.MAX_VALUE) {
424                            throw new IOException("can't process nclob larger than Integer.MAX_VALUE");
425                        }
426                        values.add(nclob.getSubString(1, (int) n));
427                        nclob.free();
428                    }
429                    break;
430                }
431                /**
432                 * The JDBC type DATALINK, new in the JDBC 3.0 API, is a column
433                 * value that references a file that is outside of a data source
434                 * but is managed by the data source. It maps to the Java type
435                 * java.net.URL and provides a way to manage external files. For
436                 * instance, if the data source is a DBMS, the concurrency
437                 * controls it enforces on its own data can be applied to the
438                 * external file as well.
439                 *
440                 * A DATALINK value is retrieved from a ResultSet object with
441                 * the ResultSet methods getURL or getObject. If the Java
442                 * platform does not support the type of URL returned by getURL
443                 * or getObject, a DATALINK value can be retrieved as a String
444                 * object with the method getString.
445                 *
446                 * java.net.URL values are stored in a database using the method
447                 * setURL. If the Java platform does not support the type of URL
448                 * being set, the method setString can be used instead.
449                 *
450                 *
451                 */
452                case Types.DATALINK: {
453                    URL url = result.getURL(i);
454                    values.add(url);
455                    break;
456                }
457                /**
458                 * The JDBC DATE type represents a date consisting of day,
459                 * month, and year. The corresponding SQL DATE type is defined
460                 * in SQL-92, but it is implemented by only a subset of the
461                 * major databases. Some databases offer alternative SQL types
462                 * that support similar semantics.
463                 */
464                case Types.DATE: {
465                    try {
466                        Date d = result.getDate(i);
467                        values.add(d != null ? DateUtil.formatDateISO(d.getTime()) : null);
468                    } catch (SQLException e) {
469                        values.add(null);
470                    }
471                    break;
472                }
473                case Types.TIME: {
474                    try {
475                        Time t = result.getTime(i);
476                        values.add(t != null ? DateUtil.formatDateISO(t.getTime()) : null);
477                    } catch (SQLException e) {
478                        values.add(null);
479                    }
480                    break;
481                }
482                case Types.TIMESTAMP: {
483                    try {
484                        Timestamp t = result.getTimestamp(i);
485                        values.add(t != null ? DateUtil.formatDateISO(t.getTime()) : null);
486                    } catch (SQLException e) {
487                        // java.sql.SQLException: Cannot convert value '0000-00-00 00:00:00' from column ... to TIMESTAMP.
488                        values.add(null);
489                    }
490                    break;
491                }
492                /**
493                 * The JDBC types DECIMAL and NUMERIC are very similar. They
494                 * both represent fixed-precision decimal values.
495                 *
496                 * The corresponding SQL types DECIMAL and NUMERIC are defined
497                 * in SQL-92 and are very widely implemented. These SQL types
498                 * take precision and scale parameters. The precision is the
499                 * total number of decimal digits supported, and the scale is
500                 * the number of decimal digits after the decimal point. For
501                 * most DBMSs, the scale is less than or equal to the precision.
502                 * So for example, the value "12.345" has a precision of 5 and a
503                 * scale of 3, and the value ".11" has a precision of 2 and a
504                 * scale of 2. JDBC requires that all DECIMAL and NUMERIC types
505                 * support both a precision and a scale of at least 15.
506                 *
507                 * The sole distinction between DECIMAL and NUMERIC is that the
508                 * SQL-92 specification requires that NUMERIC types be
509                 * represented with exactly the specified precision, whereas for
510                 * DECIMAL types, it allows an implementation to add additional
511                 * precision beyond that specified when the type was created.
512                 * Thus a column created with type NUMERIC(12,4) will always be
513                 * represented with exactly 12 digits, whereas a column created
514                 * with type DECIMAL(12,4) might be represented by some larger
515                 * number of digits.
516                 *
517                 * The recommended Java mapping for the DECIMAL and NUMERIC
518                 * types is java.math.BigDecimal. The java.math.BigDecimal type
519                 * provides math operations to allow BigDecimal types to be
520                 * added, subtracted, multiplied, and divided with other
521                 * BigDecimal types, with integer types, and with floating point
522                 * types.
523                 *
524                 * The method recommended for retrieving DECIMAL and NUMERIC
525                 * values is ResultSet.getBigDecimal. JDBC also allows access to
526                 * these SQL types as simple Strings or arrays of char. Thus,
527                 * Java programmers can use getString to receive a DECIMAL or
528                 * NUMERIC result. However, this makes the common case where
529                 * DECIMAL or NUMERIC are used for currency values rather
530                 * awkward, since it means that application writers have to
531                 * perform math on strings. It is also possible to retrieve
532                 * these SQL types as any of the Java numeric types.
533                 */
534                case Types.DECIMAL:
535                case Types.NUMERIC: {
536                    BigDecimal bd = null;
537                    try {
538                        bd = result.getBigDecimal(i);
539                    } catch (NullPointerException e) {
540                        // getBigDecimal() should get obsolete. Most seem to use getString/getObject anayway.
541                        // But is it true? JDBC NPE exists since 13 years? 
542                        // http://forums.codeguru.com/archive/index.php/t-32443.html
543                        // Null values are driving us nuts in JDBC:
544                        // http://stackoverflow.com/questions/2777214/when-accessing-resultsets-in-jdbc-is-there-an-elegant-way-to-distinguish-betwee
545                    }
546                    values.add(bd == null ? null
547                            : scale >= 0 ? bd.setScale(scale, rounding).doubleValue()
548                            : bd.toString());
549                    break;
550                }
551                /**
552                 * The JDBC type DOUBLE represents a "double precision" floating
553                 * point number that supports 15 digits of mantissa.
554                 *
555                 * The corresponding SQL type is DOUBLE PRECISION, which is
556                 * defined in SQL-92 and is widely supported by the major
557                 * databases. The SQL-92 standard leaves the precision of DOUBLE
558                 * PRECISION up to the implementation, but in practice all the
559                 * major databases supporting DOUBLE PRECISION support a
560                 * mantissa precision of at least 15 digits.
561                 *
562                 * The recommended Java mapping for the DOUBLE type is as a Java
563                 * double.
564                 */
565                case Types.DOUBLE: {
566                    double d = result.getDouble(i);
567                    values.add(d);
568                    break;
569                }
570                /**
571                 * The JDBC type FLOAT is basically equivalent to the JDBC type
572                 * DOUBLE. We provided both FLOAT and DOUBLE in a possibly
573                 * misguided attempt at consistency with previous database APIs.
574                 * FLOAT represents a "double precision" floating point number
575                 * that supports 15 digits of mantissa.
576                 *
577                 * The corresponding SQL type FLOAT is defined in SQL-92. The
578                 * SQL-92 standard leaves the precision of FLOAT up to the
579                 * implementation, but in practice all the major databases
580                 * supporting FLOAT support a mantissa precision of at least 15
581                 * digits.
582                 *
583                 * The recommended Java mapping for the FLOAT type is as a Java
584                 * double. However, because of the potential confusion between
585                 * the double precision SQL FLOAT and the single precision Java
586                 * float, we recommend that JDBC programmers should normally use
587                 * the JDBC DOUBLE type in preference to FLOAT.
588                 */
589                case Types.FLOAT: {
590                    double d = result.getDouble(i);
591                    values.add(d);
592                    break;
593                }
594                /**
595                 * The JDBC type INTEGER represents a 32-bit signed integer
596                 * value ranging between -2147483648 and 2147483647.
597                 *
598                 * The corresponding SQL type, INTEGER, is defined in SQL-92 and
599                 * is widely supported by all the major databases. The SQL-92
600                 * standard leaves the precision of INTEGER up to the
601                 * implementation, but in practice all the major databases
602                 * support at least 32 bits.
603                 *
604                 * The recommended Java mapping for the INTEGER type is as a
605                 * Java int.
606                 */
607                case Types.INTEGER: {
608                    int n = result.getInt(i);
609                    values.add(n);
610                    break;
611                }
612                /**
613                 * The JDBC type JAVA_OBJECT, added in the JDBC 2.0 core API,
614                 * makes it easier to use objects in the Java programming
615                 * language as values in a database. JAVA_OBJECT is simply a
616                 * type code for an instance of a class defined in the Java
617                 * programming language that is stored as a database object. The
618                 * type JAVA_OBJECT is used by a database whose type system has
619                 * been extended so that it can store Java objects directly. The
620                 * JAVA_OBJECT value may be stored as a serialized Java object,
621                 * or it may be stored in some vendor-specific format.
622                 *
623                 * The type JAVA_OBJECT is one of the possible values for the
624                 * column DATA_TYPE in the ResultSet objects returned by various
625                 * DatabaseMetaData methods, including getTypeInfo, getColumns,
626                 * and getUDTs. The method getUDTs, part of the new JDBC 2.0
627                 * core API, will return information about the Java objects
628                 * contained in a particular schema when it is given the
629                 * appropriate parameters. Having this information available
630                 * facilitates using a Java class as a database type.
631                 */
632                case Types.OTHER:
633                case Types.JAVA_OBJECT: {
634                    Object o = result.getObject(i);
635                    values.add(o);
636                    break;
637                }
638                /**
639                 * The JDBC type REAL represents a "single precision" floating
640                 * point number that supports seven digits of mantissa.
641                 *
642                 * The corresponding SQL type REAL is defined in SQL-92 and is
643                 * widely, though not universally, supported by the major
644                 * databases. The SQL-92 standard leaves the precision of REAL
645                 * up to the implementation, but in practice all the major
646                 * databases supporting REAL support a mantissa precision of at
647                 * least seven digits.
648                 *
649                 * The recommended Java mapping for the REAL type is as a Java
650                 * float.
651                 */
652                case Types.REAL: {
653                    float f = result.getFloat(i);
654                    values.add(f);
655                    break;
656                }
657                /**
658                 * The JDBC type SMALLINT represents a 16-bit signed integer
659                 * value between -32768 and 32767.
660                 *
661                 * The corresponding SQL type, SMALLINT, is defined in SQL-92
662                 * and is supported by all the major databases. The SQL-92
663                 * standard leaves the precision of SMALLINT up to the
664                 * implementation, but in practice, all the major databases
665                 * support at least 16 bits.
666                 *
667                 * The recommended Java mapping for the JDBC SMALLINT type is as
668                 * a Java short.
669                 */
670                case Types.SMALLINT: {
671                    int n = result.getInt(i);
672                    values.add(n);
673                    break;
674                }
675                case Types.SQLXML: {
676                    SQLXML xml = result.getSQLXML(columns);
677                    values.add(xml != null ? xml.getString() : null);
678                    break;
679                }
680                /**
681                 * The JDBC type TINYINT represents an 8-bit integer value
682                 * between 0 and 255 that may be signed or unsigned.
683                 *
684                 * The corresponding SQL type, TINYINT, is currently supported
685                 * by only a subset of the major databases. Portable code may
686                 * therefore prefer to use the JDBC SMALLINT type, which is
687                 * widely supported.
688                 *
689                 * The recommended Java mapping for the JDBC TINYINT type is as
690                 * either a Java byte or a Java short. The 8-bit Java byte type
691                 * represents a signed value from -128 to 127, so it may not
692                 * always be appropriate for larger TINYINT values, whereas the
693                 * 16-bit Java short will always be able to hold all TINYINT
694                 * values.
695                 */
696                case Types.TINYINT: {
697                    int n = result.getInt(i);
698                    values.add(n);
699                    break;
700                }
701                case Types.NULL: {
702                    values.add(null);
703                    break;
704                }
705                /**
706                 * The JDBC type DISTINCT field (Types class)>DISTINCT
707                 * represents the SQL3 type DISTINCT.
708                 *
709                 * The standard mapping for a DISTINCT type is to the Java type
710                 * to which the base type of a DISTINCT object would be mapped.
711                 * For example, a DISTINCT type based on a CHAR would be mapped
712                 * to a String object, and a DISTINCT type based on an SQL
713                 * INTEGER would be mapped to an int.
714                 *
715                 * The DISTINCT type may optionally have a custom mapping to a
716                 * class in the Java programming language. A custom mapping
717                 * consists of a class that implements the interface SQLData and
718                 * an entry in a java.util.Map object.
719                 */
720                case Types.DISTINCT: {
721                    logger.warn("JDBC type not implemented: {}", metadata.getColumnType(i));
722                    values.add(null);
723                    break;
724                }
725                /**
726                 * The JDBC type STRUCT represents the SQL99 structured type. An
727                 * SQL structured type, which is defined by a user with a CREATE
728                 * TYPE statement, consists of one or more attributes. These
729                 * attributes may be any SQL data type, built-in or
730                 * user-defined.
731                 *
732                 * The standard mapping for the SQL type STRUCT is to a Struct
733                 * object in the Java programming language. A Struct object
734                 * contains a value for each attribute of the STRUCT value it
735                 * represents.
736                 *
737                 * A STRUCT value may optionally be custom mapped to a class in
738                 * the Java programming language, and each attribute in the
739                 * STRUCT may be mapped to a field in the class. A custom
740                 * mapping consists of a class that implements the interface
741                 * SQLData and an entry in a java.util.Map object.
742                 *
743                 *
744                 */
745                case Types.STRUCT: {
746                    logger.warn("JDBC type not implemented: {}", metadata.getColumnType(i));
747                    values.add(null);
748                    break;
749                }
750                case Types.REF: {
751                    logger.warn("JDBC type not implemented: {}", metadata.getColumnType(i));
752                    values.add(null);
753                    break;
754                }
755                case Types.ROWID: {
756                    logger.warn("JDBC type not implemented: {}", metadata.getColumnType(i));
757                    values.add(null);
758                    break;
759                }
760                default: {
761                    logger.warn("unknown JDBC type ignored: {}", metadata.getColumnType(i));
762                    values.add(null);
763                    break;
764                }
765            }
766        }
767        if (id == null) {
768            id = Integer.toString(result.getRow());
769        }
770        if (listener != null) {
771            listener.row(operation, index, type, id, keys, values);
772        }
773    }

805    private void bind(PreparedStatement pstmt, int i, Object value) throws SQLException {
806        if (value == null) {
807            pstmt.setNull(i, Types.VARCHAR);
808        } else if (value instanceof String) {
809            String s = (String) value;
810            if ("$now".equals(s)) {
811                pstmt.setDate(i, new Date(new java.util.Date().getTime()));
812            } else {
813                pstmt.setString(i, (String) value);
814            }
815        } else if (value instanceof Integer) {
816            pstmt.setInt(i, ((Integer) value).intValue());
817        } else if (value instanceof Long) {
818            pstmt.setLong(i, ((Long) value).longValue());
819        } else if (value instanceof BigDecimal) {
820            pstmt.setBigDecimal(i, (BigDecimal) value);
821        } else if (value instanceof Date) {
822            pstmt.setDate(i, (Date) value);
823        } else if (value instanceof Timestamp) {
824            pstmt.setTimestamp(i, (Timestamp) value);
825        } else if (value instanceof Float) {
826            pstmt.setFloat(i, ((Float) value).floatValue());
827        } else if (value instanceof Double) {
828            pstmt.setDouble(i, ((Double) value).doubleValue());
829        } else {
830            pstmt.setObject(i, value);
831        }
832    }

834    /**
835     * Acknowledge a bulk item response back to the river table. Fill columns
836     * target_timestamp, target_operation, target_failed, target_message.
837     *
838     * @param riverName
839     * @param response
840     * @throws IOException
841     */
842    @Override
843    public void acknowledge(String riverName, BulkItemResponse[] response) throws IOException {
844        if (response == null) {
845            logger.warn("can't acknowledge null bulk response");
846        }
847        try {
848            String sql = " update " + riverName + " set target_timestamp = ?, target_operation = ?, target_failed = ?, target_message = ? where source_operation = ? and _index = ? and _type = ? and _id = ? and target_operation = 'n/a'";
849            PreparedStatement pstmt = connection.prepareStatement(sql);
850            for (BulkItemResponse resp : response) {
851                pstmt.setTimestamp(1, new Timestamp(new java.util.Date().getTime()));
852                pstmt.setString(2, resp.opType());
853                pstmt.setBoolean(3, resp.failed());
854                pstmt.setString(4, resp.failureMessage());
855                pstmt.setString(5, resp.opType());
856                pstmt.setString(6, resp.index());
857                pstmt.setString(7, resp.type());
858                pstmt.setString(8, resp.id());
859                pstmt.executeUpdate();
860            }
861            pstmt.close();
862            connection.commit();
863        } catch (SQLException ex) {
864            throw new IOException(ex);
865        }
866    }
