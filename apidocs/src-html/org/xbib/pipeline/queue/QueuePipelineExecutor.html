001package org.xbib.pipeline.queue;
002
003import org.xbib.pipeline.Pipeline;
004import org.xbib.pipeline.PipelineExecutor;
005import org.xbib.pipeline.PipelineProvider;
006import org.xbib.pipeline.PipelineRequest;
007import org.xbib.pipeline.PipelineSink;
008import org.xbib.pipeline.element.PipelineElement;
009
010import java.io.IOException;
011import java.util.Collection;
012import java.util.HashSet;
013import java.util.LinkedList;
014import java.util.Set;
015import java.util.concurrent.BlockingQueue;
016import java.util.concurrent.CountDownLatch;
017import java.util.concurrent.ExecutionException;
018import java.util.concurrent.ExecutorService;
019import java.util.concurrent.Executors;
020import java.util.concurrent.Future;
021import java.util.concurrent.SynchronousQueue;
022import java.util.concurrent.TimeUnit;
023
024/**
025 * An Queue Pipeline setExecutor. This setExecutor can execute pipelines in parallel
026 * and manage a queue that have to be processed by the pipelines.
027 * <p/>
028 * By doing this, the concurrency works on archive entry level, not URI level.
029 * <p/>
030 * Pipelines are created by a pipeline provider.
031 * The maximum number of concurrent pipelines is 256.
032 * Each pipeline can receive archive entries, which are put into a blocking queue by
033 * this setExecutor.
034 *
035 * @param <T> the result type
036 * @param <R> the pipeline request type
037 * @param <P> the pipeline type
038 * @param <E> the element type
039 */
040public class QueuePipelineExecutor<T, R extends PipelineRequest, P extends Pipeline<T, R>, E extends PipelineElement>
041        implements PipelineExecutor<T, R, P> {
042
043    private ExecutorService executorService;
044
045    private PipelineProvider<P> provider;
046
047    private Set<P> pipelines;
048
049    private Collection<Future<T>> futures;
050
051    private PipelineSink<T> sink;
052
053    private CountDownLatch latch;
054
055    private BlockingQueue<E> queue;
056
057    private int concurrency;
058
059    private volatile boolean closed;
060
061    @Override
062    public QueuePipelineExecutor<T, R, P, E> setConcurrency(int concurrency) {
063        this.concurrency = concurrency;
064        return this;
065    }
066
067    @Override
068    public QueuePipelineExecutor<T, R, P, E> setPipelineProvider(PipelineProvider<P> provider) {
069        this.provider = provider;
070        return this;
071    }
072
073    @Override
074    public QueuePipelineExecutor<T, R, P, E> setSink(PipelineSink<T> sink) {
075        this.sink = sink;
076        return this;
077    }
078
079    @Override
080    public QueuePipelineExecutor<T, R, P, E> prepare() {
081        if (provider == null) {
082            throw new IllegalArgumentException("no provider set");
083        }
084        if (executorService == null) {
085            this.executorService = Executors.newFixedThreadPool(concurrency);
086        }
087        if (pipelines == null) {
088            this.pipelines = new HashSet<P>();
089        }
090        if (concurrency < 1) {
091            concurrency = 1;
092        }
093        this.queue = new SynchronousQueue<E>(true); // true means fair queue
094        this.latch = new CountDownLatch(concurrency);
095        for (int i = 0; i < Math.min(concurrency, 256); i++) {
096            P pipeline = provider.get();
097            pipelines.add(pipeline);
098        }
099        return this;
100    }
101
102    /**
103     * Usually, the pipelines receive elements from the setExecutor.
104     * Here we ensure that all pipelines that receive elements are invoked.
105     * This method returns immediately without waiting for the completion of pipelines.
106     *
107     * @return this setExecutor
108     */
109    @Override
110    public QueuePipelineExecutor<T, R, P, E> execute() {
111        futures = new LinkedList<Future<T>>();
112        for (P pipeline : pipelines) {
113            futures.add(executorService.submit(pipeline));
114        }
115        return this;
116    }
117
118    /**
119     * Wait for all the pipelines executed.
120     *
121     * @return this setExecutor
122     * @throws InterruptedException
123     * @throws java.util.concurrent.ExecutionException
124     */
125    public QueuePipelineExecutor<T, R, P, E> waitFor() throws InterruptedException, ExecutionException, IOException {
126        for (Future<T> future : futures) {
127            T t = future.get();
128            if (sink != null) {
129                sink.write(t);
130            }
131        }
132        return this;
133    }
134
135    @Override
136    public void shutdown() throws InterruptedException, IOException {
137        if (executorService == null) {
138            return;
139        }
140        executorService.shutdown();
141        if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
142            executorService.shutdownNow();
143            if (!executorService.awaitTermination(15, TimeUnit.SECONDS)) {
144                throw new IOException("pool did not terminate");
145            }
146        }
147    }
148
149    public void shutdown(E poisonElement) throws InterruptedException, ExecutionException, IOException {
150        if (closed) {
151            return;
152        }
153        closed = true;
154        // how many pipelines are still running?
155        int active = 0;
156        for (P pipeline : pipelines) {
157            if (pipeline.isRunning()) {
158                active++;
159            }
160        }
161        for (int i = 0; i < active; i++) {
162            queue.offer(poisonElement, 30, TimeUnit.SECONDS);
163        }
164        waitFor();
165        shutdown();
166    }
167
168    /**
169     * Get queue
170     *
171     * @return the queue
172     */
173    public BlockingQueue<E> queue() {
174        return queue;
175    }
176
177    /**
178     * Get pipelines
179     *
180     * @return the pipelines
181     */
182    public Set<P> getPipelines() {
183        return pipelines;
184    }
185
186    /**
187     * Count down the latch. Decreases the number of active pipelines.
188     * Called from a pipeline when it terminates.
189     *
190     * @return this setExecutor
191     */
192    public QueuePipelineExecutor countDown() {
193        latch.countDown();
194        return this;
195    }
196
197    /**
198     * Returns the number of pipelines.  If this pipeline setExecutor can receive requests,
199     * the returned number is greater than 0.
200     *
201     * @return number of pipelines ready to receive requests
202     */
203    public long canReceive() {
204        return latch.getCount();
205    }
206
207}
