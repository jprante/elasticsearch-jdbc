001package org.xbib.pipeline.element;
002
003import org.xbib.pipeline.PipelineRequest;
004
005import java.util.concurrent.atomic.AtomicLong;
006
007public class LongPipelineElement implements PipelineElement<AtomicLong>, PipelineRequest {
008
009    private AtomicLong n;
010
011    @Override
012    public AtomicLong get() {
013        return n;
014    }
015
016    @Override
017    public LongPipelineElement set(AtomicLong n) {
018        this.n = n;
019        return this;
020    }
021
022    @Override
023    public String toString() {
024        return n.toString();
025    }
026}
