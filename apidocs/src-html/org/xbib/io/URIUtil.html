031    /**
032     * This method adds a single key/value parameter to the query
033     * string of a given URI. Existing keys will be overwritten.
034     *
035     * @param uri      the URI
036     * @param key      the key
037     * @param value    the value
038     * @param encoding the encoding
039     * @return the URI
040     * @throws java.net.URISyntaxException
041     */
042    public static URI addParameter(URI uri, String key, String value, Charset encoding)
043            throws URISyntaxException {
044        Map<String, String> m = parseQueryString(uri, encoding);
045        m.put(key, value);
046        String query = renderQueryString(m);
047        return new URI(uri.getScheme(), uri.getAuthority(), uri.getPath(), query, uri.getFragment());
048    }

051     * This method adds a map of key/value parameters to the query
052     * string of a given URI. Existing keys will be overwritten.
053     *
054     * @param uri      the URI
055     * @param map      the map
056     * @param encoding the encoding
057     * @return the URI
058     * @throws java.net.URISyntaxException
059     */
060    public static URI addParameter(URI uri, Map<String, String> map, Charset encoding)
061            throws URISyntaxException {
062        Map<String, String> oldMap = parseQueryString(uri, encoding);
063        oldMap.putAll(map);
064        String query = renderQueryString(oldMap);
065        return new URI(uri.getScheme(), uri.getAuthority(), uri.getPath(), query, uri.getFragment());
066    }

068    private static String concat(Collection<String> c) {
069        StringBuilder sb = new StringBuilder();
070        for (String s : c) {
071            if (sb.length() > 0) {
072                sb.append(",");
073            }
074            sb.append(s);
075        }
076        return sb.toString();
077    }

079    /**
080     * Decodes an octet according to RFC 2396. According to this spec,
081     * any characters outside the range 0x20 - 0x7E must be escaped because
082     * they are not printable characters, except for any characters in the
083     * fragment identifier. This method will translate any escaped characters
084     * back to the original.
085     *
086     * @param s      the URI to decode
087     * @param encoding the encoding to decode into
088     * @return The decoded URI
089     */
090    public static String decode(String s, Charset encoding) {
091        if (s == null || s.isEmpty()) {
092            return null;
093        }
094        StringBuilder sb = new StringBuilder();
095        boolean fragment = false;
096        for (int i = 0; i < s.length(); i++) {
097            char ch = s.charAt(i);
098            switch (ch) {
099                case '+':
100                    sb.append(' ');
101                    break;
102                case '#':
103                    sb.append(ch);
104                    fragment = true;
105                    break;
106                case '%':
107                    if (!fragment) {
108                        // fast hex decode
109                        sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
110                                | Character.digit(s.charAt(++i), 16)));
111                    } else {
112                        sb.append(ch);
113                    }
114                    break;
115                default:
116                    sb.append(ch);
117                    break;
118            }
119        }
120        return new String(sb.toString().getBytes(LATIN1), encoding);
121    }

123    /**
124     * <p>Escape a string into URI syntax</p>
125     * <p>This function applies the URI escaping rules defined in
126     * section 2 of [RFC 2396], as amended by [RFC 2732], to the string
127     * supplied as the first argument, which typically represents all or part
128     * of a URI, URI reference or IRI. The effect of the function is to
129     * replace any special character in the string by an escape sequence of
130     * the form %xx%yy..., where xxyy... is the hexadecimal representation of
131     * the octets used to represent the character in US-ASCII for characters
132     * in the ASCII repertoire, and a different character encoding for
133     * non-ASCII characters.</p>
134     * <p>If the second argument is true, all characters are escaped
135     * other than lower case letters a-z, upper case letters A-Z, digits 0-9,
136     * and the characters referred to in [RFC 2396] as "marks": specifically,
137     * "-" | "_" | "." | "!" | "~" | "" | "'" | "(" | ")". The "%" character
138     * itself is escaped only if it is not followed by two hexadecimal digits
139     * (that is, 0-9, a-f, and A-F).</p>
140     * <p>[RFC 2396] does not define whether escaped URIs should use
141     * lower case or upper case for hexadecimal digits. To ensure that escaped
142     * URIs can be compared using string comparison functions, this function
143     * must always use the upper-case letters A-F.</p>
144     * <p>The character encoding used as the basis for determining the
145     * octets depends on the setting of the second argument.</p>
146     *
147     * @param s        the String to convert
148     * @param encoding The encoding to use for unsafe characters
149     * @return The converted String
150     */
151    public static String encode(String s, Charset encoding) {
152        if (s == null || s.isEmpty()) {
153            return null;
154        }
155        int length = s.length();
156        int start = 0;
157        int i = 0;
158        StringBuilder result = new StringBuilder(length);
159        while (true) {
160            while ((i < length) && isSafe(s.charAt(i))) {
161                i++;
162            }
163            // Safe character can just be added
164            result.append(s.substring(start, i));
165            // Are we done?
166            if (i >= length) {
167                return result.toString();
168            } else if (s.charAt(i) == ' ') {
169                result.append('+'); // Replace space char with plus symbol.
170                i++;
171            } else {
172                // Get all unsafe characters
173                start = i;
174                char c;
175                while ((i < length) && ((c = s.charAt(i)) != ' ') && !isSafe(c)) {
176                    i++;
177                }
178                // Convert them to %XY encoded strings
179                String unsafe = s.substring(start, i);
180                byte[] bytes = unsafe.getBytes(encoding);
181                for (byte aByte : bytes) {
182                    result.append('%');
183                    result.append(hex.charAt(((int) aByte & 0xf0) >> 4));
184                    result.append(hex.charAt((int) aByte & 0x0f));
185                }
186            }
187            start = i;
188        }
189    }

191    /**
192     * This method takes a String of an URI with an unescaped query
193     * string and converts it into a URI with encoded query string format.
194     * Useful for processing command line input.
195     *
196     * @param s the URI string
197     * @return a string with the URL encoded data.
198     * @throws java.net.URISyntaxException
199     */
200    public static URI encodeQueryString(String s) throws URISyntaxException {
201        if (s == null) {
202            return null;
203        }
204        StringBuilder out = new StringBuilder();
205        int questionmark = s.indexOf('?');
206        if (questionmark > 0) {
207            StringTokenizer st = new StringTokenizer(s.substring(questionmark + 1), "&");
208            while (st.hasMoreTokens()) {
209                String pair = st.nextToken();
210                int pos = pair.indexOf('=');
211                if (pos == -1) {
212                    throw new URISyntaxException(s, "missing '='");
213                }
214                if (out.length() > 0) {
215                    out.append("&");
216                }
217                out.append(pair.substring(0, pos + 1)).append(encode(pair.substring(pos + 1), UTF8));
218            }
219            return new URI(s.substring(0, questionmark + 1) + out.toString());
220        } else {
221            return new URI(s);
222        }
223    }

225    /**
226     * Get properties from URI
227     *
228     * @param uri the URI
229     * @return the properties
230     */
231    public static Properties getPropertiesFromURI(URI uri) {
232        if (uri == null) {
233            throw new IllegalArgumentException("uri must not be null");
234        }
235        Properties properties = new Properties();
236        properties.setProperty("uri", uri.toString());
237        String scheme = uri.getScheme();
238        // ensure scheme is not null
239        scheme = scheme != null ? scheme : "";
240        properties.setProperty("scheme", scheme);
241        String type;
242        if (scheme.startsWith("jdbc:")) {
243            int pos = scheme.substring(5).indexOf(':');
244            type = (pos > 0) ? scheme.substring(5).substring(0, pos) : "default";
245            properties.setProperty("type", type);
246        }
247        String host = uri.getHost();
248        host = (host != null) ? host : "";
249        int port = uri.getPort();
250        String path = uri.getPath();
251        String[] s = (path != null) ? path.split("/") : new String[]{};
252        String cluster = (s.length > 1) ? s[1] : "";
253        String collection = (s.length > 2) ? s[2] : "";
254        String userInfo = uri.getUserInfo();
255        String[] ui = (userInfo != null) ? userInfo.split(":") : new String[]{};
256        properties.setProperty("username", (ui.length > 0) ? ui[0] : "");
257        properties.setProperty("password", (ui.length > 1) ? ui[1] : "");
258        Map<String, String> m = parseQueryString(uri);
259        if ((m != null) && (m.size() > 0)) {
260            properties.setProperty("requestkeys", concat(m.keySet()));
261            properties.setProperty("requestquery", renderQueryString(m));
262            properties.putAll(m);
263        }
264        properties.setProperty("host", host);
265        if (port != 0) {
266            properties.setProperty("port", Integer.toString(port));
267        }
268        if (path != null) {
269            properties.setProperty("path", path);
270        }
271        if (uri.getFragment() != null) {
272            properties.setProperty("fragment", uri.getFragment());
273        }
274        if (cluster != null) {
275            properties.setProperty("cluster", cluster);
276            properties.setProperty("index", cluster);
277        }
278        if (collection != null) {
279            properties.setProperty("collection", collection);
280            properties.setProperty("type", collection);
281        }
282        return properties;
283    }

285    /**
286     * Returns true if the given char is
287     * either a uppercase or lowercase letter from 'a' till 'z', or a digit
288     * froim '0' till '9', or one of the characters '-', '_', '.' or ''. Such
289     * 'safe' character don't have to be url encoded.
290     *
291     * @param c the character
292     * @return true or false
293     */
294    private static boolean isSafe(char c) {
295        return (((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z'))
296                || ((c >= '0') && (c <= '9')) || (c == '-') || (c == '_') || (c == '.') || (c == '*'));
297    }

324    /**
325     * This method parses a query string and returns a map of decoded
326     * request parameters. We do not rely on java.net.URI because it does not
327     * decode plus characters. A listener can process the parameters in order.
328     *
329     * @param uri      the URI to examine for request parameters
330     * @param encoding the encoding
331     * @param listener a listner for processing the URI parameters in order, or null
332     * @return a Map of parameters
333     */
334    public static Map<String, String> parseQueryString(URI uri, Charset encoding, ParameterListener listener) {
335        Map<String, String> m = new HashMap<String, String>();
336        if (uri == null) {
337            throw new IllegalArgumentException();
338        }
339        if (uri.getRawQuery() == null) {
340            return m;
341        }
342        // we use getRawQuery because we do our decoding by ourselves
343        StringTokenizer st = new StringTokenizer(uri.getRawQuery(), "&");
344        while (st.hasMoreTokens()) {
345            String pair = st.nextToken();
346            String k;
347            String v;
348            int pos = pair.indexOf('=');
349            if (pos < 0) {
350                k = pair;
351                v = null;
352            } else {
353                k = pair.substring(0, pos);
354                v = decode(pair.substring(pos + 1, pair.length()), encoding);
355            }
356            m.put(k, v);
357            if (listener != null) {
358                listener.received(k, v);
359            }
360        }
361        return m;
362    }

364    /**
365     * This method takes a Map of key/value elements and converts it
366     * into a URL encoded querystring format.
367     *
368     * @param m a map of key/value arrays
369     * @return a string with the URL encoded data
370     */
371    public static String renderQueryString(Map<String, String> m) {
372        String key;
373        String value;
374        StringBuilder out = new StringBuilder();
375        Charset encoding = m.containsKey("encoding") ? Charset.forName(m.get("encoding")) : UTF8;
376        for (Map.Entry<String, String> me : m.entrySet()) {
377            key = me.getKey();
378            String o = me.getValue();
379            value = o != null ? encode(o, encoding) : null;
380            if (key != null) {
381                if (out.length() > 0) {
382                    out.append("&");
383                }
384                out.append(key);
385                if ((value != null) && (value.length() > 0)) {
386                    out.append("=").append(value);
387                }
388            }
389        }
390        return out.toString();
391    }

393    /**
394     * This method takes a Map of key/value elements and generates a
395     * string for queries.
396     *
397     * @param m a map of key/value arrays.
398     * @return a string
399     */
400    public static String renderRawQueryString(Map<String, String> m) {
401        String key;
402        String value;
403        StringBuilder out = new StringBuilder();
404        for (Map.Entry<String, String> me : m.entrySet()) {
405            key = me.getKey();
406            value = me.getValue();
407            if ((key != null) && (value != null) && (value.length() > 0)) {
408                if (out.length() > 0) {
409                    out.append("&");
410                }
411                out.append(key).append("=").append(value);
412            }
413        }
414        return out.toString();
415    }
