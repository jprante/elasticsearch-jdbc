002    package org.xbib.io.keyvalue;
003    
004    import java.io.IOException;
005    import java.util.List;
006    
007    public interface KeyValueStreamListener<K,V> {
008    
009        /**
010         * Begin a key/value sequence
011         *
012         * @return this value listener
013         * @throws java.io.IOException
014         */
015        KeyValueStreamListener<K,V> begin() throws IOException;
016    
017        /**
018         * Receive key/value pair
019         * @param entity a key/value pair
020         */
021        KeyValueStreamListener<K,V> keyValue(KeyValue<K,V> entity) throws IOException;
022    
023        /**
024         * Receive key/value pair
025         * @param key the key
026         * @param value the value
027         */
028        KeyValueStreamListener<K,V> keyValue(K key, V value) throws IOException;
029    
030        /**
031         * Declare the keys for the values
032         *
033         * @param keys the keys
034         * @return this ValueListener
035         */
036        KeyValueStreamListener<K,V> keys(List<K> keys) throws IOException;
037    
038        /**
039         * Receive values for the declared keys
040         *
041         * @param values the values
042         * @return this ValueListener
043         * @throws java.io.IOException
044         */
045        KeyValueStreamListener<K,V> values(List<V> values) throws IOException;
046    
047        /**
048         * End a key/value sequence
049         *
050         * @return this listener
051         * @throws java.io.IOException
052         */
053        KeyValueStreamListener<K,V> end() throws IOException;
054    
055    }
