002package org.xbib.io.keyvalue;
003
004/**
005 * This class represents two entities with an ordered pair relationship.
006 * The first entity is the key and the second is the value.
007 *
008 * @param <K> the key type
009 * @param <V> the value type
010 */
011public class KeyValue<K,V> {
012
013    private final K key;
014
015    private final V value;
016
017    public KeyValue(K key, V value) {
018        this.key = key;
019        this.value = value;
020    }
021
022    public K key() {
023        return key;
024    }
025
026    public V value() {
027        return value;
028    }
029
030    public String toString() {
031        return key + " => " + value;
032    }
033}
