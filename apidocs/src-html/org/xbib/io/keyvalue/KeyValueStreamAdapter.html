002    package org.xbib.io.keyvalue;
003    
004    import java.io.IOException;
005    import java.util.List;
006    
007    public class KeyValueStreamAdapter<K,V> implements KeyValueStreamListener<K,V> {
008    
009        /**
010         * Begin a key/value sequence
011         *
012         * @return this value listener
013         * @throws java.io.IOException
014         */
015        public KeyValueStreamListener<K,V> begin() throws IOException {
016            return this;
017        }
018    
019        /**
020         * Receive key/value pair
021         * @param keyValue the key/value pair
022         */
023        public KeyValueStreamListener<K,V> keyValue(KeyValue<K,V> keyValue) throws IOException {
024            return this;
025        }
026    
027        /**
028         * Receive key/value pair
029         * @param key the key
030         * @param value the value
031         */
032        public KeyValueStreamListener<K,V> keyValue(K key, V value) throws IOException {
033            return this;
034        }
035    
036        /**
037         * Declare the keys for the values
038         *
039         * @param keys the keys
040         * @return this ValueListener
041         */
042        public KeyValueStreamListener<K,V> keys(List<K> keys) throws IOException {
043            return this;
044        }
045    
046        /**
047         * Receive values for the declared keys
048         *
049         * @param values the values
050         * @return this ValueListener
051         * @throws java.io.IOException
052         */
053        public KeyValueStreamListener<K,V> values(List<V> values) throws IOException {
054            return this;
055        }
056    
057        /**
058         * End a key/value sequence
059         *
060         * @return this listener
061         * @throws java.io.IOException
062         */
063        public KeyValueStreamListener<K,V> end() throws IOException {
064            return this;
065        }
066    }
