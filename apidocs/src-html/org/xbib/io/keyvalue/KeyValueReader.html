002package org.xbib.io.keyvalue;
003
004import java.io.BufferedReader;
005import java.io.IOException;
006import java.io.Reader;
007import java.util.List;
008
009/**
010 * Read text lines with a key/value pair at each line.
011 */
012public class KeyValueReader extends BufferedReader
013        implements KeyValueStreamListener<String, String> {
014
015    private KeyValueStreamListener<String, String> listener;
016
017    private boolean atStart;
018
019    private boolean atEnd;
020
021    private final char delimiter;
022
023    public KeyValueReader(Reader reader) throws IOException {
024        this(reader, '=');
025    }
026    
027    public KeyValueReader(Reader reader, char delimiter) throws IOException {
028        super(reader);
029        this.atStart = true;
030        this.atEnd = false;
031        this.delimiter = delimiter;
032    }
033
034    public KeyValueReader addListener(KeyValueStreamListener<String, String> listener) {
035        this.listener = listener;
036        return this;
037    }
038
039    @Override
040    public int read() throws IOException {
041        if (atStart) {
042            atStart = false;
043            begin();
044        }
045        return super.read();
046    }
047
048    @Override
049    public int read(char[] buf, int off, int len) throws IOException {
050        if (atStart) {
051            atStart = false;
052            begin();
053        }
054        return super.read(buf, off, len);
055    }
056
057    @Override
058    public String readLine() throws IOException {
059        if (atStart) {
060            atStart = false;
061            begin();
062        }
063        String line = super.readLine();
064        if (line != null) {
065            int pos = line.indexOf(delimiter);
066            if (pos > 0) {
067                keyValue(line.substring(0, pos), line.substring(pos+1));
068            }
069        }
070        return line;
071    }
072
073    @Override
074    public void close() throws IOException {
075        if (!atEnd) {
076            atEnd = true;
077            end();
078        }
079        super.close();
080    }
081
082    @Override
083    public KeyValueReader begin() throws IOException {
084        if (listener != null) {
085            listener.begin();
086        }
087        return this;
088    }
089
090    @Override
091    public KeyValueReader keyValue(String key, String value) throws IOException {
092        if (listener != null) {
093            listener.keyValue(key, value);
094        }
095        return this;
096    }
097
098    @Override
099    public KeyValueReader keyValue(KeyValue<String,String> keyValue) throws IOException {
100        if (listener != null) {
101            listener.keyValue(keyValue);
102        }
103        return this;
104    }
105
106    @Override
107    public KeyValueStreamListener<String, String> keys(List<String> keys) throws IOException {
108        if (listener != null) {
109            listener.keys(keys);
110        }
111        return this;
112    }
113
114    @Override
115    public KeyValueStreamListener<String, String> values(List<String> values) throws IOException {
116        if (listener != null) {
117            listener.values(values);
118        }
119        return this;
120    }
121
122    @Override
123    public KeyValueReader end() throws IOException {
124        if (listener != null) {
125            listener.end();
126        }
127        return this;
128    }
129
130}
