123        public void save(Client client) throws IOException {
124            XContentBuilder builder = jsonBuilder();
125            builder = toXContent(builder, ToXContent.EMPTY_PARAMS);
126            logger.debug("save state={}",builder.string());
127            client.index(indexRequest().index(index).type(type).id(id)
128                    .source(builder.string()))
129                    .actionGet();
130        }
131    
132        public void load(Client client) throws IOException {
133            GetResponse get = null;
134            try {
135                client.admin().indices().prepareRefresh(index).execute().actionGet();
136                get = client.prepareGet(index, type, id).execute().actionGet();
137            } catch (IndexMissingException e) {
138                logger.warn("river state missing: {}/{}/{}", index, type, id);
139            }
140            if (get != null && get.isExists()) {
141                logger.debug("load state={}", get.getSourceAsString());
142                XContentParser parser = XContentFactory.xContent(XContentType.JSON)
143                        .createParser(get.getSourceAsBytes());
144                fromXContent(parser);
145            } else {
146                counter = 0L;
147            }
148        }
149    
150        public void fromXContent(XContentParser parser) throws IOException {
151            DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"), TimeUnit.MILLISECONDS);
152            String currentFieldName = null;
153            XContentParser.Token token;
154            while ((token = parser.nextToken()) != END_OBJECT) {
155                if (token == null) {
156                    break;
157                } else if (token == FIELD_NAME) {
158                    currentFieldName = parser.currentName();
159                } else if (token == VALUE_NULL || token == VALUE_STRING
160                        || token == VALUE_BOOLEAN || token == VALUE_NUMBER) {
161                    if ("name".equals(currentFieldName)) {
162                        name(parser.text());
163                    } else if ("started".equals(currentFieldName)) {
164                        started(dateParser.parse(parser.text(), 0));
165                    } else if ("timestamp".equals(currentFieldName)) {
166                        timestamp(dateParser.parse(parser.text(), 0));
167                    } else if ("counter".equals(currentFieldName)) {
168                        counter(parser.longValue());
169                    } else if ("active".equals(currentFieldName)) {
170                        active(parser.booleanValue());
171                    }
172                } else if (token == START_OBJECT) {
173                    if ("custom".equals(currentFieldName)) {
174                        custom(parser.map());
175                    } // else skip
176                }
177            }
178        }
179    
180        @Override
181        public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
182            builder.startObject()
183                    .field("name", name)
184                    .field("started", new Date(started))
185                    .field("timestamp", new Date(timestamp))
186                    .field("counter", counter)
187                    .field("active", active)
188                    .field("custom", custom)
189                .endObject();
190            return builder;
191        }
