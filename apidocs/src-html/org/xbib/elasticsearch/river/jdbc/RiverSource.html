001package org.xbib.elasticsearch.river.jdbc;
002
003import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
004import org.xbib.keyvalue.KeyValueStreamListener;
005
006import java.io.IOException;
007import java.sql.CallableStatement;
008import java.sql.Connection;
009import java.sql.PreparedStatement;
010import java.sql.ResultSet;
011import java.sql.SQLException;
012import java.sql.Statement;
013import java.text.ParseException;
014import java.util.List;
015import java.util.Locale;
016import java.util.Map;
017
018/**
019 * The river source models the data producing side
020 */
021public interface RiverSource {
022
023    /**
024     * The strategy this river source supports.
025     *
026     * @return the strategy as a string
027     */
028    String strategy();
029
030    /**
031     * Set the river context
032     *
033     * @param context the context
034     * @return this river source
035     */
036    RiverSource setRiverContext(RiverContext context);
037
038    /**
039     * Fetch a data portion from the database and pass it to the river task
040     * for firther processing.
041     *
042     * @throws SQLException when SQL execution gives an error
043     * @throws IOException when input/output error occurs
044     */
045    void fetch() throws SQLException, IOException;
046
047    /**
048     * Set the driver URL
049     *
050     * @param url the JDBC URL
051     * @return this river source
052     */
053    RiverSource setUrl(String url);
054
055    /**
056     * Set the user authentication
057     *
058     * @param user the user
059     * @return this river source
060     */
061    RiverSource setUser(String user);
062
063    /**
064     * Set the password authentication
065     *
066     * @param password the password
067     * @return this river source
068     */
069    RiverSource setPassword(String password);
070
071
072    /**
073     * Get a connection for reading data
074     *
075     * @return connection
076     * @throws SQLException when SQL execution gives an error
077     */
078    Connection getConnectionForReading() throws SQLException;
079
080    /**
081     * Get a connection for writing data
082     *
083     * @return connection
084     * @throws SQLException when SQL execution gives an error
085     */
086    Connection getConnectionForWriting() throws SQLException;
087
088    /**
089     * Prepare query statement
090     *
091     * @param sql SQL statement
092     * @return a prepared statement
093     * @throws SQLException when SQL execution gives an error
094     */
095    PreparedStatement prepareQuery(String sql) throws SQLException;
096
097    /**
098     * Prepare insert/update statement
099     *
100     * @param sql SQL statement
101     * @return a prepared statement
102     * @throws SQLException when SQL execution gives an error
103     */
104    PreparedStatement prepareUpdate(String sql) throws SQLException;
105
106    /**
107     * Bind query variables
108     *
109     * @param statement prepared statement
110     * @param values    values
111     * @return this river source
112     * @throws SQLException when SQL execution gives an error
113     */
114    RiverSource bind(PreparedStatement statement, List<Object> values) throws SQLException;
115
116    /**
117     * Register output variables for callable statement
118     *
119     * @param statement callable statement
120     * @param values    values
121     * @return this river source
122     * @throws SQLException when SQL execution gives an error
123     */
124    RiverSource register(CallableStatement statement, Map<String, Object> values) throws SQLException;
125
126    /**
127     * Execute query
128     *
129     * @param statement prepared statement
130     * @return the result set
131     * @throws SQLException when SQL execution gives an error
132     */
133    ResultSet executeQuery(PreparedStatement statement) throws SQLException;
134
135    /**
136     * Execute query without binding parameters
137     *
138     * @param statement the SQL statement
139     * @param sql       the SQL query
140     * @return the result set
141     * @throws SQLException when SQL execution gives an error
142     */
143    ResultSet executeQuery(Statement statement, String sql) throws SQLException;
144
145    /**
146     * Execute insert/update
147     *
148     * @param statement statement
149     * @return this river source
150     * @throws SQLException when SQL execution gives an error
151     */
152    RiverSource executeUpdate(PreparedStatement statement) throws SQLException;
153
154    RiverSource executeUpdate(Statement statement, String sql) throws SQLException;
155
156    void beforeRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
157
158    /**
159     * Action for the next row of the result set to be processed
160     *
161     * @param results  result
162     * @param listener listener
163     * @return true if next row exists
164     * @throws SQLException when SQL execution gives an error
165     * @throws IOException when input/output error occurs
166     */
167    boolean nextRow(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
168
169    void afterRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
170
171    /**
172     * Parse a value in a row column
173     *
174     * @param results result set
175     * @param num     position
176     * @param type    type
177     * @param locale  locale
178     * @return object
179     * @throws SQLException when SQL execution gives an error
180     * @throws IOException when input/output error occurs
181     * @throws ParseException if number format could not be parsed
182     */
183    Object parseType(ResultSet results, Integer num, int type, Locale locale) throws SQLException, IOException, ParseException;
184
185    /**
186     * Close result set
187     *
188     * @param result result set
189     * @return this river source
190     * @throws SQLException when SQL execution gives an error
191     */
192    RiverSource close(ResultSet result) throws SQLException;
193
194    /**
195     * Close statement
196     *
197     * @param statement statement
198     * @return this river source
199     * @throws SQLException when SQL execution gives an error
200     */
201    RiverSource close(Statement statement) throws SQLException;
202
203    /**
204     * Close reading from this river source
205     *
206     * @return this river source
207     */
208    RiverSource closeReading();
209
210    /**
211     * Close writing to this river source
212     *
213     * @return this river source
214     */
215    RiverSource closeWriting();
216}
