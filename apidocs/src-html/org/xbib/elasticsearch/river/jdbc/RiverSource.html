001    
002    package org.xbib.elasticsearch.river.jdbc;
003    
004    import java.io.IOException;
005    import java.sql.CallableStatement;
006    import java.sql.Connection;
007    import java.sql.PreparedStatement;
008    import java.sql.ResultSet;
009    import java.sql.SQLException;
010    import java.sql.Statement;
011    import java.text.ParseException;
012    import java.util.List;
013    import java.util.Locale;
014    import java.util.Map;
015    
016    import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
017    import org.xbib.io.keyvalue.KeyValueStreamListener;
018    
019    /**
020     * The river source models the data producing side
021     */
022    public interface RiverSource {
023    
024        /**
025         * The strategy this river source supports.
026         *
027         * @return the strategy as a string
028         */
029        String strategy();
030    
031        /**
032         * Set the river context
033         *
034         * @param context the context
035         * @return this river source
036         */
037        RiverSource riverContext(RiverContext context);
038    
039        /**
040         * Fetch a data portion from the database and pass it to the river task
041         * for firther processing.
042         *
043         * @throws SQLException
044         * @throws IOException
045         */
046        void fetch() throws SQLException, IOException;
047    
048        /**
049         * Set the driver URL
050         *
051         * @param url the JDBC URL
052         * @return this river source
053         */
054        RiverSource url(String url);
055    
056        /**
057         * Set the user authentication
058         *
059         * @param user the user
060         * @return this river source
061         */
062        RiverSource user(String user);
063    
064        /**
065         * Set the password authentication
066         *
067         * @param password the password
068         * @return this river source
069         */
070        RiverSource password(String password);
071    
072        /**
073         * Set rounding for transporting java.math.BigDecimal
074         *
075         * @param rounding rounding
076         * @return this river source
077         */
078        RiverSource rounding(String rounding);
079    
080        /**
081         * Set scale precision for transporting java.math.BigDecimal
082         *
083         * @param scale scale
084         * @return this river source
085         */
086        RiverSource precision(int scale);
087    
088        /**
089         * Get a connection for reading data
090         *
091         * @return connection
092         * @throws SQLException
093         */
094        Connection connectionForReading() throws SQLException;
095    
096        /**
097         * Get a connection for writing data
098         *
099         * @return connection
100         * @throws SQLException
101         */
102        Connection connectionForWriting() throws SQLException;
103    
104        /**
105         * Prepare query statement
106         *
107         * @param sql SQL statement
108         * @return a prepared statement
109         * @throws SQLException
110         */
111        PreparedStatement prepareQuery(String sql) throws SQLException;
112    
113        /**
114         * Prepare insert/update statement
115         *
116         * @param sql SQL statement
117         * @return a prepared statement
118         * @throws SQLException
119         */
120        PreparedStatement prepareUpdate(String sql) throws SQLException;
121    
122        /**
123         * Bind query variables
124         *
125         * @param statement prepared statement
126         * @param values values
127         * @return this river source
128         * @throws SQLException
129         */
130        RiverSource bind(PreparedStatement statement, List<? extends Object> values) throws SQLException;
131    
132        /**
133         * Register output variables for callable statement
134         *
135         * @param statement callable statement
136         * @param values values
137         * @return this river source
138         * @throws SQLException
139         */
140        RiverSource register(CallableStatement statement, Map<String, Object> values) throws SQLException;
141    
142        /**
143         * Execute query
144         *
145         * @param statement prepared statement
146         * @return the result set
147         * @throws SQLException
148         */
149        ResultSet executeQuery(PreparedStatement statement) throws SQLException;
150    
151        /**
152         * Execute query without binding parameters
153         *
154         * @param statement the SQL statement
155         * @param sql       the SQL query
156         * @return the result set
157         * @throws SQLException
158         */
159        ResultSet executeQuery(Statement statement, String sql) throws SQLException;
160    
161        /**
162         * Execute insert/update
163         *
164         * @param statement statement
165         * @return this river source
166         * @throws SQLException
167         */
168        RiverSource executeUpdate(PreparedStatement statement) throws SQLException;
169    
170        RiverSource executeUpdate(Statement statement, String sql) throws SQLException;
171    
172        void beforeRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
173    
174        /**
175         * Action for the next row of the result set to be processed
176         *
177         * @param results result
178         * @param listener listener
179         * @return true if next row exists
180         * @throws SQLException
181         * @throws IOException
182         * @throws ParseException
183         */
184        boolean nextRow(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException, ParseException;
185    
186        void afterRows(ResultSet results, KeyValueStreamListener listener) throws SQLException, IOException;
187    
188        /**
189         * Parse a value in a row column
190         *
191         * @param results result set
192         * @param num position
193         * @param type type
194         * @param locale locale
195         * @return object
196         * @throws SQLException
197         * @throws IOException
198         * @throws ParseException
199         */
200        Object parseType(ResultSet results, Integer num, int type, Locale locale) throws SQLException, IOException, ParseException;
201    
202        /**
203         * Close result set
204         *
205         * @param result result set
206         * @return this river source
207         * @throws SQLException
208         */
209        RiverSource close(ResultSet result) throws SQLException;
210    
211        /**
212         * Close statement
213         *
214         * @param statement statement
215         * @return this river source
216         * @throws SQLException
217         */
218        RiverSource close(Statement statement) throws SQLException;
219    
220        /**
221         * Close reading from this river source
222         *
223         * @return this river source
224         */
225        RiverSource closeReading();
226    
227        /**
228         * Close writing to this river source
229         *
230         * @return this river source
231         */
232        RiverSource closeWriting();
233    }
