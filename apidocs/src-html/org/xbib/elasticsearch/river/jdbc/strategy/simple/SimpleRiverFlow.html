002package org.xbib.elasticsearch.river.jdbc.strategy.simple;
003
004import org.xbib.elasticsearch.plugin.feeder.jdbc.JDBCFeeder;
005import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
006import org.xbib.elasticsearch.river.jdbc.RiverFlow;
007
008public class SimpleRiverFlow  implements RiverFlow {
009
010    protected RiverContext context;
011
012    protected JDBCFeeder feeder;
013
014    @Override
015    public String strategy() {
016        return "simple";
017    }
018
019    @Override
020    public RiverFlow setRiverContext(RiverContext context) {
021        this.context = context;
022        return this;
023    }
024
025    @Override
026    public RiverFlow setFeeder(JDBCFeeder feeder) {
027        this.feeder = feeder;
028        return this;
029    }
030
031    @Override
032    public JDBCFeeder getFeeder() {
033        if (feeder == null) {
034            this.feeder = new JDBCFeeder();
035        }
036        return feeder;
037    }
038}
