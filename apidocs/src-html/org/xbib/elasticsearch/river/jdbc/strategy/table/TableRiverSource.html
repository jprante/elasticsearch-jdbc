019package org.xbib.elasticsearch.river.jdbc.strategy.table;

021import java.io.IOException;
022import java.sql.Connection;
023import java.sql.PreparedStatement;
024import java.sql.ResultSet;
025import java.sql.SQLException;
026import java.sql.Timestamp;
027import java.util.ArrayList;
028import java.util.List;
029
030import org.elasticsearch.action.bulk.BulkItemResponse;
031import org.elasticsearch.action.bulk.BulkResponse;
032import org.elasticsearch.common.logging.ESLogger;
033import org.elasticsearch.common.logging.ESLoggerFactory;
034import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
035import org.xbib.elasticsearch.river.jdbc.support.Operations;
036import org.xbib.elasticsearch.river.jdbc.support.ValueListener;
037import org.xbib.elasticsearch.river.jdbc.support.Operations;
038import org.xbib.elasticsearch.river.jdbc.support.ValueListener;
039
040/**
041 * River source implementation of the 'table' strategy
042 *
043 * @author JÃ¶rg Prante <joergprante@gmail.com>
044 */
045public class TableRiverSource extends SimpleRiverSource {
046
047    private final ESLogger logger = ESLoggerFactory.getLogger(TableRiverSource.class.getName());
048
049    @Override
050    public String strategy() {
051        return "table";
052    }
053
054    @Override
055    public String fetch() throws SQLException, IOException {
056        Connection connection = connectionForReading();
057        String[] optypes = new String[]{Operations.OP_CREATE, Operations.OP_INDEX, Operations.OP_DELETE};
058        for (String optype : optypes) {
059            PreparedStatement statement;
060            try {
061                statement = connection.prepareStatement("select * from \"" + context.riverName() + "\" where \"source_operation\" = ? and \"source_timestamp\" between ? and ?");
062            } catch (SQLException e) {
063                // hsqldb
064                statement = connection.prepareStatement("select * from " + context.riverName() + " where \"source_operation\" = ? and \"source_timestamp\" between ? and ?");
065            }
066            statement.setString(1, optype);
067            java.util.Date d = new java.util.Date();
068            long now = d.getTime();
069            statement.setTimestamp(2, new Timestamp(now - context.pollingInterval().millis()));
070            statement.setTimestamp(3, new Timestamp(now));
071            ResultSet results;
072            try {
073                results = executeQuery(statement);
074            } catch (SQLException e) {
075                // mysql
076                statement = connection.prepareStatement("select * from " + context.riverName() + " where source_operation = ? and source_timestamp between ? and ?");
077                statement.setString(1, optype);
078                statement.setTimestamp(2, new Timestamp(now - context.pollingInterval().millis()));
079                statement.setTimestamp(3, new Timestamp(now));
080                results = executeQuery(statement);
081            }
082            try {
083                ValueListener listener = new TableValueListener()
084                        .target(context.riverMouth())
085                        .digest(context.digesting());
086                merge(results, listener); // ignore digest
087            } catch (Exception e) {
088                throw new IOException(e);
089            }
090            close(results);
091            close(statement);
092            acknowledge();
093        }
094        return null;
095    }
096
097    /**
098     * Acknowledge a bulk item response back to the river table. Fill columns
099     * target_timestamp, taget_operation, target_failed, target_message.
100     *
101     * @param response
102     * @throws IOException
103     */
104    @Override
105    public SimpleRiverSource acknowledge(BulkResponse response) throws IOException {
106        if (response == null) {
107            logger.warn("can't acknowledge null bulk response");
108        }
109        try {
110            Connection connection = connectionForWriting();
111            String riverName = context.riverName();
112            for (BulkItemResponse resp : response.getItems()) {
113                PreparedStatement pstmt;
114                try {
115                    pstmt = prepareUpdate("update \"" + riverName + "\" set \"source_operation\" = 'ack' where \"_index\" = ? and \"_type\" = ? and \"_id\" = ?");
116                } catch (SQLException e) {
117                    try {
118                        // hsqldb
119                        pstmt = prepareUpdate("update " + riverName + " set \"source_operation\" = 'ack' where \"_index\" = ? and \"_type\" = ? and \"_id\" = ?");
120                    } catch (SQLException e1) {
121                        // mysql
122                        pstmt = prepareUpdate("update " + riverName + " set source_operation = 'ack' where _index = ? and _type = ? and _id = ?");
123                    }
124                }
125                List<Object> params = new ArrayList();
126                params.add(resp.getIndex());
127                params.add(resp.getType());
128                params.add(resp.getId());
129                bind(pstmt, params);
130                executeUpdate(pstmt);
131                close(pstmt);
132                try {
133                    pstmt = prepareUpdate("update \"" + riverName + "_ack\" set \"target_timestamp\" = ?, \"target_operation\" = ?, \"target_failed\" = ?, \"target_message\" = ? where \"_index\" = ? and \"_type\" = ? and \"_id\" = ?");
134                } catch (SQLException e) {
135                    try {
136                        // hsqldb
137                        pstmt = prepareUpdate("update " + riverName + "_ack set \"target_timestamp\" = ?, \"target_operation\" = ?, \"target_failed\" = ?, \"target_message\" = ? where \"_index\" = ? and \"_type\" = ? and \"_id\" = ?");
138                    } catch (SQLException e1) {
139                        // mysql
140                        pstmt = prepareUpdate("update " + riverName + "_ack set target_timestamp = ?, target_operation = ?, target_failed = ?, target_message = ? where _index = ? and _type = ? and _id = ?");
141                    }
142                }
143                params = new ArrayList();
144                params.add(new Timestamp(new java.util.Date().getTime()));
145                params.add(resp.getOpType());
146                params.add(resp.isFailed());
147                params.add(resp.getFailureMessage());
148                params.add(resp.getIndex());
149                params.add(resp.getType());
150                params.add(resp.getId());
151                bind(pstmt, params);
152                executeUpdate(pstmt);
153                close(pstmt);
154            }
155        } catch (SQLException ex) {
156            throw new IOException(ex);
157        }
158        return this;
159    }
160}
