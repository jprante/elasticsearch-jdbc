139    /**
140     * Get JDBC connection for reading
141     *
142     * @return the connection
143     * @throws SQLException
144     */
145    @Override
146    public Connection connectionForReading() throws SQLException {
147        boolean cond = readConnection == null || readConnection.isClosed();
148        try {
149            cond = cond || !readConnection.isValid(5);
150        } catch (AbstractMethodError e) {
151            // old/buggy JDBC driver
152            logger().debug(e.getMessage());
153        } catch (SQLFeatureNotSupportedException e) {
154            // postgresql does not support isValid()
155            logger().debug(e.getMessage());
156        }
157        if (cond) {
158            int retries = context != null ? context.retries() : 1;
159            while (retries > 0) {
160                retries--;
161                try {
162                    readConnection = DriverManager.getConnection(url, user, password);
163                    // required by MySQL for large result streaming
164                    readConnection.setReadOnly(true);
165                    // Postgresql cursor mode condition:
166                    // fetchsize > 0, no scrollable result set, no auto commit, no holdable cursors over commit
167                    // https://github.com/pgjdbc/pgjdbc/blob/master/org/postgresql/jdbc2/AbstractJdbc2Statement.java#L514
168                    //readConnection.setHoldability(ResultSet.HOLD_CURSORS_OVER_COMMIT);
169                    if (context != null) {
170                        // many drivers don't like autocommit=true
171                        readConnection.setAutoCommit(context.autocommit());
172                    }
173                    return readConnection;
174                } catch (SQLException e) {
175                    logger().error("while opening read connection: " + url + " " + e.getMessage(), e);
176                    try {
177                        Thread.sleep(context != null ? context.maxRetryWait().millis() : 1000L);
178                    } catch (InterruptedException ex) {
179                        // do nothing
180                    }
181                }
182            }
183        }
184        return readConnection;
185    }

193    @Override
194    public Connection connectionForWriting() throws SQLException {
195        boolean cond = writeConnection == null || writeConnection.isClosed();
196        try {
197            cond = cond || !writeConnection.isValid(5);
198        } catch (AbstractMethodError e) {
199            // old JDBC driver
200        } catch (SQLFeatureNotSupportedException e) {
201            // postgresql does not support isValid()
202        }
203        if (cond) {
204            int retries = context != null ? context.retries() : 1;
205            while (retries > 0) {
206                retries--;
207                try {
208                    writeConnection = DriverManager.getConnection(url, user, password);
209                    if (context != null) {
210                        // many drivers don't like autocommit=true
211                        writeConnection.setAutoCommit(context.autocommit());
212                    }
213                    return writeConnection;
214                } catch (SQLNonTransientConnectionException e) {
215                    // ignore derby drop=true
216                } catch (SQLException e) {
217                    logger().error("while opening write connection: " + url + " " + e.getMessage(), e);
218                    try {
219                        Thread.sleep(context != null ? context.maxRetryWait().millis() : 1000L);
220                    } catch (InterruptedException ex) {
221                        // do nothing
222                    }
223                }
224            }
225        }
226        return writeConnection;
227    }

229    @Override
230    public String fetch() throws SQLException, IOException {
231        String mergeDigest = null;
232        if (context.pollStatementParams().isEmpty()) {
233            Statement statement = null;
234            ResultSet results = null;
235            try {
236                // Postgresql: do not use prepareStatement.
237                // Postgresql requires direct use of executeQuery(sql) for cursor with fetchsize
238                statement = connectionForReading().createStatement();
239                results = executeQuery(statement, getSql());
240                ValueListener listener = new SimpleValueListener()
241                        .target(context.riverMouth())
242                        .digest(context.digesting());
243                mergeDigest = merge(results, listener);
244            } catch (Exception e) {
245                throw new IOException(e);
246            } finally {
247                close(results);
248                close(statement);
249                acknowledge();
250                closeReading();
251                closeWriting();
252            }
253        } else if (context.callable()) {
254            // call stored procedure
255            CallableStatement statement = null;
256            ResultSet results = null;
257            try {
258                statement = connectionForReading().prepareCall(getSql());
259                bind(statement, context.pollStatementParams());
260                results = executeQuery(statement);
261                ValueListener listener = new SimpleValueListener()
262                        .target(context.riverMouth())
263                        .digest(context.digesting());
264                mergeDigest = merge(results, listener);
265            } catch (Exception e) {
266                throw new IOException(e);
267            } finally {
268                close(results);
269                close(statement);
270                acknowledge();
271                closeReading();
272                closeWriting();
273            }
274        } else {
275            PreparedStatement statement = null;
276            ResultSet results = null;
277            try {
278                statement = prepareQuery(getSql());
279                bind(statement, context.pollStatementParams());
280                results = executeQuery(statement);
281                ValueListener listener = new SimpleValueListener()
282                        .target(context.riverMouth())
283                        .digest(context.digesting());
284                mergeDigest = merge(results, listener);
285            } catch (Exception e) {
286                throw new IOException(e);
287            } finally {
288                close(results);
289                close(statement);
290                acknowledge();
291                closeReading();
292                closeWriting();
293            }
294        }
295        return mergeDigest;
296    }

298    /**
299     * Merge rows.
300     *
301     * @param results  the ResultSet
302     * @param listener
303     * @return a digest of the merged row content
304     * @throws IOException
305     * @throws java.security.NoSuchAlgorithmException
306     * @throws SQLException
307     */
308    public String merge(ResultSet results, ValueListener listener)
309            throws SQLException, IOException, ParseException, NoSuchAlgorithmException {
310        long rows = 0L;
311        beforeFirstRow(results, listener);
312        while (nextRow(results, listener)) {
313            rows++;
314        }
315        if (rows > 0) {
316            logger().info("merged {} rows", rows);
317        } else {
318            logger().info("no rows to merge");
319        }
320        listener.reset();
321        return context.digesting() && listener.digest() != null
322                ? Base64.encodeBytes(listener.digest().digest()) : null;
323    }

326    /**
327     * Send acknowledge SQL command if exists.
328     *
329     * @throws SQLException
330     */
331    public void acknowledge() throws SQLException {
332        // send acknowledge statement if defined
333        if (context.pollAckStatement() != null) {
334            Connection connection = connectionForWriting();
335            PreparedStatement statement = prepareUpdate(context.pollAckStatement());
336            if (context.pollAckStatementParams() != null) {
337                bind(statement, context.pollAckStatementParams());
338            }
339            statement.execute();
340            close(statement);
341            try {
342                if (!connection.getAutoCommit()) {
343                    connection.commit();
344                }
345            } catch (SQLException e) {
346                //  Can't call commit when autocommit=true
347            }
348            closeWriting();
349        }
350    }

352    private String getSql() throws IOException {
353        String sql = context.pollStatement();
354        if (sql.endsWith(".sql")) {
355            Reader r = new InputStreamReader(new FileInputStream(sql), "UTF-8");
356            sql = Streams.copyToString(r);
357            r.close();
358        }
359        return sql;
360    }

371        Connection connection = connectionForReading();
372        if (connection == null) {
373            throw new SQLException("can't connect to source " + url);
374        }
375        logger().debug("preparing statement with SQL {}", sql);
376        return connection.prepareStatement(sql,
377                ResultSet.TYPE_FORWARD_ONLY,
378                ResultSet.CONCUR_READ_ONLY);

413    @Override
414    public SimpleRiverSource bind(PreparedStatement pstmt, List<? extends Object> values) throws SQLException {
415        if (values == null) {
416            logger().warn("no values given for bind");
417            return this;
418        }
419        for (int i = 1; i <= values.size(); i++) {
420            bind(pstmt, i, values.get(i - 1));
421        }
422        return this;
423    }

472    public void beforeFirstRow(ResultSet result, ValueListener listener)
473            throws SQLException, IOException, ParseException {
474        ResultSetMetaData metadata = result.getMetaData();
475        int columns = metadata.getColumnCount();
476        List<String> keys = new LinkedList();
477        for (int i = 1; i <= columns; i++) {
478            keys.add(metadata.getColumnLabel(i));
479        }
480        if (listener != null) {
481            listener.keys(keys);
482        }
483    }

485    /**
486     * Get next row and prepare the values for processing. The labels of each
487     * columns are used for the ValueListener as paths for JSON object merging.
488     *
489     * @param result   the result set
490     * @param listener the listener
491     * @return true if row exists and was processed, false otherwise
492     * @throws SQLException
493     * @throws IOException
494     */
495    @Override
496    public boolean nextRow(ResultSet result, ValueListener listener)
497            throws SQLException, IOException, ParseException {
498        if (result.next()) {
499            processRow(result, listener);
500            return true;
501        }
502        return false;
503    }
504
505    private void processRow(ResultSet result, ValueListener listener)
506            throws SQLException, IOException, ParseException {
507        Locale locale = context != null ? context.locale() != null ? context.locale() : Locale.getDefault() : Locale.getDefault();
508        List<Object> values = new LinkedList();
509        ResultSetMetaData metadata = result.getMetaData();
510        int columns = metadata.getColumnCount();
511        for (int i = 1; i <= columns; i++) {
512            Object value = parseType(result, i, metadata.getColumnType(i), locale);
513            if (logger().isTraceEnabled()) {
514                logger().trace("value={} class={}", value, value != null ? value.getClass().getName() : "");
515            }
516            values.add(value);
517        }
518        if (listener != null) {
519            listener.values(values);
520        }
521    }

559            if (readConnection != null) {
560                // always commit before close to finish cursors/transactions
561                if (!readConnection.getAutoCommit()) {
562                    readConnection.commit();
563                }
564                if (!readConnection.isClosed()) {
565                    readConnection.close();
566                }

569            logger().warn("while closing read connection: " + e.getMessage());
570        }

603    @Override
604    public SimpleRiverSource rounding(String rounding) {
605        if ("ceiling".equalsIgnoreCase(rounding)) {
606            this.rounding = BigDecimal.ROUND_CEILING;
607        } else if ("down".equalsIgnoreCase(rounding)) {
608            this.rounding = BigDecimal.ROUND_DOWN;
609        } else if ("floor".equalsIgnoreCase(rounding)) {
610            this.rounding = BigDecimal.ROUND_FLOOR;
611        } else if ("halfdown".equalsIgnoreCase(rounding)) {
612            this.rounding = BigDecimal.ROUND_HALF_DOWN;
613        } else if ("halfeven".equalsIgnoreCase(rounding)) {
614            this.rounding = BigDecimal.ROUND_HALF_EVEN;
615        } else if ("halfup".equalsIgnoreCase(rounding)) {
616            this.rounding = BigDecimal.ROUND_HALF_UP;
617        } else if ("unnecessary".equalsIgnoreCase(rounding)) {
618            this.rounding = BigDecimal.ROUND_UNNECESSARY;
619        } else if ("up".equalsIgnoreCase(rounding)) {
620            this.rounding = BigDecimal.ROUND_UP;
621        }
622        return this;
623    }

690    private void bind(PreparedStatement pstmt, int i, Object value) throws SQLException {
691        if (value == null) {
692            pstmt.setNull(i, Types.VARCHAR);
693        } else if (value instanceof String) {
694            String s = (String) value;
695            if ("$now".equals(s)) {
696                pstmt.setDate(i, new Date(new java.util.Date().getTime()));
697            } else if ("$job".equals(s)) {
698                logger().debug("job = {}", context.job());
699                pstmt.setString(i, context.job());
700            } else {
701                pstmt.setString(i, (String) value);
702            }
703        } else if (value instanceof Integer) {
704            pstmt.setInt(i, (Integer) value);
705        } else if (value instanceof Long) {
706            pstmt.setLong(i, (Long) value);
707        } else if (value instanceof BigDecimal) {
708            pstmt.setBigDecimal(i, (BigDecimal) value);
709        } else if (value instanceof Date) {
710            pstmt.setDate(i, (Date) value);
711        } else if (value instanceof Timestamp) {
712            pstmt.setTimestamp(i, (Timestamp) value);
713        } else if (value instanceof Float) {
714            pstmt.setFloat(i, (Float) value);
715        } else if (value instanceof Double) {
716            pstmt.setDouble(i, (Double) value);
717        } else {
718            pstmt.setObject(i, value);
719        }
720    }

722    /**
723     * Parse of value of resultset with the good type
724     *
725     * @param result
726     * @param i
727     * @param type
728     * @param locale
729     * @return The parse value
730     * @throws SQLException
731     * @throws IOException
732     */
733    @Override
734    public Object parseType(ResultSet result, Integer i, int type, Locale locale)
735            throws SQLException, IOException, ParseException {
736        if (logger().isTraceEnabled()) {
737            logger().trace("{} {} {}", i, type, result.getString(i));
738        }
739
740        switch (type) {
741            /**
742             * The JDBC types CHAR, VARCHAR, and LONGVARCHAR are closely
743             * related. CHAR represents a small, fixed-length character string,
744             * VARCHAR represents a small, variable-length character string, and
745             * LONGVARCHAR represents a large, variable-length character string.
746             */
747            case Types.CHAR:
748            case Types.VARCHAR:
749            case Types.LONGVARCHAR: {
750                return result.getString(i);
751            }
752            case Types.NCHAR:
753            case Types.NVARCHAR:
754            case Types.LONGNVARCHAR: {
755                return result.getNString(i);
756            }
757            /**
758             * The JDBC types BINARY, VARBINARY, and LONGVARBINARY are closely
759             * related. BINARY represents a small, fixed-length binary value,
760             * VARBINARY represents a small, variable-length binary value, and
761             * LONGVARBINARY represents a large, variable-length binary value
762             */
763            case Types.BINARY:
764            case Types.VARBINARY:
765            case Types.LONGVARBINARY: {
766                return result.getBytes(i);
767            }
768            /**
769             * The JDBC type ARRAY represents the SQL3 type ARRAY.
770             *
771             * An ARRAY value is mapped to an instance of the Array interface in
772             * the Java programming language. If a driver follows the standard
773             * implementation, an Array object logically points to an ARRAY
774             * value on the server rather than containing the elements of the
775             * ARRAY object, which can greatly increase efficiency. The Array
776             * interface contains methods for materializing the elements of the
777             * ARRAY object on the client in the form of either an array or a
778             * ResultSet object.
779             */
780            case Types.ARRAY: {
781                Array a = result.getArray(i);
782                return a != null ? a.toString() : null;
783            }
784            /**
785             * The JDBC type BIGINT represents a 64-bit signed integer value
786             * between -9223372036854775808 and 9223372036854775807.
787             *
