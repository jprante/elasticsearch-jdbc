076        private final BulkProcessor.Listener listener = new BulkProcessor.Listener() {
077    
078            @Override
079            public void beforeBulk(long executionId, BulkRequest request) {
080                long l = outstandingBulkRequests.incrementAndGet();
081                logger().info("new bulk [{}] of [{} items], {} outstanding bulk requests",
082                        executionId, request.numberOfActions(), l);
083            }
084    
085            @Override
086            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
087                outstandingBulkRequests.decrementAndGet();
088                logger().info("bulk [{}] success [{} items] [{}ms]",
089                        executionId, response.getItems().length, response.getTookInMillis());
090    
091            }
092    
093            @Override
094            public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
095                outstandingBulkRequests.decrementAndGet();
096                logger().error("bulk [" + executionId + "] error", failure);
097                error = true;
098            }
099        };

186        @Override
187        public void index(IndexableObject object, boolean create) throws IOException {
188            if (error) {
189                logger().error("error, not indexing");
190                return;
191            }
192            if (!started) {
193                started = true;
194                startup();
195            }
196            if (Strings.hasLength(object.index())) {
197                setIndex(object.index());
198            }
199            if (Strings.hasLength(object.type())) {
200                setType(object.type());
201            }
202            if (Strings.hasLength(object.id())) {
203                setId(object.id());
204            }
205            IndexRequest request = Requests.indexRequest(getIndex())
206                    .type(getType())
207                    .id(getId())
208                    .source(object.build());
209            if (object.meta(ControlKeys._version.name()) != null) {
210                request.versionType(VersionType.EXTERNAL)
211                        .version(Long.parseLong(object.meta(ControlKeys._version.name())));
212            }
213            if (object.meta(ControlKeys._routing.name()) != null) {
214                request.routing(object.meta(ControlKeys._routing.name()));
215            }
216            if (object.meta(ControlKeys._parent.name()) != null) {
217                request.parent(object.meta(ControlKeys._parent.name()));
218            }
219            if (object.meta(ControlKeys._timestamp.name()) != null) {
220                request.timestamp(object.meta(ControlKeys._timestamp.name()));
221            }
222            if (object.meta(ControlKeys._ttl.name()) != null) {
223                request.ttl(Long.parseLong(object.meta(ControlKeys._ttl.name())));
224            }
225            bulk.add(request);
226        }
227    
228        @Override
229        public void delete(IndexableObject object) {
230            if (error) {
231                logger().error("error, not indexing");
232                return;
233            }
234            if (!started) {
235                started = true;
236                startup();
237            }
238            if (Strings.hasLength(object.index())) {
239                setIndex(object.index());
240            }
241            if (Strings.hasLength(object.type())) {
242                setType(object.type());
243            }
244            if (Strings.hasLength(object.id())) {
245                setId(object.id());
246            }
247            if (getId() == null) {
248                return; // skip if no doc is specified to delete
249            }
250            DeleteRequest request = Requests.deleteRequest(getIndex()).type(getType()).id(getId());
251            if (object.meta(ControlKeys._version.name()) != null) {
252                request.versionType(VersionType.EXTERNAL)
253                        .version(Long.parseLong(object.meta(ControlKeys._version.name())));
254            }
255            if (object.meta(ControlKeys._routing.name()) != null) {
256                request.routing(object.meta(ControlKeys._routing.name()));
257            }
258            if (object.meta(ControlKeys._parent.name()) != null) {
259                request.parent(object.meta(ControlKeys._parent.name()));
260            }
261            bulk.add(request);
262        }

264        @Override
265        public void flush() throws IOException {
266            try {
267                // we must wait for flush interval... not really cool
268                Thread.sleep(flushInterval.millis());
269            } catch (InterruptedException e) {
270                Thread.currentThread().interrupt();
271                throw new IOException("interrupted (maybe harmless)");
272            }
273        }

280        private RiverMouth startup() {
281            try {
282                //updateSettings();
283                //updateMapping();
284                createIndexIfNotExists();
285            } catch (Exception e) {
286                if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
287                    logger().warn("index {} already exists", index);
288                } else {
289                    logger().warn("failed to create index", e);
290                    error = true;
291                }
292            }
293            return this;
294        }
295    
296        private void updateSettings() {
297            if (error) {
298                logger().error("error, not updating settings");
299                return;
300            }
301            if (settings != null) {
302                client.admin().indices().prepareUpdateSettings(index)
303                        .setSettings(settings)
304                        .execute().actionGet();
305            }
306        }
307    
308        private void updateMapping() {
309            if (error) {
310                logger().error("error, not updating mapping");
311                return;
312            }
313            if (mapping != null) {
314                client.admin().indices().preparePutMapping(index)
315                        .setType(type)
316                        .setSource(mapping)
317                        .execute().actionGet();
318            }
319        }
320    
321        private void createIndexIfNotExists() {
322            if (error) {
323                logger().error("error, not creating index");
324                return;
325            }
326            if (client.admin().indices().prepareExists(index).execute().actionGet().isExists()) {
327                logger().warn("index {} already exists, ignoring index creation", index);
328                return;
329            }
330            CreateIndexRequestBuilder createIndexRequestBuilder =
331                    client.admin().indices().prepareCreate(index);
332            if (settings != null) {
333                createIndexRequestBuilder.setSettings(settings);
334            }
335            if (mapping != null) {
336                createIndexRequestBuilder.addMapping(type, mapping);
337            }
338            createIndexRequestBuilder.execute().actionGet();
339        }

341        @Override
342        public void waitForCluster() throws IOException {
343            waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
344        }
345    
346        public void waitForCluster(ClusterHealthStatus status, TimeValue timeout) throws IOException {
347            try {
348                logger().info("waiting for cluster state {}", status.name());
349                ClusterHealthResponse healthResponse =
350                        client.admin().cluster().prepareHealth().setWaitForStatus(status).setTimeout(timeout).execute().actionGet();
351                if (healthResponse.isTimedOut()) {
352                    throw new IOException("cluster state is " + healthResponse.getStatus().name()
353                            + " and not " + status.name()
354                            + ", cowardly refusing to continue with operations");
355                } else {
356                    logger().info("... cluster state ok");
357                }
358            } catch (ElasticsearchTimeoutException e) {
359                throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
360            }
361        }
