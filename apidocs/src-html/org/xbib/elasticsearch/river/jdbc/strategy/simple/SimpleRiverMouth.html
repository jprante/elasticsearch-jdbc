003
004import org.elasticsearch.ExceptionsHelper;
005import org.elasticsearch.action.admin.indices.create.CreateIndexRequestBuilder;
006import org.elasticsearch.action.delete.DeleteRequest;
007import org.elasticsearch.action.index.IndexRequest;
008import org.elasticsearch.client.Requests;
009import org.elasticsearch.common.Strings;
010import org.elasticsearch.index.VersionType;
011import org.elasticsearch.indices.IndexAlreadyExistsException;
012import org.xbib.elasticsearch.plugin.support.ControlKeys;
013import org.xbib.elasticsearch.plugin.support.IndexableObject;
014import org.xbib.elasticsearch.plugin.support.RiverContext;

019
020import java.io.IOException;

023/**
024 * Simple river mouth
025 */
026public class SimpleRiverMouth implements RiverMouth {
027
028    private final Logger logger = LoggerFactory.getLogger(SimpleRiverMouth.class.getName());
029
030    protected RiverContext context;
031
032    protected Ingest ingest;
033
034    protected Map<String, Object> settings;
035
036    protected Map<String, Object> mapping;
037
038    protected String index;
039
040    protected String type;
041
042    protected String id;
043
044    private volatile boolean error;
045
046    private volatile boolean closed;
047
048    protected Logger logger() {
049        return logger;
050    }
051
052    @Override
053    public String strategy() {
054        return "simple";
055    }
056
057    @Override
058    public SimpleRiverMouth setRiverContext(RiverContext context) {
059        this.context = context;
060        return this;
061    }
062
063    @Override
064    public SimpleRiverMouth setIngest(Ingest ingest) {
065        this.ingest = ingest;
066        return this;
067    }
068
069    public Ingest getIngest() {
070        return ingest;
071    }
072
073    @Override
074    public SimpleRiverMouth setSettings(Map<String, Object> settings) {
075        this.settings = settings;
076        return this;
077    }
078
079    @Override
080    public SimpleRiverMouth setMapping(Map<String, Object> mapping) {
081        this.mapping = mapping;
082        return this;
083    }
084
085    @Override
086    public SimpleRiverMouth setIndex(String index) {
087        this.index = index;
088        return this;
089    }
090
091    @Override
092    public String getIndex() {
093        return index;
094    }
095
096    @Override
097    public SimpleRiverMouth setType(String type) {
098        this.type = type;
099        return this;
100    }
101
102    @Override
103    public String getType() {
104        return type;
105    }
106
107    @Override
108    public SimpleRiverMouth setId(String id) {
109        this.id = id;
110        return this;
111    }
112
113    @Override
114    public String getId() {
115        return id;
116    }
117
118    @Override
119    public void index(IndexableObject object, boolean create) throws IOException {
120        if (error) {
121            logger().error("error, not indexing");
122            return;
123        }
124        ensureStarted();
125        if (Strings.hasLength(object.index())) {
126            setIndex(object.index());
127        }
128        if (Strings.hasLength(object.type())) {
129            setType(object.type());
130        }
131        if (Strings.hasLength(object.id())) {
132            setId(object.id());
133        }
134        IndexRequest request = Requests.indexRequest(getIndex())
135                .type(getType())
136                .id(getId())
137                .source(object.build());
138        if (object.meta(ControlKeys._version.name()) != null) {
139            request.versionType(VersionType.EXTERNAL)
140                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
141        }
142        if (object.meta(ControlKeys._routing.name()) != null) {
143            request.routing(object.meta(ControlKeys._routing.name()));
144        }
145        if (object.meta(ControlKeys._parent.name()) != null) {
146            request.parent(object.meta(ControlKeys._parent.name()));
147        }
148        if (object.meta(ControlKeys._timestamp.name()) != null) {
149            request.timestamp(object.meta(ControlKeys._timestamp.name()));
150        }
151        if (object.meta(ControlKeys._ttl.name()) != null) {
152            request.ttl(Long.parseLong(object.meta(ControlKeys._ttl.name())));
153        }
154        if (logger.isDebugEnabled()) {
155            logger.debug("adding bulk action {}", request.source().toUtf8());
156        }
157        ingest.index(request);
158    }
159
160    @Override
161    public void delete(IndexableObject object) {
162        if (error) {
163            logger().error("error, not indexing");
164            return;
165        }
166        ensureStarted();
167        if (Strings.hasLength(object.index())) {
168            setIndex(object.index());
169        }
170        if (Strings.hasLength(object.type())) {
171            setType(object.type());
172        }
173        if (Strings.hasLength(object.id())) {
174            setId(object.id());
175        }
176        if (getId() == null) {
177            return; // skip if no doc is specified to delete
178        }
179        DeleteRequest request = Requests.deleteRequest(getIndex()).type(getType()).id(getId());
180        if (object.meta(ControlKeys._version.name()) != null) {
181            request.versionType(VersionType.EXTERNAL)
182                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
183        }
184        if (object.meta(ControlKeys._routing.name()) != null) {
185            request.routing(object.meta(ControlKeys._routing.name()));
186        }
187        if (object.meta(ControlKeys._parent.name()) != null) {
188            request.parent(object.meta(ControlKeys._parent.name()));
189        }
190        ingest.delete(request);
191    }
192
193    @Override
194    public void flush() throws IOException {
195        if (ingest != null) {
196            ingest.flush();
197        }
198    }
199
200    @Override
201    public void close() throws IOException {
202        if (!closed) {
203            // do not shut down ingest object...  we need it for cleanup
204            closed = true;
205        }
206    }
207
208    private synchronized void ensureStarted() {
209        if (!closed) {
210            return;
211        }
212        try {
213            createIndexIfNotExists();
214        } catch (Exception e) {
215            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
216                logger().warn("index {} already exists", index);
217            } else {
218                logger().warn("failed to create index", e);
219                error = true;
220            }
221        } finally {
222            closed = false;
223        }
224    }
225
226    private void createIndexIfNotExists() {
227        if (error) {
228            logger().error("error, not creating index");
229            return;
230        }
231        if (ingest == null) {
232            return;
233        }
234        if (ingest.client().admin().indices().prepareExists(index).execute().actionGet().isExists()) {
235            logger().warn("index {} already exists, ignoring index creation", index);
236            return;
237        }
238        CreateIndexRequestBuilder createIndexRequestBuilder =
239                ingest.client().admin().indices().prepareCreate(index);
240        if (settings != null) {
241            createIndexRequestBuilder.setSettings(settings);
242        }
243        if (mapping != null) {
244            createIndexRequestBuilder.addMapping(type, mapping);
245        }
246        createIndexRequestBuilder.execute().actionGet();
247    }
