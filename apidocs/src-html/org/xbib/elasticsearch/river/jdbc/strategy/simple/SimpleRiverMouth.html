001package org.xbib.elasticsearch.river.jdbc.strategy.simple;
002
003import org.elasticsearch.action.delete.DeleteRequest;
004import org.elasticsearch.action.index.IndexRequest;
005import org.elasticsearch.client.Requests;
006import org.elasticsearch.common.Strings;
007import org.elasticsearch.common.joda.time.DateTime;
008import org.elasticsearch.common.joda.time.format.DateTimeFormat;
009import org.elasticsearch.common.logging.ESLogger;
010import org.elasticsearch.common.logging.ESLoggerFactory;
011import org.elasticsearch.common.unit.TimeValue;
012import org.elasticsearch.index.VersionType;
013import org.xbib.elasticsearch.plugin.jdbc.ControlKeys;
014import org.xbib.elasticsearch.plugin.jdbc.IndexableObject;
015import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
016import org.xbib.elasticsearch.river.jdbc.RiverMouth;
017import org.xbib.elasticsearch.support.client.Ingest;
018
019import java.io.IOException;
020
021/**
022 * Simple river mouth
023 */
024public class SimpleRiverMouth implements RiverMouth {
025
026    private final ESLogger logger = ESLoggerFactory.getLogger(SimpleRiverMouth.class.getName());
027
028    protected RiverContext context;
029
030    protected Ingest ingest;
031
032    protected String index;
033
034    protected String type;
035
036    protected String id;
037
038    private boolean timeWindowed;
039
040    private volatile boolean closed;
041
042    protected ESLogger logger() {
043        return logger;
044    }
045
046    @Override
047    public String strategy() {
048        return "simple";
049    }
050
051    @Override
052    public SimpleRiverMouth setRiverContext(RiverContext context) {
053        this.context = context;
054        return this;
055    }
056
057    @Override
058    public SimpleRiverMouth setIngest(Ingest ingest) {
059        this.ingest = ingest;
060        return this;
061    }
062
063    @Override
064    public SimpleRiverMouth setIndex(String index) {
065        this.index = timeWindowed ? DateTimeFormat.forPattern(index).print(new DateTime()) : index;
066        return this;
067    }
068
069    public String getIndex() {
070        return index;
071    }
072
073    @Override
074    public SimpleRiverMouth setType(String type) {
075        this.type = type;
076        return this;
077    }
078
079    public String getType() {
080        return type;
081    }
082
083    @Override
084    public SimpleRiverMouth setId(String id) {
085        this.id = id;
086        return this;
087    }
088
089    @Override
090    public SimpleRiverMouth setTimeWindowed(boolean timeWindowed) {
091        this.timeWindowed = timeWindowed;
092        return this;
093    }
094
095    public boolean isTimeWindowed() {
096        return timeWindowed;
097    }
098
099    @Override
100    public String getId() {
101        return id;
102    }
103
104    @Override
105    public void index(IndexableObject object, boolean create) throws IOException {
106        if (Strings.hasLength(object.index())) {
107            setIndex(object.index());
108        }
109        if (Strings.hasLength(object.type())) {
110            setType(object.type());
111        }
112        if (Strings.hasLength(object.id())) {
113            setId(object.id());
114        }
115        IndexRequest request = Requests.indexRequest(this.index)
116                .type(this.type)
117                .id(getId())
118                .source(object.build());
119        if (object.meta(ControlKeys._version.name()) != null) {
120            request.versionType(VersionType.EXTERNAL)
121                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
122        }
123        if (object.meta(ControlKeys._routing.name()) != null) {
124            request.routing(object.meta(ControlKeys._routing.name()));
125        }
126        if (object.meta(ControlKeys._parent.name()) != null) {
127            request.parent(object.meta(ControlKeys._parent.name()));
128        }
129        if (object.meta(ControlKeys._timestamp.name()) != null) {
130            request.timestamp(object.meta(ControlKeys._timestamp.name()));
131        }
132        if (object.meta(ControlKeys._ttl.name()) != null) {
133            request.ttl(Long.parseLong(object.meta(ControlKeys._ttl.name())));
134        }
135        if (logger.isTraceEnabled()) {
136            logger.trace("adding bulk index action {}", request.source().toUtf8());
137        }
138        if (ingest != null) {
139            ingest.bulkIndex(request);
140        }
141    }
142
143    @Override
144    public void delete(IndexableObject object) {
145        if (Strings.hasLength(object.index())) {
146            this.index = object.index();
147        }
148        if (Strings.hasLength(object.type())) {
149            this.type = object.type();
150        }
151        if (Strings.hasLength(object.id())) {
152            setId(object.id());
153        }
154        if (getId() == null) {
155            return; // skip if no doc is specified to delete
156        }
157        DeleteRequest request = Requests.deleteRequest(this.index).type(this.type).id(getId());
158        if (object.meta(ControlKeys._version.name()) != null) {
159            request.versionType(VersionType.EXTERNAL)
160                    .version(Long.parseLong(object.meta(ControlKeys._version.name())));
161        }
162        if (object.meta(ControlKeys._routing.name()) != null) {
163            request.routing(object.meta(ControlKeys._routing.name()));
164        }
165        if (object.meta(ControlKeys._parent.name()) != null) {
166            request.parent(object.meta(ControlKeys._parent.name()));
167        }
168        if (logger.isTraceEnabled()) {
169            logger.trace("adding bulk delete action {}/{}/{}", request.index(), request.type(), request.id());
170        }
171        if (ingest != null) {
172            ingest.bulkDelete(request);
173        }
174    }
175
176    @Override
177    public void flush() throws IOException {
178        if (ingest != null) {
179            ingest.flushIngest();
180            // wait for all outstanding bulk requests before continue with river
181            try {
182                ingest.waitForResponses(TimeValue.timeValueSeconds(60));
183            } catch (InterruptedException e) {
184                Thread.currentThread().interrupt();
185                logger.warn("interrupted while waiting for responses");
186            }
187        }
188    }
189
190    @Override
191    public void close() throws IOException {
192        // keep open, do not close or shut down ingest object here...  we need it for cleanup
193    }
194
195}
