001package org.xbib.elasticsearch.river.jdbc.strategy.column;
002
003import org.elasticsearch.common.logging.ESLogger;
004import org.elasticsearch.common.logging.ESLoggerFactory;
005import org.xbib.elasticsearch.plugin.feeder.jdbc.JDBCFeeder;
006import org.xbib.elasticsearch.river.jdbc.RiverFlow;
007import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverFlow;
008
009/**
010 * River flow implementation for the 'column' strategy
011 *
012 * @author <a href="piotr.sliwa@zineinc.com">Piotr Åšliwa</a>
013 */
014public class ColumnRiverFlow extends SimpleRiverFlow {
015
016    private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverFlow.class.getName());
017
018    public static final String DOCUMENT = "_custom";
019
020    public static final String LAST_RUN_TIME = "last_run_time";
021
022    public static final String CURRENT_RUN_STARTED_TIME = "current_run_started_time";
023
024    protected ESLogger logger() {
025        return logger;
026    }
027
028    @Override
029    public String strategy() {
030        return "column";
031    }
032
033    @Override
034    public RiverFlow setFeeder(JDBCFeeder feeder) {
035        this.feeder = feeder;
036        return this;
037    }
038
039    @Override
040    public JDBCFeeder getFeeder() {
041        if (feeder == null) {
042            this.feeder = new ColumnRiverFeeder();
043        }
044        return feeder;
045    }
046
047}
