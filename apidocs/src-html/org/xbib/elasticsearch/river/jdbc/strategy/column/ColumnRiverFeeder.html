002package org.xbib.elasticsearch.river.jdbc.strategy.column;
003
004import org.elasticsearch.action.get.GetResponse;
005import org.elasticsearch.client.Requests;
006import org.elasticsearch.common.logging.ESLogger;
007import org.elasticsearch.common.logging.ESLoggerFactory;
008import org.elasticsearch.common.unit.TimeValue;
009import org.elasticsearch.common.xcontent.support.XContentMapValues;
010import org.elasticsearch.indices.IndexMissingException;
011import org.elasticsearch.common.xcontent.XContentBuilder;
012import org.elasticsearch.common.xcontent.XContentFactory;
013import org.xbib.elasticsearch.action.river.state.RiverState;
014import org.xbib.elasticsearch.plugin.feeder.jdbc.JDBCFeeder;
015import org.xbib.pipeline.Pipeline;
016import org.xbib.pipeline.PipelineProvider;
017import org.xbib.pipeline.PipelineRequest;
018
019import java.io.IOException;
020import java.util.Date;
021import java.util.HashMap;
022import java.util.Map;
023
024public class ColumnRiverFeeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
025        extends JDBCFeeder<T,R,P> {
026
027    private final static ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverFeeder.class.getSimpleName());
028
029    public ColumnRiverFeeder(){
030        super();
031    }
032    
033    @SuppressWarnings("rawtypes")
034    public ColumnRiverFeeder(ColumnRiverFeeder feeder){
035        super(feeder);
036    }
037    
038    @Override
039    protected void createRiverContext(String riverType, String riverName, Map<String, Object> mySettings) {
040        super.createRiverContext(riverType, riverName, mySettings);
041        // defaults for column strategy
042        String columnCreatedAt = XContentMapValues.nodeStringValue(mySettings.get("created_at"), "created_at");
043        String columnUpdatedAt = XContentMapValues.nodeStringValue(mySettings.get("updated_at"), "updated_at");
044        String columnDeletedAt = XContentMapValues.nodeStringValue(mySettings.get("deleted_at"), null);
045        boolean columnEscape = XContentMapValues.nodeBooleanValue(mySettings.get("column_escape"), true);
046        riverContext
047                .columnCreatedAt(columnCreatedAt)
048                .columnUpdatedAt(columnUpdatedAt)
049                .columnDeletedAt(columnDeletedAt)
050                .columnEscape(columnEscape);
051    }
052
053        @Override
054    public void executeTask(Map<String, Object> map) throws Exception {
055        logger.info("processing map {}", map);
056        createRiverContext("jdbc", "feeder", map);
057        if (riverState == null) {
058            riverState = new RiverState();
059        }
060        riverState.load(ingest.client());
061        // increment state counter
062        Long counter = riverState.getCounter() + 1;
063        this.riverState = riverState.setCounter(counter)
064                .setEnabled(true)
065                .setActive(true)
066                .setTimestamp(new Date());
067        riverState.save(ingest.client());
068        if (logger.isDebugEnabled()) {
069            logger.debug("state saved before fetch");
070        }
071        // set the job number to the state counter
072        riverContext.job(Long.toString(counter));
073        if (logger.isDebugEnabled()) {
074            logger.debug("trying to fetch ...");
075        }
076
077        // column river specific code here
078        TimeValue lastRunTime = readLastRunTimeFromCustomInfo();
079        TimeValue currentTime = new TimeValue(new java.util.Date().getTime());
080        writeTimesToJdbcSettings(lastRunTime, currentTime);
081        riverContext.getRiverSource().fetch();
082        writeCustomInfo(currentTime.millis());
083
084
085        if (logger.isDebugEnabled()) {
086            logger.debug("fetched, flushing");
087        }
088        riverContext.getRiverMouth().flush();
089        if (logger.isDebugEnabled()) {
090            logger.debug("flushed");
091        }
092        riverContext.getRiverSource().closeReading();
093        riverContext.getRiverSource().closeWriting();
094        this.riverState = riverState
095                .setActive(false)
096                .setTimestamp(new Date());
097        riverState.save(ingest.client());
098    }
099
100    private TimeValue readLastRunTimeFromCustomInfo() throws IOException {
101        try {
102            GetResponse response = getClient().prepareGet("_river", riverContext.getRiverName(), ColumnRiverFlow.DOCUMENT).execute().actionGet();
103            if (response != null && response.isExists()) {
104                Map jdbcState = (Map) response.getSourceAsMap().get("jdbc");
105
106                if (jdbcState != null) {
107                    Number lastRunTime = (Number) jdbcState.get(ColumnRiverFlow.LAST_RUN_TIME);
108
109                    if (lastRunTime != null) {
110                        return new TimeValue(lastRunTime.longValue());
111                    }
112                } else {
113                    throw new IOException("can't retrieve previously persisted state from _river/" + riverContext.getRiverName());
114                }
115            }
116        } catch (IndexMissingException e) {
117            logger.warn("river state missing: _river/{}/{}", riverContext.getRiverName(), "_custom");
118        }
119
120        return null;
121    }
122
123    private void writeCustomInfo(long lastRunAt) throws IOException {
124        XContentBuilder builder = XContentFactory.jsonBuilder()
125                .startObject()
126                .startObject("jdbc")
127                .field(ColumnRiverFlow.LAST_RUN_TIME, lastRunAt)
128                .endObject()
129                .endObject();
130        getClient().prepareBulk().add(Requests.indexRequest("_river").type(riverContext.getRiverName()).id(ColumnRiverFlow.DOCUMENT)
131                .source(builder.string())).execute().actionGet();
132    }
133
134    private void writeTimesToJdbcSettings(TimeValue lastRunTime, TimeValue currentTime) {
135        if (riverContext == null || riverContext.getRiverSettings() == null) {
136            return;
137        }
138        Map<String, Object> settings = riverContext.getRiverSettings();
139        settings.put(ColumnRiverFlow.LAST_RUN_TIME, lastRunTime);
140        settings.put(ColumnRiverFlow.CURRENT_RUN_STARTED_TIME, currentTime);
141    }
142    
143    @Override
144    @SuppressWarnings({"rawtypes", "unchecked"})
145    public PipelineProvider<P> pipelineProvider() {
146        return new PipelineProvider<P>() {
147            @Override
148            public P get() {
149                return  (P) new ColumnRiverFeeder(ColumnRiverFeeder.this);
150            }
151        };
152    }
