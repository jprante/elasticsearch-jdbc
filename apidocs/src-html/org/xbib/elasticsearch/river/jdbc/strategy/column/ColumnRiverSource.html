002package org.xbib.elasticsearch.river.jdbc.strategy.column;
003
004import org.elasticsearch.common.logging.ESLogger;
005import org.elasticsearch.common.logging.ESLoggerFactory;
006import org.elasticsearch.common.unit.TimeValue;
007import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
008import org.xbib.elasticsearch.river.jdbc.support.Operations;
009import org.xbib.elasticsearch.river.jdbc.support.SimpleValueListener;
010import org.xbib.elasticsearch.river.jdbc.support.StructuredObject;
011import org.xbib.elasticsearch.river.jdbc.support.ValueListener;
012
013import java.io.IOException;
014import java.sql.Connection;
015import java.sql.PreparedStatement;
016import java.sql.ResultSet;
017import java.sql.SQLException;
018import java.sql.Timestamp;
019import java.util.ArrayList;
020import java.util.Collections;
021import java.util.LinkedList;
022import java.util.List;
023import java.util.Map;
024
025/**
026 * River source implementation for the 'column' strategy
027 *
028 * @author Piotr Åšliwa <piotr.sliwa@zineinc.com>
029 */
030public class ColumnRiverSource extends SimpleRiverSource {
031
032    private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverSource.class.getSimpleName());
033
034    protected ESLogger logger() {
035        return logger;
036    }
037
038    @Override
039    public String strategy() {
040        return "column";
041    }
042
043    @Override
044    public String fetch() throws SQLException, IOException {
045
046        String sql = context.pollStatement();
047
048        Connection connection = connectionForReading();
049
050        List<OpInfo> opInfos = getOpInfos(connection);
051
052        Timestamp lastRunTimestamp = getLastRunTimestamp();
053
054        for (OpInfo opInfo : opInfos) {
055            fetch(connection, sql, opInfo, lastRunTimestamp);
056        }
057
058        return null;
059    }
060
061    private List<OpInfo> getOpInfos(Connection connection) throws SQLException {
062
063        String quoteString = getIdentifierQuoteString(connection);
064
065        List<OpInfo> opInfos = new LinkedList<OpInfo>();
066
067        String noDeletedWhereClause = context.columnDeletedAt() != null ? " AND " + quoteColumn(context.columnDeletedAt(), quoteString) + " IS NULL" : "";
068
069        opInfos.add(new OpInfo(Operations.OP_CREATE, quoteColumn(context.columnCreatedAt(), quoteString) + " >= ?" + noDeletedWhereClause));
070        opInfos.add(new OpInfo(Operations.OP_INDEX, quoteColumn(context.columnUpdatedAt(), quoteString) + " >= ? AND (" + quoteColumn(context.columnCreatedAt(), quoteString) + " IS NULL OR " + quoteColumn(context.columnCreatedAt(), quoteString) + " < ?)" + noDeletedWhereClause, 2));
071
072        if (context.columnDeletedAt() != null) {
073            opInfos.add(new OpInfo(Operations.OP_DELETE, quoteColumn(context.columnDeletedAt(), quoteString) + " >= ?"));
074        }
075
076        return opInfos;
077    }
078
079    private String getIdentifierQuoteString(Connection connection) throws SQLException {
080
081        if (!context.columnEscape()) {
082            return "";
083        }
084
085        String quoteString = connection.getMetaData().getIdentifierQuoteString();
086        quoteString = quoteString == null ? "" : quoteString;
087
088        return quoteString;
089    }
090
091    private String quoteColumn(String column, String quote) {
092        return quote + column + quote;
093    }
094
095    private Timestamp getLastRunTimestamp() {
096        Map jdbcSettings = (Map) context.riverSettings().get("jdbc");
097
098        if (jdbcSettings == null || jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME) == null) {
099            return new Timestamp(0);
100        }
101
102        TimeValue lastRunTime = (TimeValue) jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME);
103
104        return new Timestamp(lastRunTime.millis());
105    }
106
107    private void fetch(Connection connection, String sql, OpInfo opInfo, Timestamp lastRunTimestamp) throws IOException, SQLException {
108        String fullSql = addWhereClauseToSqlQuery(sql, opInfo.where);
109
110        PreparedStatement stmt = connection.prepareStatement(fullSql);
111        List<Object> params = createQueryParams(lastRunTimestamp, opInfo.paramsInWhere);
112
113        logger.debug("sql: {}, params {}", fullSql, params);
114
115        ResultSet result = null;
116
117        try {
118            bind(stmt, params);
119
120            result = executeQuery(stmt);
121
122            try {
123                ValueListener listener = new ColumnValueListener(opInfo.opType)
124                        .target(context.riverMouth())
125                        .digest(context.digesting());
126
127                merge(result, listener);
128            } catch (Exception e) {
129                throw new IOException(e);
130            }
131        } finally {
132            close(result);
133            close(stmt);
134        }
135
136        acknowledge();
137    }
138
139    private String addWhereClauseToSqlQuery(String sql, String whereClauseToAppend) {
140        final String whereKeyword = "where ";
141        int whereIndex = sql.toLowerCase().indexOf(whereKeyword);
142
143        return whereIndex > 0 ?
144                sql.substring(0, whereIndex + whereKeyword.length()) + whereClauseToAppend + " AND " + sql.substring(whereIndex + whereKeyword.length())
145                : sql + " WHERE " + whereClauseToAppend;
146    }
147
148    private List<Object> createQueryParams(Timestamp lastRunTimestamp, int lastRunTimestampParamsCount) {
149        List<? extends Object> statementParams = context.pollStatementParams() != null ? context.pollStatementParams() : Collections.emptyList();
150        List<Object> params = new ArrayList<Object>(statementParams.size() + lastRunTimestampParamsCount);
151
152        for (int i = 0; i < lastRunTimestampParamsCount; i++) {
153            params.add(lastRunTimestamp);
154        }
155
156        for (Object param : statementParams) {
157            params.add(param);
158        }
159
160        return params;
161    }
162
163    private static class OpInfo {
164        final String where;
165        final String opType;
166        final int paramsInWhere;
167
168        public OpInfo(String opType, String where, int paramsInWhere) {
169            this.opType = opType;
170            this.where = where;
171            this.paramsInWhere = paramsInWhere;
172        }
173
174        public OpInfo(String opType, String where) {
175            this(opType, where, 1);
176        }
177    }
178
179    private static class ColumnValueListener extends SimpleValueListener<Object> {
180
181        private String opType;
182
183        public ColumnValueListener(String opType) {
184            this.opType = opType;
185        }
186
187        @Override
188        public SimpleValueListener end(StructuredObject object) throws IOException {
189
190            if (!object.source().isEmpty()) {
191                object.optype(opType);
192            }
193
194            return super.end(object);
195        }
196    }
197}
