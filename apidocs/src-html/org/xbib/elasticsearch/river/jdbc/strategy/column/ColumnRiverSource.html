002    package org.xbib.elasticsearch.river.jdbc.strategy.column;
003    
004    import java.io.IOException;
005    import java.sql.Connection;
006    import java.sql.PreparedStatement;
007    import java.sql.ResultSet;
008    import java.sql.SQLException;
009    import java.sql.Timestamp;
010    import java.util.ArrayList;
011    import java.util.Collections;
012    import java.util.LinkedList;
013    import java.util.List;
014    import java.util.Map;
015    
016    import org.elasticsearch.common.logging.ESLogger;
017    import org.elasticsearch.common.logging.ESLoggerFactory;
018    import org.elasticsearch.common.unit.TimeValue;
019    
020    import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
021    import org.xbib.elasticsearch.river.jdbc.support.RiverKeyValueStreamListener;
022    import org.xbib.elasticsearch.river.jdbc.support.SQLCommand;
023    import org.xbib.elasticsearch.gatherer.IndexableObject;
024    import org.xbib.io.keyvalue.KeyValueStreamListener;
025    
026    /**
027     * River source implementation for the 'column' strategy
028     *
029     * @author Piotr Åšliwa <piotr.sliwa@zineinc.com>
030     */
031    public class ColumnRiverSource extends SimpleRiverSource {
032    
033        private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverSource.class.getName());
034    
035        private static final String WHERE_CLAUSE_PLACEHOLDER = "$where";
036    
037        protected ESLogger logger() {
038            return logger;
039        }
040    
041        @Override
042        public String strategy() {
043            return "column";
044        }
045    
046        @Override
047        public void fetch() throws SQLException, IOException {
048            for (SQLCommand command : context.getStatements()) {
049                Connection connection = connectionForReading();
050                List<OpInfo> opInfos = getOpInfos(connection);
051                Timestamp lastRunTimestamp = getLastRunTimestamp();
052                for (OpInfo opInfo : opInfos) {
053                    fetch(connection, command, opInfo, lastRunTimestamp);
054                }
055            }
056        }
057    
058        private List<OpInfo> getOpInfos(Connection connection) throws SQLException {
059            String quoteString = getIdentifierQuoteString(connection);
060            List<OpInfo> opInfos = new LinkedList<OpInfo>();
061            String noDeletedWhereClause = context.columnDeletedAt() != null ?
062                    " AND " + quoteColumn(context.columnDeletedAt(), quoteString) + " IS NULL" : "";
063            opInfos.add(new OpInfo("create", quoteColumn(context.columnCreatedAt(), quoteString) + " >= ?" + noDeletedWhereClause));
064            opInfos.add(new OpInfo("index", quoteColumn(context.columnUpdatedAt(), quoteString) + " >= ? AND (" + quoteColumn(context.columnCreatedAt(), quoteString) + " IS NULL OR " + quoteColumn(context.columnCreatedAt(), quoteString) + " < ?)" + noDeletedWhereClause, 2));
065            if (context.columnDeletedAt() != null) {
066                opInfos.add(new OpInfo("delete", quoteColumn(context.columnDeletedAt(), quoteString) + " >= ?"));
067            }
068    
069            return opInfos;
070        }
071    
072        private String getIdentifierQuoteString(Connection connection) throws SQLException {
073    
074            if (!context.columnEscape()) {
075                return "";
076            }
077    
078            String quoteString = connection.getMetaData().getIdentifierQuoteString();
079            quoteString = quoteString == null ? "" : quoteString;
080    
081            return quoteString;
082        }
083    
084        private String quoteColumn(String column, String quote) {
085            return quote + column + quote;
086        }
087    
088        private Timestamp getLastRunTimestamp() {
089            Map jdbcSettings = (Map) context.riverSettings().get("jdbc");
090    
091            if (jdbcSettings == null || jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME) == null) {
092                return new Timestamp(0);
093            }
094    
095            TimeValue lastRunTime = (TimeValue) jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME);
096    
097            return new Timestamp(lastRunTime.millis());
098        }
099    
100        private void fetch(Connection connection, SQLCommand command, OpInfo opInfo, Timestamp lastRunTimestamp) throws IOException, SQLException {
101            String fullSql = addWhereClauseToSqlQuery(command.getSQL(), opInfo.where);
102            PreparedStatement stmt = connection.prepareStatement(fullSql);
103            List<Object> params = createQueryParams(command, lastRunTimestamp, opInfo.paramsInWhere);
104            if (logger.isDebugEnabled()) {
105                logger.debug("sql: {}, params {}", fullSql, params);
106            }
107            ResultSet result = null;
108            try {
109                bind(stmt, params);
110                result = executeQuery(stmt);
111                try {
112                    KeyValueStreamListener listener = new ColumnKeyValueStreamListener(opInfo.opType)
113                            .output(context.riverMouth());
114                    merge(result, listener);
115                } catch (Exception e) {
116                    throw new IOException(e);
117                }
118            } finally {
119                close(result);
120                close(stmt);
121            }
122        }
123    
124        private String addWhereClauseToSqlQuery(String sql, String whereClauseToAppend) {
125            int wherePlaceholderIndex = sql.indexOf(WHERE_CLAUSE_PLACEHOLDER);
126            final String whereKeyword = "where ";
127            int whereIndex = sql.toLowerCase().indexOf(whereKeyword);
128    
129            if(wherePlaceholderIndex >= 0) {
130                return sql.replace(WHERE_CLAUSE_PLACEHOLDER, whereClauseToAppend);
131            } else if (whereIndex >= 0) {
132                return sql.substring(0, whereIndex + whereKeyword.length()) + whereClauseToAppend + " AND " + sql.substring(whereIndex + whereKeyword.length());
133            } else {
134                return sql + " WHERE " + whereClauseToAppend;
135            }
136        }
137    
138        private List<Object> createQueryParams(SQLCommand command, Timestamp lastRunTimestamp, int lastRunTimestampParamsCount) {
139            List<? extends Object> statementParams = command.getParameters() != null ?
140                    command.getParameters() : Collections.emptyList();
141            List<Object> params = new ArrayList<Object>(statementParams.size() + lastRunTimestampParamsCount);
142    
143            for (int i = 0; i < lastRunTimestampParamsCount; i++) {
144                params.add(lastRunTimestamp);
145            }
146    
147            for (Object param : statementParams) {
148                params.add(param);
149            }
150    
151            return params;
152        }
153    
154        private static class OpInfo {
155            final String where;
156            final String opType;
157            final int paramsInWhere;
158    
159            public OpInfo(String opType, String where, int paramsInWhere) {
160                
161                if(where != null && !where.equals("")) {
162                    where = "("+where+")";
163                }
164                
165                this.opType = opType;
166                this.where = where;
167                this.paramsInWhere = paramsInWhere;
168            }
169    
170            public OpInfo(String opType, String where) {
171                this(opType, where, 1);
172            }
173        }
174    
175        private static class ColumnKeyValueStreamListener extends RiverKeyValueStreamListener {
176    
177            private String opType;
178    
179            public ColumnKeyValueStreamListener(String opType) {
180                this.opType = opType;
181            }
182    
183            @Override
184            public RiverKeyValueStreamListener end(IndexableObject object) throws IOException {
185    
186                if (!object.source().isEmpty()) {
187                    object.optype(opType);
188                }
189    
190                return super.end(object);
191            }
192        }
193    }
