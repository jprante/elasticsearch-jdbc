001package org.xbib.elasticsearch.river.jdbc.strategy.column;
002
003import org.elasticsearch.common.logging.ESLogger;
004import org.elasticsearch.common.logging.ESLoggerFactory;
005import org.elasticsearch.common.unit.TimeValue;
006import org.xbib.elasticsearch.plugin.jdbc.IndexableObject;
007import org.xbib.elasticsearch.plugin.jdbc.RiverMouthKeyValueStreamListener;
008import org.xbib.elasticsearch.plugin.jdbc.SQLCommand;
009import org.xbib.elasticsearch.river.jdbc.strategy.simple.SimpleRiverSource;
010import org.xbib.keyvalue.KeyValueStreamListener;
011
012import java.io.IOException;
013import java.sql.Connection;
014import java.sql.PreparedStatement;
015import java.sql.ResultSet;
016import java.sql.SQLException;
017import java.sql.Timestamp;
018import java.util.ArrayList;
019import java.util.Collections;
020import java.util.LinkedList;
021import java.util.List;
022import java.util.Map;
023
024/**
025 * River source implementation for the 'column' strategy
026 *
027 * @author <a href="piotr.sliwa@zineinc.com">Piotr ?liwa</a>
028 */
029public class ColumnRiverSource extends SimpleRiverSource {
030
031    private final ESLogger logger = ESLoggerFactory.getLogger(ColumnRiverSource.class.getName());
032
033    private static final String WHERE_CLAUSE_PLACEHOLDER = "$where";
034
035    protected ESLogger logger() {
036        return logger;
037    }
038
039    @Override
040    public String strategy() {
041        return "column";
042    }
043
044    @Override
045    public void fetch() throws SQLException, IOException {
046        for (SQLCommand command : context.getStatements()) {
047            Connection connection = getConnectionForReading();
048            if (connection != null) {
049                List<OpInfo> opInfos = getOpInfos(connection);
050                Timestamp lastRunTimestamp = getLastRunTimestamp();
051                logger.info("lastRunTimestamp={}", lastRunTimestamp);
052                for (OpInfo opInfo : opInfos) {
053                    logger.info("opinfo={}", opInfo.toString());
054                    fetch(connection, command, opInfo, lastRunTimestamp);
055                }
056            }
057        }
058    }
059
060    private List<OpInfo> getOpInfos(Connection connection) throws SQLException {
061        String quoteString = getIdentifierQuoteString(connection);
062        List<OpInfo> opInfos = new LinkedList<OpInfo>();
063        String noDeletedWhereClause = context.columnDeletedAt() != null ?
064                " AND " + quoteColumn(context.columnDeletedAt(), quoteString) + " IS NULL" : "";
065        opInfos.add(new OpInfo("create", quoteColumn(context.columnCreatedAt(), quoteString) + " >= ?" + noDeletedWhereClause));
066        opInfos.add(new OpInfo("index", quoteColumn(context.columnUpdatedAt(), quoteString) + " >= ? AND (" + quoteColumn(context.columnCreatedAt(), quoteString) + " IS NULL OR " + quoteColumn(context.columnCreatedAt(), quoteString) + " < ?)" + noDeletedWhereClause, 2));
067        if (context.columnDeletedAt() != null) {
068            opInfos.add(new OpInfo("delete", quoteColumn(context.columnDeletedAt(), quoteString) + " >= ?"));
069        }
070        return opInfos;
071    }
072
073    private String getIdentifierQuoteString(Connection connection) throws SQLException {
074        if (!context.columnEscape()) {
075            return "";
076        }
077        String quoteString = connection.getMetaData().getIdentifierQuoteString();
078        quoteString = quoteString == null ? "" : quoteString;
079        return quoteString;
080    }
081
082    private String quoteColumn(String column, String quote) {
083        return quote + column + quote;
084    }
085
086    private Timestamp getLastRunTimestamp() {
087        Map jdbcSettings = (Map) context.getRiverSettings().get("jdbc");
088        logger.info("getLastRunTimestamp jdbcSettings={}", jdbcSettings);
089        if (jdbcSettings == null || jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME) == null) {
090            return new Timestamp(0);
091        }
092        TimeValue lastRunTime = (TimeValue) jdbcSettings.get(ColumnRiverFlow.LAST_RUN_TIME);
093        return new Timestamp(lastRunTime.millis());
094    }
095
096    private void fetch(Connection connection, SQLCommand command, OpInfo opInfo, Timestamp lastRunTimestamp) throws IOException, SQLException {
097        String fullSql = addWhereClauseToSqlQuery(command.getSQL(), opInfo.where);
098        PreparedStatement stmt = connection.prepareStatement(fullSql);
099        List<Object> params = createQueryParams(command, lastRunTimestamp, opInfo.paramsInWhere);
100        if (logger.isDebugEnabled()) {
101            logger.debug("sql: {}, params {}", fullSql, params);
102        }
103        ResultSet result = null;
104        try {
105            bind(stmt, params);
106            result = executeQuery(stmt);
107            KeyValueStreamListener<Object,Object> listener =
108                    new ColumnKeyValueStreamListener<Object,Object>(opInfo.opType)
109                        .output(context.getRiverMouth());
110                merge(result, listener);
111        } catch (Exception e) {
112            throw new IOException(e);
113        } finally {
114            close(result);
115            close(stmt);
116        }
117    }
118
119    private String addWhereClauseToSqlQuery(String sql, String whereClauseToAppend) {
120        int wherePlaceholderIndex = sql.indexOf(WHERE_CLAUSE_PLACEHOLDER);
121        final String whereKeyword = "where ";
122        int whereIndex = sql.toLowerCase().indexOf(whereKeyword);
123        if (wherePlaceholderIndex >= 0) {
124            return sql.replace(WHERE_CLAUSE_PLACEHOLDER, whereClauseToAppend);
125        } else if (whereIndex >= 0) {
126            return sql.substring(0, whereIndex + whereKeyword.length()) + whereClauseToAppend + " AND " + sql.substring(whereIndex + whereKeyword.length());
127        } else {
128            return sql + " WHERE " + whereClauseToAppend;
129        }
130    }
131
132    private List<Object> createQueryParams(SQLCommand command, Timestamp lastRunTimestamp, int lastRunTimestampParamsCount) {
133        List<? extends Object> statementParams = command.getParameters() != null ?
134                command.getParameters() : Collections.emptyList();
135        List<Object> params = new ArrayList<Object>(statementParams.size() + lastRunTimestampParamsCount);
136        for (int i = 0; i < lastRunTimestampParamsCount; i++) {
137            params.add(lastRunTimestamp);
138        }
139        for (Object param : statementParams) {
140            params.add(param);
141        }
142        return params;
143    }
144
145    private static class OpInfo {
146        final String where;
147        final String opType;
148        final int paramsInWhere;
149
150        public OpInfo(String opType, String where, int paramsInWhere) {
151            if (where != null && !where.equals("")) {
152                where = "(" + where + ")";
153            }
154            this.opType = opType;
155            this.where = where;
156            this.paramsInWhere = paramsInWhere;
157        }
158
159        public OpInfo(String opType, String where) {
160            this(opType, where, 1);
161        }
162
163        public String toString() {
164            return opType + " " + where + " " + paramsInWhere;
165        }
166    }
167
168    private static class ColumnKeyValueStreamListener<K,V> extends RiverMouthKeyValueStreamListener<K,V> {
169
170        private String opType;
171
172        public ColumnKeyValueStreamListener(String opType) {
173            this.opType = opType;
174        }
175
176        @Override
177        public ColumnKeyValueStreamListener end(IndexableObject object) throws IOException {
178            if (!object.source().isEmpty()) {
179                object.optype(opType);
180            }
181            super.end(object);
182            return this;
183        }
184    }
185}
