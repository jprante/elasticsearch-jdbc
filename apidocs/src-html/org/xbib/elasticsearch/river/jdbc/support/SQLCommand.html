002package org.xbib.elasticsearch.river.jdbc.support;
003
004import java.io.FileInputStream;
005import java.io.IOException;
006import java.io.InputStreamReader;
007import java.io.Reader;
008import java.util.Arrays;
009import java.util.List;
010import java.util.Map;
011
012import org.elasticsearch.common.io.Streams;
013import org.elasticsearch.common.logging.ESLogger;
014import org.elasticsearch.common.logging.ESLoggerFactory;
015import org.elasticsearch.common.xcontent.support.XContentMapValues;
016
017import static org.elasticsearch.common.collect.Lists.newLinkedList;
018import static org.elasticsearch.common.collect.Maps.newHashMap;
019
020public class SQLCommand {
021
022    private final static ESLogger logger = ESLoggerFactory.getLogger(SQLCommand.class.getName());
023
024    private String sql;
025
026    private List<? super Object> params = newLinkedList();
027
028    private Map<String,Object> results = newHashMap();
029
030    private boolean callable;
031
032    public SQLCommand setSQL(String sql) throws IOException {
033        if (sql.endsWith(".sql")) {
034            Reader r = new InputStreamReader(new FileInputStream(sql), "UTF-8");
035            sql = Streams.copyToString(r);
036            r.close();
037        }
038        this.sql = sql;
039        return this;
040    }
041
042    public String getSQL() {
043        return sql;
044    }
045
046    public SQLCommand setParameters(List<? super Object> params) {
047        this.params = params;
048        return this;
049    }
050
051    public List<? super Object> getParameters() {
052        return params;
053    }
054
055    public SQLCommand setCallable(boolean callable) {
056        this.callable = callable;
057        return this;
058    }
059
060    public boolean isCallable() {
061        return callable;
062    }
063
064    public boolean isQuery() {
065        int p1 = sql.toLowerCase().indexOf("select");
066        if (p1 < 0) {
067            return false;
068        }
069        int p2 = sql.toLowerCase().indexOf("update");
070        if (p2 < 0) {
071            return true;
072        }
073        int p3 = sql.toLowerCase().indexOf("insert");
074        return p3 < 0 || p1 < p2 && p1 < p3;
075    }
076
077    public void setResults(Map<String,Object> results) {
078        this.results = results;
079    }
080
081    public Map<String,Object> getResults() {
082        return results;
083    }
084
085    public static List<SQLCommand> parse(Map<String,Object> settings) {
086        List<SQLCommand> sql = newLinkedList();
087        if (!XContentMapValues.isArray(settings.get("sql"))) {
088            settings.put("sql", Arrays.asList(settings.get("sql")));
089        }
090        List<? super Object> list = (List<? super Object>) settings.get("sql");
091        for (Object entry : list) {
092            SQLCommand command = new SQLCommand();
093            try {
094                if (entry instanceof Map) {
095                    Map<String, Object> m = (Map<String, Object>) entry;
096                    if (m.containsKey("statement")) {
097                        command.setSQL((String) m.get("statement"));
098                    }
099                    if (m.containsKey("parameter")) {
100                        command.setParameters(XContentMapValues.extractRawValues("parameter", m));
101                    }
102                    if (m.containsKey("callable")) {
103                        command.setCallable(XContentMapValues.nodeBooleanValue(m.get("callable")));
104                    }
105                    if (m.containsKey("register")) {
106                        command.setResults(XContentMapValues.nodeMapValue(m.get("register"), null));
107                    }
108                } else if (entry instanceof String) {
109                    command.setSQL((String) entry);
110                }
111                sql.add(command);
112            } catch (IOException e) {
113                logger.warn("SQL command not found", e);
114            }
115        }
116        return sql;
117    }
118
119    public String toString() {
120        return "statement=" + sql + " parameter=" + params + " callable=" + callable;
121    }
122
123}
