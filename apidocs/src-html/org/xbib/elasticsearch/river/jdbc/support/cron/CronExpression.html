018    /**
019     * Provides a parser and evaluator for unix-like cron expressions. Cron
020     * expressions provide the ability to specify complex time combinations such as
021     * &quot;At 8:00am every Monday through Friday&quot; or &quot;At 1:30am every
022     * last Friday of the month&quot;.
023     * <P>
024     * Cron expressions are comprised of 6 required fields and one optional field
025     * separated by white space. The fields respectively are described as follows:
026     * <p/>
027     * <table cellspacing="8">
028     * <tr>
029     * <th align="left">Field Name</th>
030     * <th align="left">&nbsp;</th>
031     * <th align="left">Allowed Values</th>
032     * <th align="left">&nbsp;</th>
033     * <th align="left">Allowed Special Characters</th>
034     * </tr>
035     * <tr>
036     * <td align="left"><code>Seconds</code></td>
037     * <td align="left">&nbsp;</th>
038     * <td align="left"><code>0-59</code></td>
039     * <td align="left">&nbsp;</th>
040     * <td align="left"><code>, - * /</code></td>
041     * </tr>
042     * <tr>
043     * <td align="left"><code>Minutes</code></td>
044     * <td align="left">&nbsp;</th>
045     * <td align="left"><code>0-59</code></td>
046     * <td align="left">&nbsp;</th>
047     * <td align="left"><code>, - * /</code></td>
048     * </tr>
049     * <tr>
050     * <td align="left"><code>Hours</code></td>
051     * <td align="left">&nbsp;</th>
052     * <td align="left"><code>0-23</code></td>
053     * <td align="left">&nbsp;</th>
054     * <td align="left"><code>, - * /</code></td>
055     * </tr>
056     * <tr>
057     * <td align="left"><code>Day-of-month</code></td>
058     * <td align="left">&nbsp;</th>
059     * <td align="left"><code>1-31</code></td>
060     * <td align="left">&nbsp;</th>
061     * <td align="left"><code>, - * ? / L W</code></td>
062     * </tr>
063     * <tr>
064     * <td align="left"><code>Month</code></td>
065     * <td align="left">&nbsp;</th>
066     * <td align="left"><code>0-11 or JAN-DEC</code></td>
067     * <td align="left">&nbsp;</th>
068     * <td align="left"><code>, - * /</code></td>
069     * </tr>
070     * <tr>
071     * <td align="left"><code>Day-of-Week</code></td>
072     * <td align="left">&nbsp;</th>
073     * <td align="left"><code>1-7 or SUN-SAT</code></td>
074     * <td align="left">&nbsp;</th>
075     * <td align="left"><code>, - * ? / L #</code></td>
076     * </tr>
077     * <tr>
078     * <td align="left"><code>Year (Optional)</code></td>
079     * <td align="left">&nbsp;</th>
080     * <td align="left"><code>empty, 1970-2199</code></td>
081     * <td align="left">&nbsp;</th>
082     * <td align="left"><code>, - * /</code></td>
083     * </tr>
084     * </table>
085     * <P>
086     * The '*' character is used to specify all values. For example, &quot;*&quot;
087     * in the minute field means &quot;every minute&quot;.
088     * <P>
089     * The '?' character is allowed for the day-of-month and day-of-week fields. It
090     * is used to specify 'no specific value'. This is useful when you need to
091     * specify something in one of the two fields, but not the other.
092     * <P>
093     * The '-' character is used to specify ranges For example &quot;10-12&quot; in
094     * the hour field means &quot;the hours 10, 11 and 12&quot;.
095     * <P>
096     * The ',' character is used to specify additional values. For example
097     * &quot;MON,WED,FRI&quot; in the day-of-week field means &quot;the days Monday,
098     * Wednesday, and Friday&quot;.
099     * <P>
100     * The '/' character is used to specify increments. For example &quot;0/15&quot;
101     * in the seconds field means &quot;the seconds 0, 15, 30, and 45&quot;. And
102     * &quot;5/15&quot; in the seconds field means &quot;the seconds 5, 20, 35, and
103     * 50&quot;.  Specifying '*' before the  '/' is equivalent to specifying 0 is
104     * the value to start with. Essentially, for each field in the expression, there
105     * is a set of numbers that can be turned on or off. For seconds and minutes,
106     * the numbers range from 0 to 59. For hours 0 to 23, for days of the month 0 to
107     * 31, and for months 0 to 11 (JAN to DEC). The &quot;/&quot; character simply helps you turn
108     * on every &quot;nth&quot; value in the given set. Thus &quot;7/6&quot; in the
109     * month field only turns on month &quot;7&quot;, it does NOT mean every 6th
110     * month, please note that subtlety.
111     * <P>
112     * The 'L' character is allowed for the day-of-month and day-of-week fields.
113     * This character is short-hand for &quot;last&quot;, but it has different
114     * meaning in each of the two fields. For example, the value &quot;L&quot; in
115     * the day-of-month field means &quot;the last day of the month&quot; - day 31
116     * for January, day 28 for February on non-leap years. If used in the
117     * day-of-week field by itself, it simply means &quot;7&quot; or
118     * &quot;SAT&quot;. But if used in the day-of-week field after another value, it
119     * means &quot;the last xxx day of the month&quot; - for example &quot;6L&quot;
120     * means &quot;the last friday of the month&quot;. You can also specify an offset
121     * from the last day of the month, such as "L-3" which would mean the third-to-last
122     * day of the calendar month. <i>When using the 'L' option, it is important not to
123     * specify lists, or ranges of values, as you'll get confusing/unexpected results.</i>
124     * <P>
125     * The 'W' character is allowed for the day-of-month field.  This character
126     * is used to specify the weekday (Monday-Friday) nearest the given day.  As an
127     * example, if you were to specify &quot;15W&quot; as the value for the
128     * day-of-month field, the meaning is: &quot;the nearest weekday to the 15th of
129     * the month&quot;. So if the 15th is a Saturday, the trigger will fire on
130     * Friday the 14th. If the 15th is a Sunday, the trigger will fire on Monday the
131     * 16th. If the 15th is a Tuesday, then it will fire on Tuesday the 15th.
132     * However if you specify &quot;1W&quot; as the value for day-of-month, and the
133     * 1st is a Saturday, the trigger will fire on Monday the 3rd, as it will not
134     * 'jump' over the boundary of a month's days.  The 'W' character can only be
135     * specified when the day-of-month is a single day, not a range or list of days.
136     * <P>
137     * The 'L' and 'W' characters can also be combined for the day-of-month
138     * expression to yield 'LW', which translates to &quot;last weekday of the
139     * month&quot;.
140     * <P>
141     * The '#' character is allowed for the day-of-week field. This character is
142     * used to specify &quot;the nth&quot; XXX day of the month. For example, the
143     * value of &quot;6#3&quot; in the day-of-week field means the third Friday of
144     * the month (day 6 = Friday and &quot;#3&quot; = the 3rd one in the month).
145     * Other examples: &quot;2#1&quot; = the first Monday of the month and
146     * &quot;4#5&quot; = the fifth Wednesday of the month. Note that if you specify
147     * &quot;#5&quot; and there is not 5 of the given day-of-week in the month, then
148     * no firing will occur that month.  If the '#' character is used, there can
149     * only be one expression in the day-of-week field (&quot;3#1,6#3&quot; is
150     * not valid, since there are two expressions).
151     * <P>
152     * The 'C' character is allowed for the day-of-month and day-of-week fields.
153     * This character is short-hand for "calendar". This means values are
154     * calculated against the associated calendar, if any. If no calendar is
155     * associated, then it is equivalent to having an all-inclusive calendar. A
156     * value of "5C" in the day-of-month field means "the first day included by the
157     * calendar on or after the 5th". A value of "1C" in the day-of-week field
158     * means "the first day included by the calendar on or after Sunday".-->
159     * <P>
160     * The legal characters and the names of months and days of the week are not
161     * case sensitive.
162     * <p/>
163     * <p>
164     * <b>NOTES:</b>
165     * <ul>
166     * <li>Support for specifying both a day-of-week and a day-of-month value is
167     * not complete (you'll need to use the '?' character in one of these fields).
168     * </li>
169     * <li>Overflowing ranges is supported - that is, having a larger number on
170     * the left hand side than the right. You might do 22-2 to catch 10 o'clock
171     * at night until 2 o'clock in the morning, or you might have NOV-FEB. It is
172     * very important to note that overuse of overflowing ranges creates ranges
173     * that don't make sense and no effort has been made to determine which
174     * interpretation CronExpression chooses. An example would be
175     * "0 0 14-6 ? * FRI-MON". </li>
176     * </ul>
177     * </p>
178     */

283        /**
284         * Indicates whether the given date satisfies the cron expression. Note that
285         * milliseconds are ignored, so two Dates falling on different milliseconds
286         * of the same second will always have the same result here.
287         *
288         * @param date the date to evaluate
289         * @return a boolean indicating whether the given date satisfies the cron
290         * expression
291         */
292        public boolean isSatisfiedBy(Date date) {
293            Calendar testDateCal = Calendar.getInstance(getTimeZone());
294            testDateCal.setTime(date);
295            testDateCal.set(Calendar.MILLISECOND, 0);
296            Date originalDate = testDateCal.getTime();
297            testDateCal.add(Calendar.SECOND, -1);
298            Date timeAfter = getTimeAfter(testDateCal.getTime());
299            return ((timeAfter != null) && (timeAfter.equals(originalDate)));
300        }

314        /**
315         * Returns the next date/time <I>after</I> the given date/time which does
316         * <I>not</I> satisfy the expression
317         *
318         * @param date the date/time at which to begin the search for the next
319         *             invalid date/time
320         * @return the next valid date/time
321         */
322        public Date getNextInvalidTimeAfter(Date date) {
323            long difference = 1000;
324            Calendar adjustCal = Calendar.getInstance(getTimeZone());
325            adjustCal.setTime(date);
326            adjustCal.set(Calendar.MILLISECOND, 0);
327            Date lastDate = adjustCal.getTime();
328            Date newDate;
329            while (difference == 1000) {
330                newDate = getTimeAfter(lastDate);
331                if (newDate == null) {
332                    break;
333                }
334                difference = newDate.getTime() - lastDate.getTime();
335                if (difference == 1000) {
336                    lastDate = newDate;
337                }
338            }
339            return new Date(lastDate.getTime() + 1000);
340        }

384        protected void buildExpression(String expression) {
385            expressionParsed = true;
386            try {
387                if (seconds == null) {
388                    seconds = new TreeSet<Integer>();
389                }
390                if (minutes == null) {
391                    minutes = new TreeSet<Integer>();
392                }
393                if (hours == null) {
394                    hours = new TreeSet<Integer>();
395                }
396                if (daysOfMonth == null) {
397                    daysOfMonth = new TreeSet<Integer>();
398                }
399                if (months == null) {
400                    months = new TreeSet<Integer>();
401                }
402                if (daysOfWeek == null) {
403                    daysOfWeek = new TreeSet<Integer>();
404                }
405                if (years == null) {
406                    years = new TreeSet<Integer>();
407                }
408                int exprOn = SECOND;
409                StringTokenizer exprsTok = new StringTokenizer(expression, " \t", false);
410                while (exprsTok.hasMoreTokens() && exprOn <= YEAR) {
411                    String expr = exprsTok.nextToken().trim();
412                    if (exprOn == DAY_OF_MONTH && expr.indexOf('L') != -1 && expr.length() > 1 && expr.contains(",")) {
413                        throw new ParseException("Support for specifying 'L' and 'LW' with other days of the month is not implemented", -1);
414                    }
415                    if (exprOn == DAY_OF_WEEK && expr.indexOf('L') != -1 && expr.length() > 1 && expr.contains(",")) {
416                        throw new ParseException("Support for specifying 'L' with other days of the week is not implemented", -1);
417                    }
418                    if (exprOn == DAY_OF_WEEK && expr.indexOf('#') != -1 && expr.indexOf('#', expr.indexOf('#') + 1) != -1) {
419                        throw new ParseException("Support for specifying multiple \"nth\" days is not implemented.", -1);
420                    }
421                    StringTokenizer vTok = new StringTokenizer(expr, ",");
422                    while (vTok.hasMoreTokens()) {
423                        String v = vTok.nextToken();
424                        storeExpressionVals(0, v, exprOn);
425                    }
426                    exprOn++;
427                }
428                if (exprOn <= DAY_OF_WEEK) {
429                    throw new ParseException("Unexpected end of expression.",
430                            expression.length());
431                }
432                if (exprOn <= YEAR) {
433                    storeExpressionVals(0, "*", YEAR);
434                }
435                TreeSet<Integer> dow = getSet(DAY_OF_WEEK);
436                TreeSet<Integer> dom = getSet(DAY_OF_MONTH);
437                boolean dayOfMSpec = !dom.contains(NO_SPEC);
438                boolean dayOfWSpec = !dow.contains(NO_SPEC);
439                if (!dayOfMSpec || dayOfWSpec) {
440                    if (!dayOfWSpec || dayOfMSpec) {
441                        throw new ParseException("support for specifying both a day-of-week AND a day-of-month parameter is not implemented.", 0);
442                    }
443                }
444            } catch (Exception e) {
445                throw new IllegalArgumentException("invalid cron expression format: " + e.getMessage());
446            }
447        }

449        protected int storeExpressionVals(int pos, String s, int type)
450                throws ParseException {
451            int incr = 0;
452            int i = skipWhiteSpace(pos, s);
453            if (i >= s.length()) {
454                return i;
455            }
456            char c = s.charAt(i);
457            if ((c >= 'A') && (c <= 'Z') && (!s.equals("L")) && (!s.equals("LW")) && (!s.matches("^L-[0-9]*[W]?"))) {
458                String sub = s.substring(i, i + 3);
459                int sval = -1;
460                int eval = -1;
461                if (type == MONTH) {
462                    sval = getMonthNumber(sub) + 1;
463                    if (sval <= 0) {
464                        throw new ParseException("invalid Month value: '" + sub + "'", i);
465                    }
466                    if (s.length() > i + 3) {
467                        c = s.charAt(i + 3);
468                        if (c == '-') {
469                            i += 4;
470                            sub = s.substring(i, i + 3);
471                            eval = getMonthNumber(sub) + 1;
472                            if (eval <= 0) {
473                                throw new ParseException("invalid Mmnth value: '" + sub + "'", i);
474                            }
475                        }
476                    }
477                } else if (type == DAY_OF_WEEK) {
478                    sval = getDayOfWeekNumber(sub);
479                    if (sval < 0) {
480                        throw new ParseException("invalid day-of-week value: '"
481                                + sub + "'", i);
482                    }
483                    if (s.length() > i + 3) {
484                        c = s.charAt(i + 3);
485                        if (c == '-') {
486                            i += 4;
487                            sub = s.substring(i, i + 3);
488                            eval = getDayOfWeekNumber(sub);
489                            if (eval < 0) {
490                                throw new ParseException("invalid day-of-Week value: '" + sub + "'", i);
491                            }
492                        } else if (c == '#') {
493                            try {
494                                i += 4;
495                                nthdayOfWeek = Integer.parseInt(s.substring(i));
496                                if (nthdayOfWeek < 1 || nthdayOfWeek > 5) {
497                                    throw new Exception();
498                                }
499                            } catch (Exception e) {
500                                throw new ParseException("numeric value between 1 and 5 must follow the '#' option", i);
501                            }
502                        } else if (c == 'L') {
503                            lastdayOfWeek = true;
504                            i++;
505                        }
506                    }
507    
508                } else {
509                    throw new ParseException("illegal characters for this position: '" + sub + "'", i);
510                }
511                if (eval != -1) {
512                    incr = 1;
513                }
514                addToSet(sval, eval, incr, type);
515                return (i + 3);
516            }
517    
518            if (c == '?') {
519                i++;
520                if ((i + 1) < s.length()
521                        && (s.charAt(i) != ' ' && s.charAt(i + 1) != '\t')) {
522                    throw new ParseException("illegal character after '?': " + s.charAt(i), i);
523                }
524                if (type != DAY_OF_WEEK && type != DAY_OF_MONTH) {
525                    throw new ParseException("'?' can only be specfied for day-of-month or day-of-week", i);
526                }
527                if (type == DAY_OF_WEEK && !lastdayOfMonth) {
528                    int val = daysOfMonth.last();
529                    if (val == NO_SPEC_INT) {
530                        throw new ParseException("'?' can only be specfied for day-of-month or day-of-week", i);
531                    }
532                }
533    
534                addToSet(NO_SPEC_INT, -1, 0, type);
535                return i;
536            }
537            if (c == '*' || c == '/') {
538                if (c == '*' && (i + 1) >= s.length()) {
539                    addToSet(ALL_SPEC_INT, -1, incr, type);
540                    return i + 1;
541                } else if (c == '/'
542                        && ((i + 1) >= s.length() || s.charAt(i + 1) == ' ' || s
543                        .charAt(i + 1) == '\t')) {
544                    throw new ParseException("'/' must be followed by an integer", i);
545                } else if (c == '*') {
546                    i++;
547                }
548                c = s.charAt(i);
549                if (c == '/') {
550                    i++;
551                    if (i >= s.length()) {
552                        throw new ParseException("unexpected end of string", i);
553                    }
554    
555                    incr = getNumericValue(s, i);
556    
557                    i++;
558                    if (incr > 10) {
559                        i++;
560                    }
561                    if (incr > 59 && (type == SECOND || type == MINUTE)) {
562                        throw new ParseException("increment > 60 : " + incr, i);
563                    } else if (incr > 23 && (type == HOUR)) {
564                        throw new ParseException("increment > 24 : " + incr, i);
565                    } else if (incr > 31 && (type == DAY_OF_MONTH)) {
566                        throw new ParseException("increment > 31 : " + incr, i);
567                    } else if (incr > 7 && (type == DAY_OF_WEEK)) {
568                        throw new ParseException("increment > 7 : " + incr, i);
569                    } else if (incr > 12 && (type == MONTH)) {
570                        throw new ParseException("increment > 12 : " + incr, i);
571                    }
572                } else {
573                    incr = 1;
574                }
575                addToSet(ALL_SPEC_INT, -1, incr, type);
576                return i;
577            } else if (c == 'L') {
578                i++;
579                if (type == DAY_OF_MONTH) {
580                    lastdayOfMonth = true;
581                }
582                if (type == DAY_OF_WEEK) {
583                    addToSet(7, 7, 0, type);
584                }
585                if (type == DAY_OF_MONTH && s.length() > i) {
586                    c = s.charAt(i);
587                    if (c == '-') {
588                        ValueSet vs = getValue(0, s, i + 1);
589                        lastdayOffset = vs.value;
590                        if (lastdayOffset > 30) {
591                            throw new ParseException("offset from last day must be <= 30", i + 1);
592                        }
593                        i = vs.pos;
594                    }
595                    if (s.length() > i) {
596                        c = s.charAt(i);
597                        if (c == 'W') {
598                            nearestWeekday = true;
599                            i++;
600                        }
601                    }
602                }
603                return i;
604            } else if (c >= '0' && c <= '9') {
605                int val = Integer.parseInt(String.valueOf(c));
606                i++;
607                if (i >= s.length()) {
608                    addToSet(val, -1, -1, type);
609                } else {
610                    c = s.charAt(i);
611                    if (c >= '0' && c <= '9') {
612                        ValueSet vs = getValue(val, s, i);
613                        val = vs.value;
614                        i = vs.pos;
615                    }
616                    i = checkNext(i, s, val, type);
617                    return i;
618                }
619            } else {
620                throw new ParseException("unexpected character: " + c, i);
621            }
622            return i;
623        }

625        protected int checkNext(int pos, String s, int val, int type)
626                throws ParseException {
627            int end = -1;
628            int i = pos;
629            if (i >= s.length()) {
630                addToSet(val, end, -1, type);
631                return i;
632            }
633            char c = s.charAt(pos);
634            if (c == 'L') {
635                if (type == DAY_OF_WEEK) {
636                    if (val < 1 || val > 7) {
637                        throw new ParseException("day-of-week values must be between 1 and 7", -1);
638                    }
639                    lastdayOfWeek = true;
640                } else {
641                    throw new ParseException("'L' option is not valid here (pos=" + i + ")", i);
642                }
643                TreeSet<Integer> set = getSet(type);
644                set.add(val);
645                i++;
646                return i;
647            }
648    
649            if (c == 'W') {
650                if (type == DAY_OF_MONTH) {
651                    nearestWeekday = true;
652                } else {
653                    throw new ParseException("'W' option is not valid here (pos=" + i + ")", i);
654                }
655                if (val > 31) {
656                    throw new ParseException("'W' option does not make sense with values larger than 31 (max number of days in a month)", i);
657                }
658                TreeSet<Integer> set = getSet(type);
659                set.add(val);
660                i++;
661                return i;
662            }
663            if (c == '#') {
664                if (type != DAY_OF_WEEK) {
665                    throw new ParseException("'#' option is not valid here. (pos=" + i + ")", i);
666                }
667                i++;
668                try {
669                    nthdayOfWeek = Integer.parseInt(s.substring(i));
670                    if (nthdayOfWeek < 1 || nthdayOfWeek > 5) {
671                        throw new Exception();
672                    }
673                } catch (Exception e) {
674                    throw new ParseException("numeric value between 1 and 5 must follow the '#' option", i);
675                }
676                TreeSet<Integer> set = getSet(type);
677                set.add(val);
678                i++;
679                return i;
680            }
681    
682            if (c == '-') {
683                i++;
684                c = s.charAt(i);
685                int v = Integer.parseInt(String.valueOf(c));
686                end = v;
687                i++;
688                if (i >= s.length()) {
689                    addToSet(val, end, 1, type);
690                    return i;
691                }
692                c = s.charAt(i);
693                if (c >= '0' && c <= '9') {
694                    ValueSet vs = getValue(v, s, i);
695                    end = vs.value;
696                    i = vs.pos;
697                }
698                if (i < s.length() && ((c = s.charAt(i)) == '/')) {
699                    i++;
700                    c = s.charAt(i);
701                    int v2 = Integer.parseInt(String.valueOf(c));
702                    i++;
703                    if (i >= s.length()) {
704                        addToSet(val, end, v2, type);
705                        return i;
706                    }
707                    c = s.charAt(i);
708                    if (c >= '0' && c <= '9') {
709                        ValueSet vs = getValue(v2, s, i);
710                        int v3 = vs.value;
711                        addToSet(val, end, v3, type);
712                        i = vs.pos;
713                        return i;
714                    } else {
715                        addToSet(val, end, v2, type);
716                        return i;
717                    }
718                } else {
719                    addToSet(val, end, 1, type);
720                    return i;
721                }
722            }
723            if (c == '/') {
724                i++;
725                c = s.charAt(i);
726                int v2 = Integer.parseInt(String.valueOf(c));
727                i++;
728                if (i >= s.length()) {
729                    addToSet(val, end, v2, type);
730                    return i;
731                }
732                c = s.charAt(i);
733                if (c >= '0' && c <= '9') {
734                    ValueSet vs = getValue(v2, s, i);
735                    int v3 = vs.value;
736                    addToSet(val, end, v3, type);
737                    i = vs.pos;
738                    return i;
739                } else {
740                    throw new ParseException("unexpected character '" + c + "' after '/'", i);
741                }
742            }
743            addToSet(val, end, 0, type);
744            i++;
745            return i;
746        }

849        protected void addToSet(int val, int end, int incr, int type)
850                throws ParseException {
851            TreeSet<Integer> set = getSet(type);
852            if (type == SECOND || type == MINUTE) {
853                if ((val < 0 || val > 59 || end > 59) && (val != ALL_SPEC_INT)) {
854                    throw new ParseException(
855                            "Minute and Second values must be between 0 and 59",
856                            -1);
857                }
858            } else if (type == HOUR) {
859                if ((val < 0 || val > 23 || end > 23) && (val != ALL_SPEC_INT)) {
860                    throw new ParseException(
861                            "Hour values must be between 0 and 23", -1);
862                }
863            } else if (type == DAY_OF_MONTH) {
864                if ((val < 1 || val > 31 || end > 31) && (val != ALL_SPEC_INT)
865                        && (val != NO_SPEC_INT)) {
866                    throw new ParseException(
867                            "Day of month values must be between 1 and 31", -1);
868                }
869            } else if (type == MONTH) {
870                if ((val < 1 || val > 12 || end > 12) && (val != ALL_SPEC_INT)) {
871                    throw new ParseException(
872                            "Month values must be between 1 and 12", -1);
873                }
874            } else if (type == DAY_OF_WEEK) {
875                if ((val == 0 || val > 7 || end > 7) && (val != ALL_SPEC_INT)
876                        && (val != NO_SPEC_INT)) {
877                    throw new ParseException(
878                            "Day-of-Week values must be between 1 and 7", -1);
879                }
880            }
881            if ((incr == 0 || incr == -1) && val != ALL_SPEC_INT) {
882                if (val != -1) {
883                    set.add(val);
884                } else {
885                    set.add(NO_SPEC);
886                }
887    
888                return;
889            }
890            int startAt = val;
891            int stopAt = end;
892    
893            if (val == ALL_SPEC_INT && incr <= 0) {
894                incr = 1;
895                set.add(ALL_SPEC);
896            }
897            if (type == SECOND || type == MINUTE) {
898                if (stopAt == -1) {
899                    stopAt = 59;
900                }
901                if (startAt == -1 || startAt == ALL_SPEC_INT) {
902                    startAt = 0;
903                }
904            } else if (type == HOUR) {
905                if (stopAt == -1) {
906                    stopAt = 23;
907                }
908                if (startAt == -1 || startAt == ALL_SPEC_INT) {
909                    startAt = 0;
910                }
911            } else if (type == DAY_OF_MONTH) {
912                if (stopAt == -1) {
913                    stopAt = 31;
914                }
915                if (startAt == -1 || startAt == ALL_SPEC_INT) {
916                    startAt = 1;
917                }
918            } else if (type == MONTH) {
919                if (stopAt == -1) {
920                    stopAt = 12;
921                }
922                if (startAt == -1 || startAt == ALL_SPEC_INT) {
923                    startAt = 1;
924                }
925            } else if (type == DAY_OF_WEEK) {
926                if (stopAt == -1) {
927                    stopAt = 7;
928                }
929                if (startAt == -1 || startAt == ALL_SPEC_INT) {
930                    startAt = 1;
931                }
932            } else if (type == YEAR) {
933                if (stopAt == -1) {
934                    stopAt = MAX_YEAR;
935                }
936                if (startAt == -1 || startAt == ALL_SPEC_INT) {
937                    startAt = 1970;
938                }
939            }
940            int max = -1;
941            if (stopAt < startAt) {
942                switch (type) {
943                    case SECOND:
944                        max = 60;
945                        break;
946                    case MINUTE:
947                        max = 60;
948                        break;
949                    case HOUR:
950                        max = 24;
951                        break;
952                    case MONTH:
953                        max = 12;
954                        break;
955                    case DAY_OF_WEEK:
956                        max = 7;
957                        break;
958                    case DAY_OF_MONTH:
959                        max = 31;
960                        break;
961                    case YEAR:
962                        throw new IllegalArgumentException("Start year must be less than stop year");
963                    default:
964                        throw new IllegalArgumentException("Unexpected type encountered");
965                }
966                stopAt += max;
967            }
968    
969            for (int i = startAt; i <= stopAt; i += incr) {
970                if (max == -1) {
971                    set.add(i);
972                } else {
973                    int i2 = i % max;
974                    if (i2 == 0 && (type == MONTH || type == DAY_OF_WEEK || type == DAY_OF_MONTH)) {
975                        i2 = max;
976                    }
977                    set.add(i2);
978                }
979            }
980        }

1086                }
1087                cl.set(Calendar.MINUTE, min);

1129                        } else {

1256                            dayShifted = true;
1257                        }
1258                        day += daysToAdd;
1259                        int weekOfMonth = day / 7;
1260                        if (day % 7 > 0) {
1261                            weekOfMonth++;
1262                        }
1263                        daysToAdd = (nthdayOfWeek - weekOfMonth) * 7;
1264                        day += daysToAdd;
1265                        if (daysToAdd < 0
1266                                || day > getLastDayOfMonth(mon, cl
1267                                .get(Calendar.YEAR))) {
1268                            cl.set(Calendar.SECOND, 0);
1269                            cl.set(Calendar.MINUTE, 0);
1270                            cl.set(Calendar.HOUR_OF_DAY, 0);
1271                            cl.set(Calendar.DAY_OF_MONTH, 1);
1272                            cl.set(Calendar.MONTH, mon);
1273                            // no '- 1' here because we are promoting the month
1274                            continue;
1275                        } else if (daysToAdd > 0 || dayShifted) {
1276                            cl.set(Calendar.SECOND, 0);
1277                            cl.set(Calendar.MINUTE, 0);
1278                            cl.set(Calendar.HOUR_OF_DAY, 0);
1279                            cl.set(Calendar.DAY_OF_MONTH, day);
1280                            cl.set(Calendar.MONTH, mon - 1);
1281                            // '- 1' here because we are NOT promoting the month
1282                            continue;
1283                        }
1284                    } else {
1285                        int cDow = cl.get(Calendar.DAY_OF_WEEK); // current d-o-w
1286                        int dow = daysOfWeek.first(); // desired
1287                        // d-o-w
1288                        st = daysOfWeek.tailSet(cDow);
1289                        if (st.size() > 0) {
1290                            dow = st.first();
1291                        }
1292    
1293                        int daysToAdd = 0;
1294                        if (cDow < dow) {
1295                            daysToAdd = dow - cDow;
1296                        }
1297                        if (cDow > dow) {
1298                            daysToAdd = dow + (7 - cDow);
1299                        }
1300    
1301                        int lDay = getLastDayOfMonth(mon, cl.get(Calendar.YEAR));
1302    
1303                        if (day + daysToAdd > lDay) { // will we pass the end of
1304                            // the month?
1305                            cl.set(Calendar.SECOND, 0);
1306                            cl.set(Calendar.MINUTE, 0);
1307                            cl.set(Calendar.HOUR_OF_DAY, 0);
1308                            cl.set(Calendar.DAY_OF_MONTH, 1);
1309                            cl.set(Calendar.MONTH, mon);
1310                            continue;
1311                        } else if (daysToAdd > 0) { // are we swithing days?
1312                            cl.set(Calendar.SECOND, 0);
1313                            cl.set(Calendar.MINUTE, 0);
1314                            cl.set(Calendar.HOUR_OF_DAY, 0);
1315                            cl.set(Calendar.DAY_OF_MONTH, day + daysToAdd);
1316                            cl.set(Calendar.MONTH, mon - 1);
1317                            continue;
1318                        }
1319                    }
1320                } else {
1321                    throw new UnsupportedOperationException(
1322                            "Support for specifying both a day-of-week AND a day-of-month parameter is not implemented.");
1323                }
1324                cl.set(Calendar.DAY_OF_MONTH, day);
1325    
1326                mon = cl.get(Calendar.MONTH) + 1;
1327                int year = cl.get(Calendar.YEAR);
1328                t = -1;
1329                if (year > MAX_YEAR) {
1330                    return null;
1331                }
1332                st = months.tailSet(mon);
1333                if (st.size() != 0) {
1334                    t = mon;
1335                    mon = st.first();
1336                } else {
1337                    mon = months.first();
1338                    year++;
1339                }
1340                if (mon != t) {
1341                    cl.set(Calendar.SECOND, 0);
1342                    cl.set(Calendar.MINUTE, 0);
1343                    cl.set(Calendar.HOUR_OF_DAY, 0);
1344                    cl.set(Calendar.DAY_OF_MONTH, 1);
1345                    cl.set(Calendar.MONTH, mon - 1);
1346                    cl.set(Calendar.YEAR, year);
1347                    continue;
1348                }
1349                cl.set(Calendar.MONTH, mon - 1);
1350                year = cl.get(Calendar.YEAR);
1351                st = years.tailSet(year);
1352                if (st.size() != 0) {
1353                    t = year;
1354                    year = st.first();
1355                } else {
1356                    return null;
1357                }
1358                if (year != t) {
1359                    cl.set(Calendar.SECOND, 0);
1360                    cl.set(Calendar.MINUTE, 0);
1361                    cl.set(Calendar.HOUR_OF_DAY, 0);
1362                    cl.set(Calendar.DAY_OF_MONTH, 1);
1363                    cl.set(Calendar.MONTH, 0);
1364                    cl.set(Calendar.YEAR, year);
1365                    continue;
1366                }
1367                cl.set(Calendar.YEAR, year);
1368                gotOne = true;
1369            }
1370            return cl.getTime();
1371        }
