001package org.xbib.elasticsearch.river.jdbc.support;
002
003import java.lang.Runnable;
004import java.lang.Thread;
005
006import org.elasticsearch.action.admin.cluster.health.ClusterHealthResponse;
007import org.elasticsearch.client.Client;
008import org.elasticsearch.common.logging.ESLogger;
009import org.elasticsearch.common.logging.ESLoggerFactory;
010import org.elasticsearch.common.unit.TimeValue;
011
012/**
013 * Cluster health monitor thread. Checks the cluster status every 5 minutes and is used by
014 * all JDBC river mouths to check if actions should be performed.
015 * @author pdegeus
016 */
017public class HealthMonitorThread implements Runnable {
018
019    private static final ESLogger logger = ESLoggerFactory.getLogger(HealthMonitorThread.class.getName());
020
021    private boolean stop = false;
022    private boolean healthy = false;
023    private final Client client;
024
025    /**
026     * Constructor.
027     * @param client Client instance to check health with.
028     */
029    public HealthMonitorThread(Client client) {
030        this.client = client;
031    }
032
033    @Override
034    public void run() {
035        while (!stop) {
036            healthy = false;
037            int curTime = 0;
038
039            while (!healthy) {
040                ClusterHealthResponse health = client.admin().cluster().prepareHealth()
041                    .setWaitForYellowStatus()
042                    .setTimeout(TimeValue.timeValueMinutes(1))
043                    .execute().actionGet();
044                if (health.isTimedOut()) {
045                    logger.info("Waiting for cluster ({} minutes elapsed)...", curTime);
046                    curTime++;
047                } else {
048                    healthy = true;
049                }
050            }
051
052            try {
053                Thread.sleep(TimeValue.timeValueMinutes(5).getMillis());
054            } catch (InterruptedException e) {
055                Thread.interrupted();
056                if (!stop) {
057                    logger.warn("Thread interrupted unexpectedly, stopping", e);
058                    stop = true;
059                }
060            }
061        }
062    }
063
064    /**
065     * Stops this thread.
066     */
067    public void stop() {
068        stop = true;
069    }
070
071    /**
072     * @return True if the cluster is healthy.
073     */
074    public boolean isHealthy() {
075        return healthy;
076    }
077
078}
