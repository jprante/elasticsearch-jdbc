019package org.xbib.elasticsearch.river.jdbc.support;
020
021import org.elasticsearch.common.unit.TimeValue;
022import org.xbib.elasticsearch.river.jdbc.RiverMouth;
023import org.xbib.elasticsearch.river.jdbc.RiverSource;
024import org.xbib.elasticsearch.river.jdbc.RiverMouth;
025import org.xbib.elasticsearch.river.jdbc.RiverSource;
026
027import java.util.List;
028import java.util.Locale;
029import java.util.Map;
030
031/**
032 * The river context consists of the parameters that span over source and target, and the source and target.
033 * It represents the river state, for supporting the river task, and river scripting.
034 *
035 * @author JÃ¶rg Prante <joergprante@gmail.com>
036 */
037public class RiverContext {
038
039    /**
040     * The name of the river.
041     */
042    private String name;
043    /**
044     * The river index name
045     */
046    private String riverIndexName;
047    /**
048     * The settings of the river
049     */
050    private Map<String, Object> settings;
051    /**
052     * The source of the river
053     */
054    private RiverSource source;
055    /**
056     * The target of the river
057     */
058    private RiverMouth mouth;
059    /**
060     * The polling interval
061     */
062    private TimeValue poll;
063
064    /**
065     * The job name of the current river task
066     */
067    private String job;
068    /**
069     * The SQL statement
070     */
071    private String sql;
072    /**
073     * Parameters for the SQL statement
074     */
075    private List<? extends Object> sqlparams;
076    /**
077     * The acknowledge SQL statement
078     */
079    private String acksql;
080    /**
081     * Parameters for the acknowledge SQL statement
082     */
083    private List<? extends Object> acksqlparams;
084    /**
085     * Autocomit enabled or not
086     */
087    private boolean autocommit;
088    /**
089     * The fetch size
090     */
091    private int fetchSize;
092    /**
093     * The maximum numbe rof rows per statement execution
094     */
095    private int maxRows;
096    /**
097     * The number of retries
098     */
099    private int retries;
100    /**
101     * The time to wait between retries
102     */
103    private TimeValue maxretrywait;
104
105    /**
106     * The locale for numerical format
107     */
108    private Locale locale;
109
110    /**
111     * If digesting should be used or not
112     */
113    private boolean digesting;
114
115    public RiverContext riverSettings(Map<String, Object> settings) {
116        this.settings = settings;
117        return this;
118    }
119
120    public Map<String, Object> riverSettings() {
121        return settings;
122    }
123
124    public RiverContext riverIndexName(String name) {
125        this.riverIndexName = name;
126        return this;
127    }
128
129    public String riverIndexName() {
130        return riverIndexName;
131    }
132
133    public RiverContext riverName(String name) {
134        this.name = name;
135        return this;
136    }
137
138    public String riverName() {
139        return name;
140    }
141
142    public RiverContext riverSource(RiverSource source) {
143        this.source = source;
144        return this;
145    }
146
147    public RiverSource riverSource() {
148        return source;
149    }
150
151    public RiverContext riverMouth(RiverMouth mouth) {
152        this.mouth = mouth;
153        return this;
154    }
155
156    public RiverMouth riverMouth() {
157        return mouth;
158    }
159
160    public RiverContext job(String job) {
161        this.job = job;
162        return this;
163    }
164
165    public String job() {
166        return job;
167    }
168
169    public RiverContext pollInterval(TimeValue poll) {
170        this.poll = poll;
171        return this;
172    }
173
174    public TimeValue pollingInterval() {
175        return poll;
176    }
177
178    public RiverContext pollStatement(String sql) {
179        this.sql = sql;
180        return this;
181    }
182
183    public String pollStatement() {
184        return sql;
185    }
186
187    public RiverContext pollStatementParams(List<? extends Object> params) {
188        this.sqlparams = params;
189        return this;
190    }
191
192    public List<? extends Object> pollStatementParams() {
193        return sqlparams;
194    }
195
196    public RiverContext pollAckStatement(String acksql) {
197        this.acksql = acksql;
198        return this;
199    }
200
201    public String pollAckStatement() {
202        return acksql;
203    }
204
205    public RiverContext pollAckStatementParams(List<? extends Object> params) {
206        this.acksqlparams = params;
207        return this;
208    }
209
210    public List<? extends Object> pollAckStatementParams() {
211        return acksqlparams;
212    }
213
214    public RiverContext autocommit(boolean enabled) {
215        this.autocommit = enabled;
216        return this;
217    }
218
219    public boolean autocommit() {
220        return autocommit;
221    }
222
223    public RiverContext fetchSize(int fetchSize) {
224        this.fetchSize = fetchSize;
225        return this;
226    }
227
228    public int fetchSize() {
229        return fetchSize;
230    }
231
232    public RiverContext maxRows(int maxRows) {
233        this.maxRows = maxRows;
234        return this;
235    }
236
237    public int maxRows() {
238        return maxRows;
239    }
240
241    public RiverContext retries(int retries) {
242        this.retries = retries;
243        return this;
244    }
245
246    public int retries() {
247        return retries;
248    }
249
250    public RiverContext maxRetryWait(TimeValue maxretrywait) {
251        this.maxretrywait = maxretrywait;
252        return this;
253    }
254
255    public TimeValue maxRetryWait() {
256        return maxretrywait;
257    }
258
259    public RiverContext locale(String languageTag) {
260        this.locale = LocaleUtil.toLocale(languageTag);
261        Locale.setDefault(locale); // for JDBC drivers internals
262        return this;
263    }
264
265    public Locale locale() {
266        return locale;
267    }
268
269    public RiverContext digesting(boolean digesting) {
270        this.digesting = digesting;
271        return this;
272    }
273
274    public boolean digesting() {
275        return digesting;
276    }
277
278    public RiverContext contextualize() {
279        if (source != null) {
280            source.riverContext(this);
281        }
282        if (mouth != null) {
283            mouth.riverContext(this);
284        }
285        return this;
286    }
287}
