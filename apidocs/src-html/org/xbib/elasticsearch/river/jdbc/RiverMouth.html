001package org.xbib.elasticsearch.river.jdbc;
002
003import org.xbib.elasticsearch.plugin.jdbc.IndexableObject;
004import org.xbib.elasticsearch.plugin.jdbc.RiverContext;
005import org.xbib.elasticsearch.support.client.Ingest;
006
007import java.io.IOException;
008
009/**
010 * The river mouth is the abstraction of the destination where all the data
011 * is flowing from the river source.
012 */
013public interface RiverMouth {
014
015    /**
016     * The river strategy
017     *
018     * @return the strategy
019     */
020    String strategy();
021
022    /**
023     * The river context
024     *
025     * @param context the river context
026     * @return this river mouth
027     */
028    RiverMouth setRiverContext(RiverContext context);
029
030    /**
031     * Set the Elasticsearch ingester
032     *
033     * @param ingester the ingester
034     * @return this river mouth
035     */
036    RiverMouth setIngest(Ingest ingester);
037
038    RiverMouth setIndex(String index);
039
040    RiverMouth setType(String type);
041
042    RiverMouth setTimeWindowed(boolean timeWindowed);
043
044    RiverMouth setId(String id);
045
046    String getId();
047
048    /**
049     * Index. Indicating that an object has been built and is ready to be indexed.
050     * The source of the document is held by the XContentBuilder.
051     *
052     * @param object the indexable object
053     * @param create true if the document should be created
054     * @throws IOException when indexing fails
055     */
056    void index(IndexableObject object, boolean create) throws IOException;
057
058    /**
059     * Delete. Indicating that an object should be deleted from the index.
060     *
061     * @param object the structured object
062     * @throws IOException when delete fails
063     */
064    void delete(IndexableObject object) throws IOException;
065
066    /**
067     * Flush data to the river mouth
068     *
069     * @throws IOException when flush fails
070     */
071    void flush() throws IOException;
072
073    /**
074     * Close this river mouth
075     */
076    void close() throws IOException;
077
078}
