001
002package org.xbib.elasticsearch.river.jdbc;
003
004import java.io.IOException;
005import java.util.Map;
006
007import org.elasticsearch.client.Client;
008import org.elasticsearch.common.unit.ByteSizeValue;
009import org.elasticsearch.common.unit.TimeValue;
010import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
011import org.xbib.elasticsearch.gatherer.IndexableObject;
012
013/**
014 * The river mouth is the abstraction of the destination where all the data
015 * is flowing from the river source.
016 */
017public interface RiverMouth {
018
019    /**
020     * The river strategy
021     *
022     * @return the strategy
023     */
024    String strategy();
025
026    /**
027     * The river context
028     *
029     * @param context the river context
030     * @return this river mouth
031     */
032    RiverMouth riverContext(RiverContext context);
033
034    /**
035     * Set the Elasticsearch client
036     *
037     * @param client the client
038     * @return this river mouth
039     */
040    RiverMouth client(Client client);
041
042    /**
043     * Return the Elasticsearch client
044     *
045     * @return this river mouth
046     */
047    Client client();
048
049    /**
050     * Set settings for Elasticsearch to be used in index creation
051     * @param settings the settings
052     * @return this river mouth
053     */
054    RiverMouth setSettings(Map<String,Object> settings);
055
056    /**
057     * Set mapping for Elasticsearch to be used in mapping creation
058     * @param mapping the mapping
059     * @return this river mouth
060     */
061    RiverMouth setMapping(Map<String,Object> mapping);
062
063    /**
064     * Set index
065     *
066     * @param index the index
067     * @return this river mouth
068     */
069    RiverMouth setIndex(String index);
070
071    /**
072     * Return index
073     *
074     * @return the index
075     */
076    String getIndex();
077
078    /**
079     * Set type
080     *
081     * @param type the type
082     * @return this river mouth
083     */
084    RiverMouth setType(String type);
085
086    /**
087     * Return type
088     *
089     * @return the type
090     */
091    String getType();
092
093    /**
094     * Set ID
095     *
096     * @param id the id
097     * @return this river mouth
098     */
099    RiverMouth setId(String id);
100
101    String getId();
102
103    /**
104     * Set maximum number of bulk actions
105     *
106     * @param actions the number of bulk actions
107     * @return this river mouth
108     */
109    RiverMouth setMaxBulkActions(int actions);
110
111    /**
112     * Set maximum number of concurrent bulk requests
113     *
114     * @param max the  maximum number of concurrent bulk requests
115     * @return this river mouth
116     */
117    RiverMouth setMaxConcurrentBulkRequests(int max);
118
119    /**
120     * Set maximum volume of a bulk request
121     * @param maxVolumePerBulkRequest the maxiumum byte size of a buk request
122     * @return this river mouth
123     */
124    RiverMouth setMaxVolumePerBulkRequest(ByteSizeValue maxVolumePerBulkRequest);
125
126    /**
127     * Set flush interval
128     * @param flushInterval flush interval
129     * @return this river mouth
130     */
131    RiverMouth setFlushInterval(TimeValue flushInterval);
132
133    /**
134     * Index. Indicating that an object has been built and is ready to be indexed.
135     * The source of the document is held by the XContentBuilder.
136     *
137     * @param object the indexable object
138     * @param create true if the document should be created
139     * @throws IOException
140     */
141    void index(IndexableObject object, boolean create) throws IOException;
142
143    /**
144     * Delete. Indicating that an object should be deleted from the index.
145     *
146     * @param object the structured object
147     * @throws IOException
148     */
149    void delete(IndexableObject object) throws IOException;
150
151    /**
152     * Flush data to the river mouth
153     *
154     * @throws IOException
155     */
156    void flush() throws IOException;
157
158    /**
159     * Close this river mouth
160     */
161    void close();
162
163    void waitForCluster() throws IOException;
164
165}
