001    
002    package org.xbib.elasticsearch.river.jdbc;
003    
004    import java.io.IOException;
005    import java.util.Map;
006    
007    import org.elasticsearch.client.Client;
008    import org.elasticsearch.common.unit.TimeValue;
009    import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
010    import org.xbib.elasticsearch.gatherer.IndexableObject;
011    
012    /**
013     * The river mouth is the abstraction of the destination where all the data
014     * is flowing from the river source.
015     */
016    public interface RiverMouth {
017    
018        /**
019         * The river strategy
020         *
021         * @return the strategy
022         */
023        String strategy();
024    
025        /**
026         * The river context
027         *
028         * @param context the river context
029         * @return this river mouth
030         */
031        RiverMouth riverContext(RiverContext context);
032    
033        /**
034         * Set the Elasticsearch client
035         *
036         * @param client the client
037         * @return this river mouth
038         */
039        RiverMouth client(Client client);
040    
041        /**
042         * Return the Elasticsearch client
043         *
044         * @return this river mouth
045         */
046        Client client();
047    
048        /**
049         * Set settings for Elasticsearch to be used in index creation
050         * @param settings the settings
051         * @return this river mouth
052         */
053        RiverMouth setSettings(Map<String,Object> settings);
054    
055        /**
056         * Set mapping for Elasticsearch to be used in mapping creation
057         * @param mapping the mapping
058         * @return this river mouth
059         */
060        RiverMouth setMapping(Map<String,Object> mapping);
061    
062        /**
063         * Set index
064         *
065         * @param index the index
066         * @return this river mouth
067         */
068        RiverMouth setIndex(String index);
069    
070        /**
071         * Return index
072         *
073         * @return the index
074         */
075        String getIndex();
076    
077        /**
078         * Set type
079         *
080         * @param type the type
081         * @return this river mouth
082         */
083        RiverMouth setType(String type);
084    
085        /**
086         * Return type
087         *
088         * @return the type
089         */
090        String getType();
091    
092        /**
093         * Set ID
094         *
095         * @param id the id
096         * @return this river mouth
097         */
098        RiverMouth setId(String id);
099    
100        String getId();
101    
102        /**
103         * Set maximum number of bulk actions
104         *
105         * @param actions the number of bulk actions
106         * @return this river mouth
107         */
108        RiverMouth setMaxBulkActions(int actions);
109    
110        /**
111         * Set maximum number of concurrent bulk requests
112         *
113         * @param max the  maximum number of concurrent bulk requests
114         * @return this river mouth
115         */
116        RiverMouth setMaxConcurrentBulkRequests(int max);
117    
118        /**
119         * Set flish interval
120         * @param flushInterval flush interval
121         * @return this river mouth
122         */
123        RiverMouth setFlushInterval(TimeValue flushInterval);
124    
125        /**
126         * Index. Indicating that an object has been built and is ready to be indexed.
127         * The source of the document is held by the XContentBuilder.
128         *
129         * @param object the indexable object
130         * @param create true if the document should be created
131         * @throws IOException
132         */
133        void index(IndexableObject object, boolean create) throws IOException;
134    
135        /**
136         * Delete. Indicating that an object should be deleted from the index.
137         *
138         * @param object the structured object
139         * @throws IOException
140         */
141        void delete(IndexableObject object) throws IOException;
142    
143        /**
144         * Flush data to the river mouth
145         *
146         * @throws IOException
147         */
148        void flush() throws IOException;
149    
150        /**
151         * Close this river mouth
152         */
153        void close();
154    
155        void waitForCluster() throws IOException;
156    
157    }
