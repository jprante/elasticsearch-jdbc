019package org.xbib.elasticsearch.river.jdbc;
020
021import org.elasticsearch.client.Client;
022import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
023import org.xbib.elasticsearch.river.jdbc.support.StructuredObject;
024import org.xbib.elasticsearch.river.jdbc.support.StructuredObject;
025
026import java.io.IOException;
027
028/**
029 * The river mouth is the abstraction of the destination where all the data
030 * is flowing from the river source.
031 *
032 * @author JÃ¶rg Prante <joergprante@gmail.com>
033 */
034public interface RiverMouth {
035
036    /**
037     * The river strategy
038     *
039     * @return the strategy
040     */
041    String strategy();
042
043    /**
044     * The river context
045     *
046     * @param context the river context
047     * @return this river mouth
048     */
049    RiverMouth riverContext(RiverContext context);
050
051    /**
052     * Set the Elasticsearch client
053     *
054     * @param client the client
055     * @return this river mouth
056     */
057    RiverMouth client(Client client);
058
059    /**
060     * Return the Elasticsearch client
061     *
062     * @return this river mouth
063     */
064    Client client();
065
066    /**
067     * Set index
068     *
069     * @param index the index
070     * @return this river mouth
071     */
072    RiverMouth index(String index);
073
074    /**
075     * Return index
076     * @return the index
077     */
078    String index();
079
080    /**
081     * Set type
082     * @param type the type
083     * @return this river mouth
084     */
085    RiverMouth type(String type);
086
087    /**
088     * Return type
089     * @return the type
090     */
091    String type();
092
093    /**
094     * Set ID
095     * @param id the id
096     * @return this river mouth
097     */
098    RiverMouth id(String id);
099
100    /**
101     * Set maximum number of bulk actions
102     *
103     * @param actions the number of bulk actions
104     * @return this river mouth
105     */
106    RiverMouth maxBulkActions(int actions);
107
108    /**
109     * Get maximum number of bulk actions
110     * @return max bulk actions
111     */
112    int maxBulkActions();
113
114    /**
115     * Set maximum number of concurrent bulk requests
116     * @param max the  maximum number of concurrent bulk requests
117     * @return this river mouth
118     */
119    RiverMouth maxConcurrentBulkRequests(int max);
120
121    /**
122     * Get maximum concurrent bulk requests
123     * @return maximum concurrent requests
124     */
125    int maxConcurrentBulkRequests();
126
127    /**
128     * Set versioning
129     * @param enable true if versioning
130     * @return this river mouth
131     */
132    RiverMouth versioning(boolean enable);
133
134    /**
135     * Get versioning
136     * @return the versioning
137     */
138    boolean versioning();
139
140    /**
141     * Set acknowledge
142     * @param enable true for enable
143     * @return this river mouth
144     */
145    RiverMouth acknowledge(boolean enable);
146
147    /**
148     * Get acknowledge
149     * @return true if the acknowledge is enabled
150     */
151    boolean acknowledge();
152
153    /**
154     * Create.  Indicating that an object has been built and is ready to be indexed.
155     * The source of the document is held by the XContentBuilder.
156     * The document is only indexed if no document with that index/type/id exists.
157     *
158     * @param object the structured object
159     * @throws IOException
160     */
161    void create(StructuredObject object) throws IOException;
162
163    /**
164     * Index. Indicating that an object has been built and is ready to be indexed.
165     * The source of the document is held by the XContentBuilder.
166     *
167     * @param object the indexable object
168     * @throws IOException
169     */
170    void index(StructuredObject object) throws IOException;
171
172    /**
173     * Delete. Indicating that an object should be deleted from the index.
174     *
175     * @param object the structured object
176     * @throws IOException
177     */
178    void delete(StructuredObject object) throws IOException;
179
180    /**
181     * Flush data to the river mouth
182     *
183     * @throws IOException
184     */
185    void flush() throws IOException;
186
187    /**
188     * Close this river mouth
189     */
190    void close();
191
192    /**
193     * Create index with optional settings and mapping
194     * @param settings the settings
195     * @param mapping the mapping
196     */
197    void createIndexIfNotExists(String settings, String mapping);
198
199}
