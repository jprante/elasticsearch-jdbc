002package org.xbib.elasticsearch.river.jdbc;
003
004import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
005
006/**
007 * RiverFlow API.
008 * The RiverFlow is the abstraction to the thread which
009 * performs data fetching from the river source and transports it
010 * to the river mouth
011 *
012 */
013public interface RiverFlow extends Runnable {
014
015    /**
016     * The doc ID of the info document in the river index
017     */
018    String ID_INFO_RIVER_INDEX = "_custom";
019
020    /**
021     * The strategy name
022     *
023     * @return the strategy of this river task
024     */
025    String strategy();
026
027    /**
028     * Set river context
029     *
030     * @param context the context
031     * @return this river flow
032     */
033    RiverFlow riverContext(RiverContext context);
034
035    /**
036     * Get river context
037     *
038     * @return the river context
039     */
040    RiverContext riverContext();
041
042    /**
043     * Get river state
044     * @return the river state
045     */
046    RiverState riverState();
047
048    /**
049     * Schedule thread
050     * @param thread the thread
051     */
052    void schedule(Thread thread);
053
054    /**
055     * Run thread once
056     * @param thread the thread
057     */
058    void once(Thread thread);
059
060    /**
061     * Run river once
062     */
063    void move();
064
065    /**
066     * Abort river task. Set signal to interrupt thread and free resources.
067     */
068    void abort();
069
070    boolean isActive();
071
072}
