019package org.xbib.elasticsearch.river.jdbc;
020
021import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
022
023import java.util.Date;
024
025/**
026 * RiverFlow fluent API
027 *
028 * The RiverFlow is the abstraction to the thread which
029 * performs data fetching from the river source and transports it
030 * to the river mouth
031 *
032 * @author JÃ¶rg Prante <joergprante@gmail.com>
033 */
034public interface RiverFlow extends Runnable {
035
036    /**
037     * The doc ID of the info document in the river index
038     */
039    String ID_INFO_RIVER_INDEX = "_custom";
040
041    /**
042     * The strategy
043     *
044     * @return the strategy of this river task
045     */
046    String strategy();
047
048    /**
049     * Set river context
050     *
051     * @param context the context
052     * @return this river flow
053     */
054    RiverFlow riverContext(RiverContext context);
055
056    /**
057     * Get river context
058     *
059     * @return the river context
060     */
061    RiverContext riverContext();
062
063    /**
064     * Set river connection start date
065     *
066     * @param creationDate the creation date
067     * @return this river flow
068     */
069    RiverFlow startDate(Date creationDate);
070
071    /**
072     * Get river connection start date
073     *
074     * @return the date
075     */
076    Date startDate();
077
078    /**
079     * Run river once
080     */
081    void move();
082
083    /**
084     * Delay between river actions.
085     *
086     * @param reason the reson
087     * @return this river flow
088     */
089    RiverFlow delay(String reason);
090
091    /**
092     * Abort river task. Set signal to interrupt thread and free resources.
093     */
094    void abort();
095
096}
