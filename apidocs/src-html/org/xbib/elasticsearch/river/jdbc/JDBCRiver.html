046/**
047 * The JDBC river
048 *
049 * @author JÃ¶rg Prante <joergprante@gmail.com>
050 */
051public class JDBCRiver extends AbstractRiverComponent implements River {
052
053    public final static String NAME = "jdbc-river";
054    public final static String TYPE = "jdbc";
055
056    private final String strategy;
057    private final TimeValue poll;
058    private final String url;
059    private final String driver;
060    private final String user;
061    private final String password;
062    private final String rounding;
063    private final int scale;
064    private final String sql;
065    private final List<? super Object> sqlparams;
066    private final String acksql;
067    private final List<? super Object> acksqlparams;
068    private final boolean autocommit;
069    private final int fetchsize;
070    private final int maxrows;
071    private final int maxretries;
072    private final TimeValue maxretrywait;
073    private final String locale;
074    private final String indexName;
075    private final String typeName;
076    private final int bulkSize;
077    private final int maxBulkRequests;
078    private final String indexSettings;
079    private final String typeMapping;
080    private final boolean versioning;
081    private final boolean digesting;
082    private final boolean acknowledgeBulk;
083    private final RiverSource riverSource;
084    private final RiverMouth riverMouth;
085    private final RiverContext riverContext;
086    private final RiverFlow riverFlow;
087    private volatile Thread thread;
088    private volatile boolean closed;
089
090    @Inject
091    public JDBCRiver(RiverName riverName, RiverSettings riverSettings,
092                     @RiverIndexName String riverIndexName,
093                     Client client) {
094        super(riverName, riverSettings);
095        // riverIndexName = _river
096
097        Map<String, Object> sourceSettings =
098                riverSettings.settings().containsKey(TYPE)
099                        ? (Map<String, Object>) riverSettings.settings().get(TYPE)
100                        : new HashMap<String, Object>();
101        // default is a single run
102        strategy = XContentMapValues.nodeStringValue(sourceSettings.get("strategy"), "oneshot");
103        url = XContentMapValues.nodeStringValue(sourceSettings.get("url"), null);
104        driver = XContentMapValues.nodeStringValue(sourceSettings.get("driver"), null);
105        user = XContentMapValues.nodeStringValue(sourceSettings.get("user"), null);
106        password = XContentMapValues.nodeStringValue(sourceSettings.get("password"), null);
107        poll = XContentMapValues.nodeTimeValue(sourceSettings.get("poll"), TimeValue.timeValueMinutes(60));
108        sql = XContentMapValues.nodeStringValue(sourceSettings.get("sql"), null);
109        sqlparams = XContentMapValues.extractRawValues("sqlparams", sourceSettings);
110        rounding = XContentMapValues.nodeStringValue(sourceSettings.get("rounding"), null);
111        scale = XContentMapValues.nodeIntegerValue(sourceSettings.get("scale"), 2);
112        autocommit = XContentMapValues.nodeBooleanValue(sourceSettings.get("autocommit"), Boolean.FALSE);
113        fetchsize = url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
114                XContentMapValues.nodeIntegerValue(sourceSettings.get("fetchsize"), 10);
115        maxrows = XContentMapValues.nodeIntegerValue(sourceSettings.get("max_rows"), 0);
116        maxretries = XContentMapValues.nodeIntegerValue(sourceSettings.get("max_retries"), 3);
117        maxretrywait = TimeValue.parseTimeValue(XContentMapValues.nodeStringValue(sourceSettings.get("max_retries_wait"), "10s"), TimeValue.timeValueMillis(30000));
118        locale = XContentMapValues.nodeStringValue(sourceSettings.get("locale"), LocaleUtil.fromLocale(Locale.getDefault()));
119        digesting = XContentMapValues.nodeBooleanValue(sourceSettings.get("digesting"), Boolean.TRUE);
120        acksql = XContentMapValues.nodeStringValue(sourceSettings.get("acksql"), null);
121        acksqlparams = XContentMapValues.extractRawValues("acksqlparams", sourceSettings);
122
123        Map<String, Object> targetSettings =
124                riverSettings.settings().containsKey("index")
125                        ? (Map<String, Object>) riverSettings.settings().get("index")
126                        : new HashMap<String, Object>();
127        indexName = XContentMapValues.nodeStringValue(targetSettings.get("index"), TYPE);
128        typeName = XContentMapValues.nodeStringValue(targetSettings.get("type"), TYPE);
129        bulkSize = XContentMapValues.nodeIntegerValue(targetSettings.get("bulk_size"), 100);
130        maxBulkRequests = XContentMapValues.nodeIntegerValue(targetSettings.get("max_bulk_requests"), 30);
131        indexSettings = XContentMapValues.nodeStringValue(targetSettings.get("index_settings"), null);
132        typeMapping = XContentMapValues.nodeStringValue(targetSettings.get("type_mapping"), null);
133        versioning = XContentMapValues.nodeBooleanValue(sourceSettings.get("versioning"), Boolean.FALSE);
134        acknowledgeBulk = XContentMapValues.nodeBooleanValue(sourceSettings.get("acknowledge"), Boolean.FALSE);
135
136        riverSource = RiverServiceLoader.findRiverSource(strategy);
137        logger.debug("found river source {} for strategy {}", riverSource.getClass().getName(), strategy);
138        riverSource.driver(driver)
139                .url(url)
140                .user(user)
141                .password(password)
142                .rounding(rounding)
143                .precision(scale);
144
145        riverMouth = RiverServiceLoader.findRiverMouth(strategy);
146        logger.debug("found river mouth {} for strategy {}", riverMouth.getClass().getName(), strategy);
147        riverMouth.index(indexName)
148                .type(typeName)
149                .maxBulkActions(bulkSize)
150                .maxConcurrentBulkRequests(maxBulkRequests)
151                .acknowledge(acknowledgeBulk)
152                .versioning(versioning)
153                .client(client);
154
155        // scripting ...
156
157        riverContext = new RiverContext()
158                .riverName(riverName.getName())
159                .riverIndexName(riverIndexName)
160                .riverSettings(riverSettings.settings())
161                .riverSource(riverSource)
162                .riverMouth(riverMouth)
163                .pollInterval(poll)
164                .pollStatement(sql)
165                .pollStatementParams(sqlparams)
166                .pollAckStatement(acksql)
167                .pollAckStatementParams(acksqlparams)
168                .autocommit(autocommit)
169                .maxRows(maxrows)
170                .fetchSize(fetchsize)
171                .retries(maxretries)
172                .maxRetryWait(maxretrywait)
173                .locale(locale)
174                .digesting(digesting)
175                .contextualize();
176
177        riverFlow = RiverServiceLoader.findRiverFlow(strategy);
178        // prepare task for run
179        riverFlow.riverContext(riverContext);
180
181        logger.debug("found river flow {} for strategy {}", riverFlow.getClass().getName(), strategy);
182    }
183
184    @Override
185    public void start() {
186        logger.info("starting JDBC river: URL [{}], driver [{}], strategy [{}], index [{}]/[{}]",
187                url, driver, strategy, indexName, typeName);
188        try {
189            riverFlow.startDate(new Date());
190            riverMouth.createIndexIfNotExists(indexSettings, typeMapping);
191        } catch (Exception e) {
192            if (ExceptionsHelper.unwrapCause(e) instanceof IndexAlreadyExistsException) {
193                riverFlow.startDate(null);
194                // that's fine, continue.
195            } else if (ExceptionsHelper.unwrapCause(e) instanceof ClusterBlockException) {
196                // ok, not recovered yet..., lets start indexing and hope we recover by the first bulk
197            } else {
198                logger.warn("failed to create index [{}], disabling JDBC river...", e, indexName);
199                return;
200            }
201        }
202        thread = EsExecutors.daemonThreadFactory(settings.globalSettings(), "JDBC river [" + riverName.name() + '/' + strategy + ']')
203                .newThread(riverFlow);
204        thread.start();
205    }
206
207    @Override
208    public void close() {
209        if (closed) {
210            return;
211        }
212        logger.info("closing JDBC river [" + riverName.name() + '/' + strategy + ']');
213        if (thread != null) {
214            thread.interrupt();
215        }
216        if (riverFlow != null) {
217            riverFlow.abort();
218        }
219        if (riverSource != null) {
220            riverSource.closeReading();
221            riverSource.closeWriting();
222        }
223        if (riverMouth != null) {
224            riverMouth.close();
225        }
226        closed = true; // abort only once
227    }
228
229    /**
230     * Induce a river run once, but in a synchronous manner.
231     */
232    public void once() {
233        if (riverFlow != null) {
234            riverFlow.move();
235        }
236    }
237
238    /**
239     * Induce a river run once, but in an asynchronous manner.
240     */
241    public void induce() {
242        RiverFlow riverTask = RiverServiceLoader.findRiverFlow(strategy);
243        // prepare task for run
244        riverTask.riverContext(riverContext);
245        thread = EsExecutors.daemonThreadFactory(settings.globalSettings(), "JDBC river (fired) [" + riverName.name() + '/' + strategy + ')')
246                .newThread(riverTask);
247        riverTask.abort();
248        thread.start(); // once
249    }
