002package org.xbib.elasticsearch.river.jdbc;
003
004import java.util.List;
005import java.util.Locale;
006import java.util.Map;
007
008import org.elasticsearch.client.Client;
009import org.elasticsearch.common.inject.Inject;
010import org.elasticsearch.common.unit.TimeValue;
011import org.elasticsearch.common.util.concurrent.EsExecutors;
012import org.elasticsearch.common.xcontent.support.XContentMapValues;
013import org.elasticsearch.river.AbstractRiverComponent;
014import org.elasticsearch.river.River;
015import org.elasticsearch.river.RiverName;
016import org.elasticsearch.river.RiverSettings;
017
018import org.xbib.elasticsearch.river.jdbc.support.LocaleUtil;
019import org.xbib.elasticsearch.river.jdbc.support.RiverContext;
020import org.xbib.elasticsearch.river.jdbc.support.RiverServiceLoader;
021import org.xbib.elasticsearch.river.jdbc.support.SQLCommand;
022
023import static org.elasticsearch.common.collect.Maps.newHashMap;
024
025/**
026 * The JDBC river
027 */
028public class JDBCRiver extends AbstractRiverComponent implements River {
029
030    public final static String NAME = "river-jdbc";
031
032    public final static String TYPE = "jdbc";
033
034    private final String strategy;
035
036    private final String url;
037
038    private final String indexName;
039
040    private final String typeName;
041
042    private final RiverSource riverSource;
043
044    private final RiverMouth riverMouth;
045
046    private final RiverFlow riverFlow;
047
048    private final RiverContext riverContext;
049
050    private volatile boolean closed;
051
052    @Inject
053    public JDBCRiver(RiverName riverName,
054                     RiverSettings riverSettings,
055                     Client client) {
056        super(riverName, riverSettings);
057        logger.debug("JDBC river initializing");
058
059        Map<String, Object> mySettings = newHashMap();
060        if (riverSettings.settings().containsKey(TYPE)) {
061            mySettings = (Map<String, Object>) riverSettings.settings().get(TYPE);
062        }
063        strategy = XContentMapValues.nodeStringValue(mySettings.get("strategy"), "simple");
064        String schedule = mySettings.containsKey("schedule") ?
065            XContentMapValues.nodeStringValue(mySettings.get("schedule"), null) :
066            null;
067        Integer poolsize = XContentMapValues.nodeIntegerValue(mySettings.get("poolsize"), 1);
068        // disable interval
069        TimeValue interval = XContentMapValues.nodeTimeValue(mySettings.get("interval"), TimeValue.timeValueMinutes(-1));
070
071        // JDBC
072        url = XContentMapValues.nodeStringValue(mySettings.get("url"), null);
073        String user = XContentMapValues.nodeStringValue(mySettings.get("user"), null);
074        String password = XContentMapValues.nodeStringValue(mySettings.get("password"), null);
075        List<SQLCommand> sql = SQLCommand.parse(mySettings);
076        String rounding = XContentMapValues.nodeStringValue(mySettings.get("rounding"), null);
077        int scale = XContentMapValues.nodeIntegerValue(mySettings.get("scale"), 2);
078        boolean autocommit = XContentMapValues.nodeBooleanValue(mySettings.get("autocommit"), Boolean.FALSE);
079
080        int fetchsize = url.startsWith("jdbc:mysql") ? Integer.MIN_VALUE :
081                XContentMapValues.nodeIntegerValue(mySettings.get("fetchsize"), 10);
082        int maxrows = XContentMapValues.nodeIntegerValue(mySettings.get("max_rows"), 0);
083        int maxretries = XContentMapValues.nodeIntegerValue(mySettings.get("max_retries"), 3);
084        TimeValue maxretrywait =
085                XContentMapValues.nodeTimeValue(mySettings.get("max_retries_wait"), TimeValue.timeValueSeconds(30));
086        String locale = XContentMapValues.nodeStringValue(mySettings.get("locale"), LocaleUtil.fromLocale(Locale.getDefault()));
087
088        // defaults for column strategy
089        String columnCreatedAt = XContentMapValues.nodeStringValue(mySettings.get("column_created_at"), "created_at");
090        String columnUpdatedAt = XContentMapValues.nodeStringValue(mySettings.get("column_updated_at"), "updated_at");
091        String columnDeletedAt = XContentMapValues.nodeStringValue(mySettings.get("column_deleted_at"), null);
092        boolean columnEscape = XContentMapValues.nodeBooleanValue(mySettings.get("column_escape"), true);
093
094        // set up bulk indeser
095        indexName = XContentMapValues.nodeStringValue(mySettings.get("index"), TYPE);
096        typeName = XContentMapValues.nodeStringValue(mySettings.get("type"), TYPE);
097        int bulkSize = XContentMapValues.nodeIntegerValue(mySettings.get("bulk_size"), 100);
098        int maxBulkRequests = XContentMapValues.nodeIntegerValue(mySettings.get("max_bulk_requests"), 30);
099        // flush interval for bulk indexer
100        TimeValue flushInterval = XContentMapValues.nodeTimeValue(mySettings.get("bulk_flush_interval"),
101                TimeValue.timeValueSeconds(5));
102        String indexSettings = XContentMapValues.nodeStringValue(mySettings.get("index_settings"), null);
103        String typeMapping = XContentMapValues.nodeStringValue(mySettings.get("type_mapping"), null);
104
105        riverSource = RiverServiceLoader.findRiverSource(strategy);
106        logger.debug("found river source class {} for strategy {}", riverSource.getClass().getName(), strategy);
107        riverSource.url(url)
108                .user(user)
109                .password(password)
110                .rounding(rounding)
111                .precision(scale);
112
113        riverMouth = RiverServiceLoader.findRiverMouth(strategy);
114        logger.debug("found river mouth class {} for strategy {}", riverMouth.getClass().getName(), strategy);
115        riverMouth.setSettings(indexSettings)
116                .setMapping(typeMapping)
117                .setIndex(indexName)
118                .setType(typeName)
119                .setMaxBulkActions(bulkSize)
120                .setMaxConcurrentBulkRequests(maxBulkRequests)
121                .setFlushInterval(flushInterval)
122                .client(client);
123
124        riverContext = new RiverContext()
125                .riverName(riverName.getName())
126                .riverSettings(riverSettings.settings())
127                .riverSource(riverSource)
128                .riverMouth(riverMouth)
129                .locale(locale)
130                .setSchedule(schedule)
131                .setPoolSize(poolsize)
132                .setInterval(interval)
133                .setStatements(sql)
134                .setAutoCommit(autocommit)
135                .setMaxRows(maxrows)
136                .setFetchSize(fetchsize)
137                .setRetries(maxretries)
138                .setMaxRetryWait(maxretrywait)
139                .columnCreatedAt(columnCreatedAt)
140                .columnUpdatedAt(columnUpdatedAt)
141                .columnDeletedAt(columnDeletedAt)
142                .columnEscape(columnEscape)
143                .contextualize();
144
145        riverFlow = RiverServiceLoader.findRiverFlow(strategy);
146
147        // prepare task for run
148        riverFlow.riverContext(riverContext);
149        logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
150
151        logger.debug("JDBC river initialized");
152
153    }
154
155    @Override
156    public void start() {
157        logger.info("starting JDBC river: URL [{}], strategy [{}], index/type [{}/{}]",
158                url, strategy, indexName, typeName);
159        Thread thread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
160                "jdbc-river-[" + riverName.name() + '/' + strategy + ']')
161                .newThread(riverFlow);
162        riverFlow.schedule(thread);
163    }
164
165    @Override
166    public void close() {
167        if (closed) {
168            return;
169        }
170        closed = true;
171        logger.info("closing JDBC river: URL [{}], strategy [{}], index/type [{}/{}]",
172                url, strategy, indexName, typeName);
173        if (riverFlow != null) {
174            riverFlow.abort();
175        }
176        if (riverSource != null) {
177            riverSource.closeReading();
178            riverSource.closeWriting();
179        }
180        if (riverMouth != null) {
181            riverMouth.close();
182        }
183    }
184
185    /**
186     * Induce a river run once, but in a synchronous manner. Mainly used for tests.
187     */
188    public void once() {
189        if (riverFlow != null) {
190            riverFlow.move();
191        }
192    }
193
194    /**
195     * Induce a river run once, but in an asynchronous manner.
196     */
197    public void induce() {
198        RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
199        // prepare task for run
200        riverFlow.riverContext(riverContext);
201        Thread thread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
202                "JDBC river: [" + riverName.name() + '/' + strategy + ')')
203                .newThread(riverFlow);
204        riverFlow.once(thread);
205    }
206
207    public RiverFlow riverFlow() {
208        return riverFlow;
209    }
