018    /**
019     */
020    public class PlainKeyValueStreamListener<K,V> implements KeyValueStreamListener<K,V> {
021    
022        /**
023         * The current structured object
024         */
025        private IndexableObject current;
026        /**
027         * The object before the current object
028         */
029        private IndexableObject prev;
030    
031        /**
032         * The keys of the values. They are examined for the Elasticsearch index
033         * attributes.
034         */
035        private List<K> keys;
036        /**
037         * The delimiter between the key names in the path, used for string split.
038         * Should not be modified.
039         */
040        private char delimiter = '.';
041    
042        /**
043         * Set custom delimiter
044         * @param delimiter the delimiter
045         * @return this listener
046         */
047        public PlainKeyValueStreamListener delimiter(char delimiter) {
048            this.delimiter = delimiter;
049            return this;
050        }
051    
052        /**
053         *
054         * @return this value listener
055         * @throws java.io.IOException
056         */
057        @Override
058        public KeyValueStreamListener<K, V> begin() throws IOException {
059            return this;
060        }
061    
062        @Override
063        public KeyValueStreamListener<K, V> keyValue(K key, V value) throws IOException {
064            return this;
065        }
066    
067        @Override
068        public KeyValueStreamListener<K, V> keyValue(KeyValue<K,V> value) throws IOException {
069            return this;
070        }
071    
072        /**
073         * Set the keys.
074         *
075         * @param keys the keys
076         * @return this value listener
077         */
078        @Override
079        public KeyValueStreamListener<K, V> keys(List<K> keys) throws IOException {
080            this.keys = keys;
081            return this;
082        }
083    
084        /**
085         * Receive values.
086         *
087         * @param values the values
088         * @return this value listener
089         * @throws java.io.IOException
090         */
091        @Override
092        public KeyValueStreamListener<K, V> values(List<V> values) throws IOException {
093            boolean hasSource = false;
094            if (current == null) {
095                current = newObject();
096            }
097            if (prev == null) {
098                prev = newObject();
099            }
100            // execute meta operations
101            for (int i = 0; i < values.size(); i++) {
102                // v may be null, then continue
103                V v = values.get(i);
104                if (v == null) {
105                    continue;
106                }
107                K k = keys.get(i);
108                map(k, v, current);
109                if (ControlKeys._source.name().equals(k)) {
110                    hasSource = true;
111                }
112            }
113            if (hasSource) {
114                end(current);
115                current = newObject();
116                return this;
117            }
118            // switch to next structured object if current is not equal to previous
119            if (!current.equals(prev) || current.isEmpty()) {
120                prev.source(current.source()); // "steal" source
121                end(prev); // here, the element is being prepared for bulk indexing
122                prev = current;
123                current = newObject();
124            }
125            // create current object from values by sequentially merging the values
126            for (int i = 0; i < keys.size(); i++) {
127                Map map = null;
128                try {
129                    // JSON content?
130                    map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
131                } catch (Exception e) {
132                    // ignore
133                }
134                Object v = map != null && map.size() > 0 ? map : values.get(i);
135                Map m = merge(current.source(), keys.get(i), v);
136                current.source(m);
137            }
138            return this;
139        }
140    
141        protected void map(K k, V v, IndexableObject current) throws IOException {
142            if (ControlKeys._optype.name().equals(k)) {
143                current.optype(v.toString());
144            } else if (ControlKeys._index.name().equals(k)) {
145                current.index(v.toString());
146            } else if (ControlKeys._type.name().equals(k)) {
147                current.type(v.toString());
148            } else if (ControlKeys._id.name().equals(k)) {
149                current.id(v.toString());
150            } else if (ControlKeys._version.name().equals(k)) {
151                current.meta(k.toString(), v.toString());
152            } else if (ControlKeys._routing.name().equals(k)) {
153                current.meta(k.toString(), v.toString());
154            } else if (ControlKeys._parent.name().equals(k)) {
155                current.meta(k.toString(), v.toString());
156            } else if (ControlKeys._timestamp.name().equals(k)) {
157                current.meta(k.toString(), v.toString());
158            } else if (ControlKeys._ttl.name().equals(k)) {
159                current.meta(k.toString(), v.toString());
160            } else if (ControlKeys._job.name().equals(k)) {
161                current.meta(k.toString(), v.toString());
162            } else if (ControlKeys._source.name().equals(k)) {
163                current.source(JsonXContent.jsonXContent.createParser(v.toString()).mapAndClose());
164            }
165        }
166    
167        /**
168         * End of values.
169         *
170         * @return this value listener
171         * @throws java.io.IOException
172         */
173        public KeyValueStreamListener<K, V> end() throws IOException {
174            if (prev != null) {
175                prev.source(current.source());
176                end(prev);
177            }
178            prev = newObject();
179            current = newObject();
180            return this;
181        }
182    
183        /**
184         * The object is complete
185         *
186         * @param object the object
187         * @return this value listener
188         * @throws java.io.IOException
189         */
190        public KeyValueStreamListener<K, V> end(IndexableObject object) throws IOException {
191            return this;
192        }
193    
194        /**
195         * Merge key/value pair to a map holding a JSON object. The key consists of
196         * a path pointing to the value position in the JSON object. The key,
197         * representing a path, is divided into head/tail. The recursion terminates
198         * if there is only a head and no tail. In this case, the value is added as
199         * a tuple to the map. If the head key exists, the merge process is
200         * continued by following the path represented by the key. If the path does
201         * not exist, a new map is created. A conflict arises if there is no map at
202         * a head key position. Then, the prefix given in the path is considered
203         * illegal.
204         *
205         */
206        private final static Set<String> controlKeys = ControlKeys.makeSet();
207    
208        protected Map<String, Values<Object>> merge(Map map, Object k, Object value) {
209            String key = k.toString();
210            if (controlKeys.contains(key)) {
211                return map;
212            }
213            int i = key.indexOf(delimiter);
214            String index = null;
215            if (i <= 0) {
216                Matcher matcher = p.matcher(key);
217                boolean isSequence = matcher.matches();
218                String head = key;
219                if (isSequence) {
220                    head = matcher.group(1);
221                    index = matcher.group(2);
222                }
223                if (index == null || index.isEmpty()) {
224                    map.put(head, new Values(map.get(head), value, isSequence));
225                } else {
226                    if (!map.containsKey(head)) {
227                        map.put(head, newLinkedList());
228                    }
229                    Object o = map.get(head);
230                    if (o instanceof List) {
231                        List l = (List) o;
232                        int j = l.isEmpty() ? -1 : l.size() - 1;
233                        if (j >= 0) {
234                            Map<String, Values<Object>> m = (Map<String, Values<Object>>) l.get(j);
235                            if (!m.containsKey(index)) {
236                                l.set(j, merge(m, index, value)); // append
237                            } else {
238                                l.add(merge(new HashMap(), index, value));
239                            }
240                        } else {
241                            l.add(merge(new HashMap(), index, value));
242                        }
243                    }
244                }
245            } else {
246                String head = key.substring(0, i);
247                Matcher matcher = p.matcher(head);
248                boolean isSequence = matcher.matches();
249                if (isSequence) {
250                    head = matcher.group(1);
251                }
252                String tail = key.substring(i + 1);
253                if (map.containsKey(head)) {
254                    Object o = map.get(head);
255                    if (o instanceof Map) {
256                        merge((Map<String, Values<Object>>) o, tail, value);
257                    } else if (o instanceof Values) {
258                        // head of Values a Map?
259                        o = ((Values)o).getValues()[0];
260                        if (o instanceof Map) {
261                            merge((Map<String, Values<Object>>) o, tail, value);
262                        } else {
263                            throw new IllegalArgumentException("illegal head: " + head);
264                        }
265                    } else {
266                        throw new IllegalArgumentException("illegal head: " + head);
267                    }
268                } else {
269                    Map<String, Values<Object>> m = new HashMap<String, Values<Object>>();
270                    map.put(head, m);
271                    merge(m, tail, value);
272                }
273            }
274            return map;
275        }
276    
277        private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
278    
279        /**
280         * Create a new structured object
281         *
282         * @return a new structured object
283         */
284        private IndexableObject newObject() {
285            return new PlainIndexableObject();
286        }
