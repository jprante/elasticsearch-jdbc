001package org.xbib.elasticsearch.rest.action.support;
002
003import org.apache.lucene.util.BytesRef;
004import org.apache.lucene.util.UnicodeUtil;
005import org.elasticsearch.ElasticsearchIllegalArgumentException;
006import org.elasticsearch.rest.AbstractRestResponse;
007import org.elasticsearch.rest.RestRequest;
008import org.elasticsearch.rest.RestStatus;
009import org.xbib.common.xcontent.XContentBuilder;
010
011import java.io.IOException;
012
013public class XContentRestResponse extends AbstractRestResponse {
014
015    private static final byte[] END_JSONP;
016
017    static {
018        BytesRef U_END_JSONP = new BytesRef();
019        UnicodeUtil.UTF16toUTF8(");", 0, ");".length(), U_END_JSONP);
020        END_JSONP = new byte[U_END_JSONP.length];
021        System.arraycopy(U_END_JSONP.bytes, U_END_JSONP.offset, END_JSONP, 0, U_END_JSONP.length);
022    }
023
024    private final BytesRef prefixUtf8Result;
025
026    private final RestStatus status;
027
028    private final XContentBuilder builder;
029
030    public XContentRestResponse(RestRequest request, RestStatus status, XContentBuilder builder) throws IOException {
031        if (request == null) {
032            throw new ElasticsearchIllegalArgumentException("request must be set");
033        }
034        this.builder = builder;
035        this.status = status;
036        this.prefixUtf8Result = startJsonp(request);
037    }
038
039    public XContentBuilder builder() {
040        return this.builder;
041    }
042
043    @Override
044    public String contentType() {
045        return builder.contentType().restContentType();
046    }
047
048    @Override
049    public boolean contentThreadSafe() {
050        return true;
051    }
052
053    @Override
054    public byte[] content() throws IOException {
055        return builder.bytes().array();
056    }
057
058    @Override
059    public int contentLength() throws IOException {
060        return builder.bytes().length();
061    }
062
063    @Override
064    public int contentOffset() throws IOException {
065        return 0;
066    }
067
068    @Override
069    public RestStatus status() {
070        return this.status;
071    }
072
073    @Override
074    public byte[] prefixContent() {
075        if (prefixUtf8Result != null) {
076            return prefixUtf8Result.bytes;
077        }
078        return null;
079    }
080
081    @Override
082    public int prefixContentLength() {
083        if (prefixUtf8Result != null) {
084            return prefixUtf8Result.length;
085        }
086        return 0;
087    }
088
089    @Override
090    public int prefixContentOffset() {
091        if (prefixUtf8Result != null) {
092            return prefixUtf8Result.offset;
093        }
094        return 0;
095    }
096
097    @Override
098    public byte[] suffixContent() {
099        if (prefixUtf8Result != null) {
100            return END_JSONP;
101        }
102        return null;
103    }
104
105    @Override
106    public int suffixContentLength() {
107        if (prefixUtf8Result != null) {
108            return END_JSONP.length;
109        }
110        return 0;
111    }
112
113    @Override
114    public int suffixContentOffset() {
115        return 0;
116    }
117
118    private static BytesRef startJsonp(RestRequest request) {
119        String callback = request.param("callback");
120        if (callback == null) {
121            return null;
122        }
123        final BytesRef result = new BytesRef();
124        UnicodeUtil.UTF16toUTF8(callback, 0, callback.length(), result);
125        result.bytes[result.length] = '(';
126        result.length++;
127        return result;
128    }
129}
