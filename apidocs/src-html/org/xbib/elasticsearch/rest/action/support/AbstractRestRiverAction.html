001package org.xbib.elasticsearch.rest.action.support;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Injector;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.rest.BaseRestHandler;
007import org.elasticsearch.rest.RestChannel;
008import org.elasticsearch.rest.RestRequest;
009import org.elasticsearch.rest.RestStatus;
010import org.xbib.common.xcontent.XContentBuilder;
011
012import java.io.IOException;
013
014public abstract class AbstractRestRiverAction extends BaseRestHandler {
015
016    protected Injector injector;
017
018    public AbstractRestRiverAction(Settings settings, Client client, Injector injector) {
019        super(settings, client);
020        this.injector = injector;
021    }
022
023    protected void respond(boolean success, RestRequest request, RestChannel channel, String error, RestStatus status) {
024        try {
025            XContentBuilder builder = RestXContentBuilder.restContentBuilder(request);
026            if (request.paramAsBoolean("pretty", false)) {
027                builder.prettyPrint();
028            }
029            builder.startObject()
030                    .field("success", success);
031            if (error != null) {
032                builder.field("error", error);
033            }
034            builder.endObject();
035            channel.sendResponse(new XContentRestResponse(request, status, builder));
036        } catch (IOException e) {
037            errorResponse(request, channel, e);
038        }
039    }
040
041    protected void errorResponse(RestRequest request, RestChannel channel, Throwable e) {
042        try {
043            channel.sendResponse(new XContentThrowableRestResponse(request, e));
044        } catch (IOException e1) {
045            logger.error("Failed to send failure response", e1);
046        }
047    }
048
049}
