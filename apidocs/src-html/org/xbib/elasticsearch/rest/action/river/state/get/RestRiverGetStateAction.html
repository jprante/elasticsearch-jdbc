001package org.xbib.elasticsearch.rest.action.river.state.get;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.common.xcontent.XContentBuilder;
007import org.elasticsearch.rest.BaseRestHandler;
008import org.elasticsearch.rest.BytesRestResponse;
009import org.elasticsearch.rest.RestChannel;
010import org.elasticsearch.rest.RestController;
011import org.elasticsearch.rest.RestRequest;
012import org.elasticsearch.rest.RestResponse;
013import org.elasticsearch.rest.action.support.RestBuilderListener;
014import org.xbib.elasticsearch.action.river.jdbc.state.RiverState;
015import org.xbib.elasticsearch.action.river.jdbc.state.get.GetRiverStateAction;
016import org.xbib.elasticsearch.action.river.jdbc.state.get.GetRiverStateRequest;
017import org.xbib.elasticsearch.action.river.jdbc.state.get.GetRiverStateResponse;
018
019import static org.elasticsearch.rest.RestStatus.OK;
020
021public class RestRiverGetStateAction extends BaseRestHandler {
022
023    @Inject
024    public RestRiverGetStateAction(Settings settings, Client client, RestController controller) {
025        super(settings, client);
026        controller.registerHandler(RestRequest.Method.GET, "/_river/jdbc/{riverName}/_state", this);
027    }
028
029    @Override
030    public void handleRequest(RestRequest request, RestChannel channel, Client client) {
031        String riverName = request.param("riverName");
032        String riverType = "jdbc";
033        GetRiverStateRequest riverStateRequest = new GetRiverStateRequest();
034        riverStateRequest.setRiverName(riverName)
035                .setRiverType(riverType)
036                .masterNodeTimeout(request.paramAsTime("master_timeout", riverStateRequest.masterNodeTimeout()))
037                .local(request.paramAsBoolean("local", riverStateRequest.local()));
038        client.admin().cluster().execute(GetRiverStateAction.INSTANCE, riverStateRequest, new RestBuilderListener<GetRiverStateResponse>(channel) {
039
040            @Override
041            public RestResponse buildResponse(GetRiverStateResponse getRiverStateResponse, XContentBuilder builder) throws Exception {
042                builder.startObject();
043                builder.startArray("state");
044                for (RiverState state : getRiverStateResponse.getStates()) {
045                    builder.startObject();
046                    builder.field("name", state.getName());
047                    builder.field("type", state.getType());
048                    builder.field("settings", state.getSettings().getAsMap());
049                    builder.field("custom", state.getCustom());
050                    builder.endObject();
051                }
052                builder.endArray().endObject();
053                return new BytesRestResponse(OK, builder);
054            }
055        });
056    }
057
058}
