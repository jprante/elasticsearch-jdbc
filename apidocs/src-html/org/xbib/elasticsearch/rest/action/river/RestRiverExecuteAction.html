001package org.xbib.elasticsearch.rest.action.river;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.common.settings.Settings;
007import org.elasticsearch.rest.RestChannel;
008import org.elasticsearch.rest.RestController;
009import org.elasticsearch.rest.RestRequest;
010import org.elasticsearch.rest.RestStatus;
011import org.xbib.elasticsearch.action.river.execute.RiverExecuteAction;
012import org.xbib.elasticsearch.action.river.execute.RiverExecuteRequest;
013import org.xbib.elasticsearch.action.river.execute.RiverExecuteResponse;
014import org.xbib.elasticsearch.rest.action.support.AbstractRestRiverAction;
015
016/**
017 * Run a river. The river can be executed once with such a call.
018 * <p>
019 * Example:
020 * <p>
021 * curl -XPOST 'localhost:9200/_river/my_jdbc_river/_execute'
022 */
023public class RestRiverExecuteAction extends AbstractRestRiverAction {
024
025    @Inject
026    public RestRiverExecuteAction(Settings settings, Client client, RestController controller, Injector injector) {
027        super(settings, client, injector);
028        controller.registerHandler(RestRequest.Method.POST, "/_river/{riverName}/_execute", this);
029        controller.registerHandler(RestRequest.Method.POST, "/_river/{riverType}/{riverName}/_execute", this);
030    }
031
032    @Override
033    public void handleRequest(final RestRequest request, RestChannel channel) {
034        String riverName = request.param("riverName");
035        if (riverName == null || riverName.isEmpty()) {
036            respond(false, request, channel, "parameter 'riverName' is required", RestStatus.BAD_REQUEST);
037            return;
038        }
039        String riverType = request.param("riverType");
040        RiverExecuteRequest riverExecuteRequest = new RiverExecuteRequest()
041                .setRiverName(riverName)
042                .setRiverType(riverType);
043        RiverExecuteResponse riverExecuteResponse = client
044                .execute(RiverExecuteAction.INSTANCE, riverExecuteRequest).actionGet();
045        boolean isExecuted = false;
046        for (int i = 0; i < riverExecuteResponse.isExecuted().length; i++) {
047            isExecuted = isExecuted || riverExecuteResponse.isExecuted()[i];
048        }
049        respond(isExecuted, request, channel, "ok", RestStatus.OK);
050    }
051
052}
