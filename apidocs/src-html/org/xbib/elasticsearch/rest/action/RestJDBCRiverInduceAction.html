002package org.xbib.elasticsearch.rest.action;
003
004import org.elasticsearch.client.Client;
005import org.elasticsearch.common.inject.Inject;
006import org.elasticsearch.common.inject.Injector;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.rest.RestChannel;
009import org.elasticsearch.rest.RestController;
010import org.elasticsearch.rest.RestRequest;
011import org.elasticsearch.rest.RestStatus;
012import org.elasticsearch.river.River;
013import org.elasticsearch.river.RiverName;
014
015import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
016
017import java.util.Map;
018
019/**
020 * The JDBC River REST inducer. The river can be fired once to run
021 * when this move is called from REST.
022 * <p/>
023 * Example:<br/>
024 * <code>
025 * curl -XPOST 'localhost:9200/_river/my_jdbc_river/_induce'
026 * </code>
027 */
028public class RestJDBCRiverInduceAction extends AbstractRestRiverAction {
029
030    @Inject
031    public RestJDBCRiverInduceAction(Settings settings, Client client,
032                                     RestController controller, Injector injector) {
033        super(settings, client, injector);
034        controller.registerHandler(RestRequest.Method.POST, "/_river/jdbc/{river}/induce", this);
035    }
036
037    @Override
038    public void handleRequest(final RestRequest request, RestChannel channel) {
039        //Get and check river name parameter
040        String riverName = request.param("river");
041        if (riverName == null || riverName.isEmpty()) {
042            respond(false, request, channel, "Parameter 'river' is required", RestStatus.BAD_REQUEST);
043            return;
044        }
045        // find river class
046        //GetResponse get = client.get(Requests.getRequest("_river").type(riverName).id("_meta")).actionGet();
047        boolean found = false;
048        for (Map.Entry<RiverName, River> entry : rivers(injector).entrySet()) {
049            RiverName name = entry.getKey();
050            if (name.getName().equals(riverName)) {
051                if (!name.getType().equals(JDBCRiver.TYPE)) {
052                    respond(false, request, channel,
053                            "River '" + riverName + "' is not a jdbc-river, but has type " + name.getType(),
054                            RestStatus.UNPROCESSABLE_ENTITY
055                    );
056                    return;
057                }
058                JDBCRiver jdbcRiver = (JDBCRiver) entry.getValue();
059                jdbcRiver.induce();
060                found = true;
061                break;
062            }
063        }
064        String error = found ? null : "River not found: " + riverName;
065        respond(found, request, channel, error, RestStatus.OK);
066    }
067
068}
