001package org.xbib.elasticsearch.plugin.support;
002
003import java.util.EnumSet;
004import java.util.Set;
005
006import static org.elasticsearch.common.collect.Sets.newHashSet;
007
008/**
009 * The names of keys with a special meaning for controlling Elasticsearch indexing.
010 * Mostly, they map to the Elasticsearch bulk item control keys.
011 * The _job column denotes an ID for the event of a fetch execution.
012 */
013public enum ControlKeys {
014
015    _optype, _index, _type, _id, _version, _timestamp, _ttl, _routing, _parent, _source, _job;
016
017    public static Set<String> makeSet() {
018        Set<String> set = newHashSet();
019        for (ControlKeys k : EnumSet.allOf(ControlKeys.class)) {
020            set.add(k.name());
021        }
022        return set;
023    }
024
025}
