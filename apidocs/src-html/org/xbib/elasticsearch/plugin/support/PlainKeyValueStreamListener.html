001package org.xbib.elasticsearch.plugin.support;

003import org.xbib.common.xcontent.json.JsonXContent;
004import org.xbib.keyvalue.KeyValueStreamListener;
005
006import java.io.IOException;
007import java.util.HashMap;
008import java.util.LinkedList;
009import java.util.List;
010import java.util.Map;
011import java.util.Set;
012import java.util.regex.Matcher;

015/**
016 */
017public class PlainKeyValueStreamListener<K, V> implements KeyValueStreamListener<K, V> {
018
019    /**
020     * The current structured object
021     */
022    private IndexableObject current;
023    /**
024     * The object before the current object
025     */
026    private IndexableObject prev;
027
028    /**
029     * The keys of the values. They are examined for the Elasticsearch index
030     * attributes.
031     */
032    private List<K> keys;
033    /**
034     * The delimiter between the key names in the path, used for string split.
035     * Should not be modified.
036     */
037    private char delimiter = '.';
038
039    private boolean shouldAutoGenID;
040
041    /**
042     * Set custom delimiter
043     *
044     * @param delimiter the delimiter
045     * @return this listener
046     */
047    public PlainKeyValueStreamListener delimiter(char delimiter) {
048        this.delimiter = delimiter;
049        return this;
050    }
051
052    /**
053     * @return this value listener
054     * @throws java.io.IOException when beginning the object gives an error
055     */
056    @Override
057    public KeyValueStreamListener<K, V> begin() throws IOException {
058        return this;
059    }
060
061    @Override
062    public KeyValueStreamListener<K, V> keyValue(K key, V value) throws IOException {
063        return this;
064    }
065
066    /**
067     * Set the keys.
068     *
069     * @param keys the keys
070     * @return this value listener
071     */
072    @Override
073    public KeyValueStreamListener<K, V> keys(List<K> keys) throws IOException {
074        this.keys = keys;
075        this.shouldAutoGenID = shouldAutoGenID();
076        return this;
077    }
078
079    /**
080     * Receive values.
081     *
082     * @param values the values
083     * @return this value listener
084     * @throws java.io.IOException when value processing giv an error
085     */
086    @Override
087    public KeyValueStreamListener<K, V> values(List<V> values) throws IOException {
088        boolean hasSource = false;
089        if (current == null) {
090            current = newObject();
091        }
092        if (prev == null) {
093            prev = newObject();
094        }
095        // execute meta operations
096        for (int i = 0; i < values.size(); i++) {
097            // v may be null, then continue
098            V v = values.get(i);
099            if (v == null) {
100                continue;
101            }
102            K k = keys.get(i);
103            map(k, v, current);
104            if (ControlKeys._source.name().equals(k)) {
105                hasSource = true;
106            }
107        }
108        if (hasSource) {
109            end(current);
110            current = newObject();
111            return this;
112        }
113        // switch to next structured object if current is not equal to previous
114        if (!current.equals(prev) || current.isEmpty() || shouldAutoGenID) {
115            prev.source(current.source()); // "steal" source
116            end(prev); // here, the element is being prepared for bulk indexing
117            prev = current;
118            current = newObject();
119        }
120        // create current object from values by sequentially merging the values
121        for (int i = 0; i < keys.size(); i++) {
122            Map map = null;
123            try {
124                // JSON content?
125                map = JsonXContent.jsonXContent.createParser(values.get(i).toString()).mapAndClose();
126            } catch (Exception e) {
127                // ignore
128            }
129            Object v = map != null && map.size() > 0 ? map : values.get(i);
130            Map<String, Object> m = merge(current.source(), keys.get(i), v);
131            current.source(m);
132        }
133        return this;
134    }
135
136    protected void map(K k, V v, IndexableObject current) throws IOException {
137        if (ControlKeys._optype.name().equals(k)) {
138            current.optype(v.toString());
139        } else if (ControlKeys._index.name().equals(k)) {
140            current.index(v.toString());
141        } else if (ControlKeys._type.name().equals(k)) {
142            current.type(v.toString());
143        } else if (ControlKeys._id.name().equals(k)) {
144            current.id(v.toString());
145        } else if (ControlKeys._version.name().equals(k)) {
146            current.meta(k.toString(), v.toString());
147        } else if (ControlKeys._routing.name().equals(k)) {
148            current.meta(k.toString(), v.toString());
149        } else if (ControlKeys._parent.name().equals(k)) {
150            current.meta(k.toString(), v.toString());
151        } else if (ControlKeys._timestamp.name().equals(k)) {
152            current.meta(k.toString(), v.toString());
153        } else if (ControlKeys._ttl.name().equals(k)) {
154            current.meta(k.toString(), v.toString());
155        } else if (ControlKeys._job.name().equals(k)) {
156            current.meta(k.toString(), v.toString());
157        } else if (ControlKeys._source.name().equals(k)) {
158            current.source(JsonXContent.jsonXContent.createParser(v.toString()).mapAndClose());
159        }
160    }
161
162    /**
163     * End of values.
164     *
165     * @return this value listener
166     * @throws java.io.IOException
167     */
168    public KeyValueStreamListener<K, V> end() throws IOException {
169        if (prev != null) {
170            prev.source(current.source());
171            end(prev);
172        }
173        prev = newObject();
174        current = newObject();
175        return this;
176    }
177
178    /**
179     * The object is complete
180     *
181     * @param object the object
182     * @return this value listener
183     * @throws java.io.IOException when ending the object gives an error
184     */
185    public KeyValueStreamListener<K, V> end(IndexableObject object) throws IOException {
186        return this;
187    }
188
189    /**
190     * Merge key/value pair to a map holding a JSON object. The key consists of
191     * a path pointing to the value position in the JSON object. The key,
192     * representing a path, is divided into head/tail. The recursion terminates
193     * if there is only a head and no tail. In this case, the value is added as
194     * a tuple to the map. If the head key exists, the merge process is
195     * continued by following the path represented by the key. If the path does
196     * not exist, a new map is created. A conflict arises if there is no map at
197     * a head key position. Then, the prefix given in the path is considered
198     * illegal.
199     */
200    private final static Set<String> controlKeys = ControlKeys.makeSet();
201
202    @SuppressWarnings({"unchecked"})
203    protected Map<String, Object> merge(Map<String, Object> map, Object k, Object value) {
204        String key = k.toString();
205        if (controlKeys.contains(key)) {
206            return map;
207        }
208        int i = key.indexOf(delimiter);
209        String index = null;
210        if (i <= 0) {
211            Matcher matcher = p.matcher(key);
212            boolean isSequence = matcher.matches();
213            String head = key;
214            if (isSequence) {
215                head = matcher.group(1);
216                index = matcher.group(2);
217            }
218            if (index == null || index.isEmpty()) {
219                map.put(head, new Values(map.get(head), value, isSequence));
220            } else {
221                if (!map.containsKey(head)) {
222                    map.put(head, new LinkedList());
223                }
224                Object o = map.get(head);
225                if (o instanceof List) {
226                    List l = (List) o;
227                    int j = l.isEmpty() ? -1 : l.size() - 1;
228                    if (j >= 0) {
229                        Map<String, Object> m = (Map<String, Object>) l.get(j);
230                        if (!m.containsKey(index)) {
231                            l.set(j, merge(m, index, value)); // append
232                        } else {
233                            l.add(merge(new HashMap(), index, value));
234                        }
235                    } else {
236                        l.add(merge(new HashMap(), index, value));
237                    }
238                }
239            }
240        } else {
241            String head = key.substring(0, i);
242            Matcher matcher = p.matcher(head);
243            boolean isSequence = matcher.matches();
244            if (isSequence) {
245                head = matcher.group(1);
246            }
247            String tail = key.substring(i + 1);
248            if (map.containsKey(head)) {
249                Object o = map.get(head);
250                if (o instanceof Map) {
251                    merge((Map<String, Object>) o, tail, value);
252                } else if (o instanceof Values) {
253                    // head of Values a Map?
254                    o = ((Values) o).getValues()[0];
255                    if (o instanceof Map) {
256                        merge((Map<String, Object>) o, tail, value);
257                    } else {
258                        throw new IllegalArgumentException("illegal head: " + head);
259                    }
260                } else {
261                    throw new IllegalArgumentException("illegal head: " + head);
262                }
263            } else {
264                Map<String, Object> m = new HashMap<String, Object>();
265                map.put(head, m);
266                merge(m, tail, value);
267            }
268        }
269        return map;
270    }
271
272    private final static Pattern p = Pattern.compile("^(.*)\\[(.*?)\\]$");
273
274    /**
275     * Create a new structured object
276     *
277     * @return a new structured object
278     */
279    private IndexableObject newObject() {
280        return new PlainIndexableObject();
281    }
282
283    private boolean shouldAutoGenID() {
284        boolean b = true;
285        for (K key : keys) {
286            if (ControlKeys._id.name().equals(key)) {
287                b = false;
288                break;
289            }
290        }
291        return b;
292    }
