001package org.xbib.elasticsearch.plugin.support;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.collect.ImmutableMap;
005import org.elasticsearch.common.inject.Injector;
006import org.elasticsearch.indices.IndexMissingException;
007import org.elasticsearch.river.River;
008import org.elasticsearch.river.RiverName;
009import org.elasticsearch.river.RiversService;
010import org.xbib.elasticsearch.action.river.state.RiverStateAction;
011import org.xbib.elasticsearch.action.river.state.RiverStateRequest;
012import org.xbib.elasticsearch.action.river.state.RiverStateResponse;
013
014import java.io.IOException;
015import java.lang.reflect.Field;
016import java.util.Collections;
017
018public class RiverHelper {
019
020    private RiverHelper() {
021    }
022
023    public void waitForRiverEnabled(Client client, String riverName) throws InterruptedException, IOException {
024        waitForRiverEnabled(client, riverName, 15);
025    }
026
027    public static void waitForRiverEnabled(Client client, String riverName, int seconds) throws InterruptedException, IOException {
028        RiverStateRequest riverStateRequest = new RiverStateRequest()
029                .setRiverName(riverName);
030        RiverStateResponse riverStateResponse = client
031                .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
032        while (seconds-- > 0 && !isEnabled(riverName, riverStateResponse)) {
033            Thread.sleep(1000L);
034            try {
035                riverStateResponse = client.execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
036            } catch (IndexMissingException e) {
037                return;
038            }
039        }
040    }
041
042    private static boolean isEnabled(String riverName, RiverStateResponse riverStateResponse) {
043        if (riverStateResponse == null) {
044            return false;
045        }
046        if (riverStateResponse.getStates() == null) {
047            return false;
048        }
049        if (riverStateResponse.getStates().isEmpty()) {
050            return false;
051        }
052        for (RiverState state : riverStateResponse.getStates()) {
053            if (state.getName().equals(riverName)) {
054                return state.isEnabled();
055            }
056        }
057        return false;
058    }
059
060    public static void waitForInactiveRiver(Client client, String riverName) throws InterruptedException, IOException {
061        waitForInactiveRiver(client, riverName, 30);
062    }
063
064    public static void waitForInactiveRiver(Client client, String riverName, int seconds) throws InterruptedException, IOException {
065        RiverStateRequest riverStateRequest = new RiverStateRequest()
066                .setRiverName(riverName);
067        RiverStateResponse riverStateResponse = client
068                .execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
069        while (seconds-- > 0 && isActive(riverName, riverStateResponse)) {
070            Thread.sleep(1000L);
071            try {
072                riverStateResponse = client.execute(RiverStateAction.INSTANCE, riverStateRequest).actionGet();
073            } catch (IndexMissingException e) {
074                return;
075            }
076        }
077        if (seconds < 0) {
078            throw new IOException("timeout waiting for inactive river");
079        }
080    }
081
082    private static boolean isActive(String riverName, RiverStateResponse riverStateResponse) {
083        if (riverStateResponse == null) {
084            return false;
085        }
086        if (riverStateResponse.getStates() == null) {
087            return false;
088        }
089        if (riverStateResponse.getStates().isEmpty()) {
090            return false;
091        }
092        for (RiverState state : riverStateResponse.getStates()) {
093            if (state.getName().equals(riverName)) {
094                return state.isActive();
095            }
096        }
097        return false;
098    }
099
100    /**
101     * Retrieve the registered rivers using reflection. This hack can be removed when RiversService gets a public API.
102     *
103     * @param injector injector
104     * @return map of rivers or null if not possible
105     */
106    @SuppressWarnings({"unchecked"})
107    public static ImmutableMap<RiverName, River> rivers(Injector injector) {
108        RiversService riversService = injector.getInstance(RiversService.class);
109        try {
110            Field field = RiversService.class.getDeclaredField("rivers");
111            if (field != null) {
112                field.setAccessible(true);
113                return (ImmutableMap<RiverName, River>) field.get(riversService);
114            }
115        } catch (Throwable e) {
116            // ignore
117        }
118        return ImmutableMap.copyOf(Collections.emptyMap());
119    }
