001package org.xbib.elasticsearch.plugin.feeder;
002
003import java.io.InputStreamReader;
004
005/**
006 * Stub for loading a class (tool) and execute it. Isolating the main method from other methods is of advantage
007 * for JVMs that thoroughly check if this class ca be executed.
008 */
009public class Runner {
010
011    public static void main(String[] args) {
012        try {
013            Class clazz = Class.forName(args[0]);
014            CommandLineInterpreter commandLineInterpreter = (CommandLineInterpreter) clazz.newInstance();
015            Runtime.getRuntime().addShutdownHook(commandLineInterpreter.shutdownHook());
016            commandLineInterpreter.readFrom(new InputStreamReader(System.in, "UTF-8")).run();
017        } catch (Throwable e) {
018            e.printStackTrace();
019            System.exit(1);
020        }
021        System.exit(0);
022    }
023}
