002package org.xbib.elasticsearch.plugin.feeder;
003
004import java.io.InputStreamReader;
005
006public class Runner {
007
008    public static void main(String[] args) {
009        try {
010            Class clazz = Class.forName(args[0]);
011            Tool tool = (Tool) clazz.newInstance();
012            Runtime.getRuntime().addShutdownHook(tool.shutdownHook());
013            tool.readFrom(new InputStreamReader(System.in, "UTF-8")).run();
014        } catch (Throwable e) {
015            e.printStackTrace();
016            System.exit(1);
017        }
018        System.exit(0);
019    }
020}
