042/**
043 * Base class for all feeders and feed pipelines.
044 * <p>
045 * A feed pipeline is a sequence of URI in form of pipeline elements that can be consumed
046 * in serial or parallel manner.
047 * <p>
048 * Variables - mostly read-only - that are common to all pipeline executions are declared
049 * as static variables.
050 *
051 * @param <T> the pipeline element
052 * @param <R> the pipeline request
053 * @param <P> the pipeline exception
054 */
055public abstract class AbstractFeeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
056        extends AbstractPipeline<MapPipelineElement, PipelineException>
057        implements Feeder<T,R,P> {
058

170    @Override
171    public void schedule(Thread thread) {
172        if (settings == null) {
173            throw new IllegalArgumentException("no settings?");
174        }
175        String[] schedule = settings.getAsArray("schedule");
176        Long seconds = settings.getAsTime("interval", TimeValue.timeValueSeconds(0)).seconds();
177        if (schedule != null && schedule.length > 0) {
178            CronThreadPoolExecutor e = new CronThreadPoolExecutor(settings.getAsInt("cronpoolsize", 4));
179            for (String cron : schedule) {
180                e.schedule(thread, new CronExpression(cron));
181            }
182            this.scheduledThreadPoolExecutor = e;
183            logger.info("scheduled with cron expressions {}", Arrays.asList(schedule));
184        } else if (seconds > 0L) {
185            this.scheduledThreadPoolExecutor = new ScheduledThreadPoolExecutor(settings.getAsInt("cronpoolsize", 4));
186            scheduledThreadPoolExecutor.scheduleAtFixedRate(thread, 0L, seconds, TimeUnit.SECONDS);
187            logger.info("scheduled at fixed rate of {} seconds", seconds);
188        } else {
189            thread.start();
190            logger.info("started");
191        }
192    }

194    @Override
195    public Feeder<T, R, P> setClient(Client client) {
196        if (settings == null) {
197            throw new IllegalArgumentException("no settings?");
198        }
199        Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
200        Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
201                Runtime.getRuntime().availableProcessors());
202        ByteSizeValue maxvolume = settings.getAsBytesSize("maxbulkvolume", ByteSizeValue.parseBytesSizeValue("10m"));
203        this.ingest = new NodeClient()
204                .maxActionsPerBulkRequest(maxbulkactions)
205                .maxConcurrentBulkRequests(maxconcurrentbulkrequests)
206                .maxVolumePerBulkRequest(org.elasticsearch.common.unit.ByteSizeValue.parseBytesSizeValue(maxvolume.toString()))
207                .newClient(client);
208        return this;
209    }

227    @Override
228    public void run() {
229        if (settings == null) {
230            throw new IllegalArgumentException("no settings?");
231        }
232        logger.info("starting run with settings {}", settings.getAsMap());
233        try {
234            prepare();
235            this.executor = new MetricSimplePipelineExecutor<T, R, P>()
236                    .setConcurrency(settings.getAsInt("concurrency", 1))
237                    .setPipelineProvider(pipelineProvider())
238                    .prepare()
239                    .execute()
240                    .waitFor();
241        } catch (InterruptedException e) {
242            Thread.currentThread().interrupt();
243            logger.warn("interrupted");
244        } catch (Throwable t) {
245            logger.error(t.getMessage(), t);
246        } finally {
247            try {
248                if (executor != null) {
249                    executor.shutdown();
250                    executor = null;
251                }
252                cleanup();
253            } catch (InterruptedException e) {
254                Thread.currentThread().interrupt();
255                logger.warn("interrupted");
256            } catch (IOException e) {
257                logger.error(e.getMessage(), e);
258            }
259            logger.info("run completed with settings {}", settings.getAsMap());
260        }
261    }

263    @Override
264    @SuppressWarnings({"unchecked"})
265    public Feeder<T, R, P> prepare() throws IOException {
266        if (spec == null) {
267            throw new IllegalArgumentException("no spec?");
268        }
269        if (settings == null) {
270            throw new IllegalArgumentException("no settings?");
271        }
272        Integer maxbulkactions = settings.getAsInt("maxbulkactions", 1000);
273        Integer maxconcurrentbulkrequests = settings.getAsInt("maxconcurrentbulkrequests",
274                Runtime.getRuntime().availableProcessors());
275        if (ingest == null) {
276            ingest = "ingest".equals(settings.get("client")) ?
277                    new IngestTransportClient()
278                    : new BulkTransportClient();
279            ingest.maxActionsPerBulkRequest(maxbulkactions)
280                    .maxConcurrentBulkRequests(maxconcurrentbulkrequests);
281            ingest.newClient(URI.create(settings.get("elasticsearch")));
282        }
283        String index = settings.get("index");
284        String type = settings.get("type");
285        try {
286            ingest.setting(AbstractFeeder.class.getResourceAsStream("/" + index + "/settings"));
287        } catch (Exception e) {
288            // ignore
289        }
290        try {
291            ingest.mapping(type, AbstractFeeder.class.getResourceAsStream("/" + index + "/" + type + ".mapping"));
292        } catch (Exception e) {
293            // ignore
294        }
295        ingest.setIndex(index)
296                .setType(type)
297                .newIndex()
298                .startBulk();
299
300        // build input queue
301
302        this.queue = new ConcurrentLinkedQueue<Map<String, Object>>();
303        Object input = getType() != null && spec.containsKey(getType()) ? spec.get(getType()) : spec;
304        if (input instanceof Object[]) {
305            for (Object map : (Object[])input) {
306                if (map instanceof Map) {
307                    queue.offer((Map<String, Object>) map);
308                } else {
309                    logger.warn("'input' field does not contain a map, skipping: {}", map);
310                }
311            }
312        } else if (input instanceof List) {
313            for (Object map : (List)input) {
314                if (map instanceof Map) {
315                    queue.offer((Map<String, Object>) map);
316                } else {
317                    logger.warn("'input' field does not contain a map, skipping: {}", map);
318                }
319            }
320        } else if (input instanceof Map) {
321            queue.offer((Map<String,Object>)input);
322        }
323        logger.info("queue = {}", queue);
324        return this;
325    }
326
327    @Override
328    public Feeder<T, R, P> cleanup() throws IOException {
329        if (ingest != null) {
330            ingest.flush();
331            ingest.stopBulk();
332            ingest.updateReplicaLevel(ingest.settings().getAsInt("replica", 0));
333        }
334        if (riverState != null) {
335            riverState.setActive(false).save(ingest.client());
336        }
337        return this;
338    }

340    /**
341     * Interrupt this feeder. The feeder pipelines will terminate as soon as the next cycle is
342     * executed.
343     *
344     * @param state true if the feeder should be interrupted
345     */
346    public void setInterrupted(boolean state) {
347        interrupted = state;
348        registry.stream().filter(feeder -> this != feeder).forEach(feeder -> feeder.setInterrupted(state));
349    }
350
351    public boolean isInterrupted() {
352        return interrupted;
353    }

355    @Override
356    public void close() throws IOException {
357        setInterrupted(true);
358        logger.info("closing ...");
359    }

385    @Override
386    public Thread shutdownHook() {
387        return new Thread() {
388            public void run() {
389                for (AbstractFeeder feeder : registry) {
390                    try {
391                        if (feeder.scheduledThreadPoolExecutor != null) {
392                            logger.info("shutting down scheduler");
393                            feeder.scheduledThreadPoolExecutor.shutdownNow();
394                            feeder.scheduledThreadPoolExecutor = null;
395                        }
396                        if (feeder.executor != null) {
397                            logger.info("shutting down executor");
398                            feeder.executor.shutdown();
399                        }
400                    } catch (InterruptedException e) {
401                        logger.warn("interrupted", e);
402                    } catch (IOException e) {
403                        logger.error(e.getMessage(), e);
404                    }
405                }
406                if (ingest != null) {
407                    logger.info("shutting down ingester");
408                    try {
409                        ingest.stopBulk();
410                        ingest.updateReplicaLevel(settings.getAsInt("replica", 0));
411                        ingest.shutdown();
412                    } catch (IOException e) {
413                        logger.error(e.getMessage(), e);
414                    }
415                }
416                logger.info("shutdown completed");
417            }
418        };
419    }
