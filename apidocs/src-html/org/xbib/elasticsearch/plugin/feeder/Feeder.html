001package org.xbib.elasticsearch.plugin.feeder;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.settings.Settings;
005import org.xbib.elasticsearch.action.river.jdbc.state.RiverState;
006import org.xbib.pipeline.Pipeline;
007import org.xbib.pipeline.PipelineProvider;
008import org.xbib.pipeline.PipelineRequest;
009
010import java.io.Closeable;
011import java.io.IOException;
012import java.util.Map;
013
014/**
015 * A feeder is a helper for executing feeds of documents to Elastisearch, either in push or pull mode.
016 * Rivers use the pull mode, standalone programs operates usually in the the push mode.
017 * The feed is executing by help of multiple threads with configurable concurrency.
018 * Each thread starts a pipeline which iterates over the feed sources. Each pipeline
019 * processes specifications that describe how the feed source is used.
020 *
021 * @param <T> the pipeline element
022 * @param <R> the pipeline request
023 * @param <P> the pipeline exception
024 */
025public interface Feeder<T, R extends PipelineRequest, P extends Pipeline<T, R>>
026        extends CommandLineInterpreter, Runnable, Closeable {
027
028    /**
029     * Set the type of the feeder
030     *
031     * @param type the type
032     * @return the feeder
033     */
034    Feeder<T, R, P> setType(String type);
035
036    /**
037     * Set the specification for the feeder. The specification contains information
038     * about a feeding step.
039     *
040     * @param spec the specification
041     * @return the specification
042     */
043    Feeder<T, R, P> setSpec(Map<String, Object> spec);
044
045    /**
046     * Get the specification of this feeder
047     *
048     * @return the specification
049     */
050    Map<String, Object> getSpec();
051
052    /**
053     * Set settings for the feeder. The settings control the behaviour of the feeder.
054     *
055     * @param settings the settings
056     * @return the feeder
057     */
058    Feeder<T, R, P> setSettings(Settings settings);
059
060    /**
061     * Get the feeder settings
062     *
063     * @return the settings
064     */
065    Settings getSettings();
066
067    /**
068     * Set the client for the feeder
069     *
070     * @param client the client
071     * @return the feeder
072     */
073    Feeder<T, R, P> setClient(Client client);
074
075    /**
076     * Get the client of the feeder.
077     *
078     * @return the client
079     */
080    Client getClient();
081
082    /**
083     * Get the pipeline provider. The pipeline provider is called once per thread.
084     *
085     * @return the pipeline provider
086     */
087    PipelineProvider<P> pipelineProvider();
088
089    /**
090     * Prepare the feeder. This is the first phase of the feeder run. It is called before feeder threads
091     * are started.
092     *
093     * @return the feeder
094     * @throws java.io.IOException if prepare fails
095     */
096    Feeder<T, R, P> beforeRun() throws IOException;
097
098    /**
099     * Executing a task
100     *
101     * @param parameters the parameters for the task execution
102     * @throws Exception if processing fails
103     */
104    void executeTask(Map<String, Object> parameters) throws Exception;
105
106    /**
107     * Clean up the feeder. This is the last phase of the feeder run. It is called after feeder threads
108     * have ended.
109     *
110     * @return this feeder
111     * @throws java.io.IOException if cleanup fails
112     */
113    Feeder<T, R, P> afterRun() throws IOException;
114
115    /**
116     * Set the state of the feeder if the feeder runs in a river.
117     *
118     * @param riverState the river state
119     * @return the feeder
120     */
121    Feeder<T, R, P> setRiverState(RiverState riverState);
122
123    /**
124     * Get the feeder state if the feeder runs in a river.
125     *
126     * @return the river state
127     */
128    RiverState getRiverState();
129
130    /**
131     * Interrupt the feeder. The interruption request is delegated to all threads.
132     *
133     * @param state true if feeder should be interrupted
134     */
135    void setInterrupted(boolean state);
136
137    /**
138     * Returns if the feeder has been interrupted.
139     *
140     * @return true if interrupted
141     */
142    boolean isInterrupted();
143
144}
