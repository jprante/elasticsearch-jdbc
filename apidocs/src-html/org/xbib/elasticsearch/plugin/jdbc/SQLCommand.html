001package org.xbib.elasticsearch.plugin.jdbc;
002
003import org.elasticsearch.common.io.Streams;
004import org.elasticsearch.common.xcontent.support.XContentMapValues;
005
006import java.io.FileInputStream;
007import java.io.IOException;
008import java.io.InputStreamReader;
009import java.io.Reader;
010import java.util.Arrays;
011import java.util.HashMap;
012import java.util.LinkedList;
013import java.util.List;
014import java.util.Map;
015import java.util.regex.Pattern;
016
017/**
018 * The SQL command
019 */
020public class SQLCommand {
021
022    private String sql;
023
024    private static final Pattern STATEMENT_PATTERN = Pattern.compile("^\\s*(update|insert)", Pattern.CASE_INSENSITIVE);
025
026    private List<Object> params = new LinkedList<Object>();
027
028    private Map<String, Object> results = new HashMap<String, Object>();
029
030    private boolean callable;
031
032    public SQLCommand setSQL(String sql) throws IOException {
033        if (sql.endsWith(".sql")) {
034            Reader r = new InputStreamReader(new FileInputStream(sql), "UTF-8");
035            sql = Streams.copyToString(r);
036            r.close();
037        }
038        this.sql = sql;
039        return this;
040    }
041
042    public String getSQL() {
043        return sql;
044    }
045
046    public SQLCommand setParameters(List<Object> params) {
047        this.params = params;
048        return this;
049    }
050
051    public List<Object> getParameters() {
052        return params;
053    }
054
055    public SQLCommand setCallable(boolean callable) {
056        this.callable = callable;
057        return this;
058    }
059
060    public boolean isCallable() {
061        return callable;
062    }
063
064    public boolean isQuery() {
065        if (sql == null) {
066            throw new IllegalArgumentException("no SQL found");
067        }
068        if (STATEMENT_PATTERN.matcher(sql).find()) {
069            return false;
070        }
071        int p1 = sql.toLowerCase().indexOf("select");
072        if (p1 < 0) {
073            return false;
074        }
075        int p2 = sql.toLowerCase().indexOf("update");
076        if (p2 < 0) {
077            return true;
078        }
079        int p3 = sql.toLowerCase().indexOf("insert");
080        return p3 < 0 || p1 < p2 && p1 < p3;
081    }
082
083    public void setResults(Map<String, Object> results) {
084        this.results = results;
085    }
086
087    public Map<String, Object> getResults() {
088        return results;
089    }
090
091    @SuppressWarnings({"unchecked"})
092    public static List<SQLCommand> parse(Map<String, Object> settings) {
093        List<SQLCommand> sql = new LinkedList<SQLCommand>();
094        if (!XContentMapValues.isArray(settings.get("sql"))) {
095            settings.put("sql", Arrays.asList(settings.get("sql")));
096        }
097        List<Object> list = (List<Object>) settings.get("sql");
098        for (Object entry : list) {
099            SQLCommand command = new SQLCommand();
100            try {
101                if (entry instanceof Map) {
102                    Map<String, Object> m = (Map<String, Object>) entry;
103                    if (m.containsKey("statement")) {
104                        command.setSQL((String) m.get("statement"));
105                    }
106                    if (m.containsKey("parameter")) {
107                        command.setParameters(XContentMapValues.extractRawValues("parameter", m));
108                    }
109                    if (m.containsKey("callable")) {
110                        command.setCallable(XContentMapValues.nodeBooleanValue(m.get("callable")));
111                    }
112                    if (m.containsKey("register")) {
113                        command.setResults(XContentMapValues.nodeMapValue(m.get("register"), null));
114                    }
115                } else if (entry instanceof String) {
116                    command.setSQL((String) entry);
117                }
118                sql.add(command);
119            } catch (IOException e) {
120                throw new IllegalArgumentException("SQL command not found", e);
121            }
122        }
123        return sql;
124    }
125
126    public String toString() {
127        return "statement=" + sql + " parameter=" + params + " callable=" + callable;
128    }
129
130}
