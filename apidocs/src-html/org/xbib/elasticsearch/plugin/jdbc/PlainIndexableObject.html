001package org.xbib.elasticsearch.plugin.jdbc;
002
003import org.elasticsearch.common.xcontent.ToXContent;
004import org.elasticsearch.common.xcontent.XContentBuilder;
005
006import java.io.IOException;
007import java.util.LinkedHashMap;
008import java.util.List;
009import java.util.Map;
010
011import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
012
013/**
014 * A plain indexable object. The indexable object can store meta data and core data.
015 * The indexable object can be iterated and passed to an XContentBuilder. The individual values
016 * are formatted for JSON, which should be the correct format
017 */
018public class PlainIndexableObject implements IndexableObject, ToXContent, Comparable<IndexableObject> {
019
020    private Map<String, String> meta;
021
022    private Map<String, Object> core;
023
024    private boolean ignoreNull;
025
026    public PlainIndexableObject() {
027        this.meta = new LinkedHashMap<String, String>();
028        this.core = new LinkedHashMap<String, Object>();
029    }
030
031    @Override
032    public IndexableObject ignoreNull(boolean ignorenull) {
033        this.ignoreNull = ignorenull;
034        return this;
035    }
036
037    @Override
038    public IndexableObject optype(String optype) {
039        meta.put(ControlKeys._optype.name(), optype);
040        return this;
041    }
042
043    @Override
044    public String optype() {
045        return meta.get(ControlKeys._optype.name());
046    }
047
048    @Override
049    public IndexableObject index(String index) {
050        meta.put(ControlKeys._index.name(), index);
051        return this;
052    }
053
054    @Override
055    public String index() {
056        return meta.get(ControlKeys._index.name());
057    }
058
059    @Override
060    public IndexableObject type(String type) {
061        meta.put(ControlKeys._type.name(), type);
062        return this;
063    }
064
065    @Override
066    public String type() {
067        return meta.get(ControlKeys._type.name());
068    }
069
070    @Override
071    public IndexableObject id(String id) {
072        meta.put(ControlKeys._id.name(), id);
073        return this;
074    }
075
076    @Override
077    public String id() {
078        return meta.get(ControlKeys._id.name());
079    }
080
081    @Override
082    public IndexableObject meta(String key, String value) {
083        meta.put(key, value);
084        return this;
085    }
086
087    @Override
088    public String meta(String key) {
089        return meta.get(key);
090    }
091
092    @Override
093    public IndexableObject source(Map<String, Object> source) {
094        this.core = source;
095        return this;
096    }
097
098    @Override
099    public Map<String, Object> source() {
100        return core;
101    }
102
103    /**
104     * Build a string that can be used for indexing.
105     *
106     * @throws java.io.IOException when build gave an error
107     */
108    @Override
109    public String build() throws IOException {
110        XContentBuilder builder = jsonBuilder();
111        toXContent(builder, ToXContent.EMPTY_PARAMS);
112        return builder.string();
113    }
114
115    @Override
116    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
117        toXContent(builder, params, core);
118        return builder;
119    }
120
121    /**
122     * Recursive method to build XContent from a key/value map of Values
123     *
124     * @param builder the builder
125     * @param map     the map
126     * @return the XContent builder
127     * @throws java.io.IOException when method gave an error
128     */
129    @SuppressWarnings({"unchecked"})
130    protected XContentBuilder toXContent(XContentBuilder builder, Params params, Map<String, Object> map) throws IOException {
131        if (checkCollapsedMapLength(map)) {
132            return builder;
133        }
134        builder.startObject();
135        for (Map.Entry<String, Object> k : map.entrySet()) {
136            Object o = k.getValue();
137            if (ignoreNull && (o == null || (o instanceof Values) && ((Values) o).isNull())) {
138                continue;
139            }
140            builder.field(k.getKey());
141            if (o instanceof Values) {
142                Values v = (Values) o;
143                v.toXContent(builder, params);
144            } else if (o instanceof Map) {
145                toXContent(builder, params, (Map<String, Object>) o);
146            } else if (o instanceof List) {
147                toXContent(builder, params, (List) o);
148            } else {
149                try {
150                    builder.value(o);
151                } catch (Throwable e) {
152                    throw new IOException("unknown object class for value:" + o.getClass().getName() + " " + o);
153                }
154            }
155        }
156        builder.endObject();
157        return builder;
158    }
159
160    /**
161     * Check if the map is empty, after optional null value removal.
162     *
163     * @param map the map to check
164     * @return true if map is empty, false if not
165     */
166    protected boolean checkCollapsedMapLength(Map<String, Object> map) {
167        int exists = 0;
168        for (Map.Entry<String, Object> k : map.entrySet()) {
169            Object o = k.getValue();
170            if (ignoreNull && (o == null || (o instanceof Values) && ((Values) o).isNull())) {
171                continue;
172            }
173            exists++;
174        }
175        return exists == 0;
176    }
177
178    @SuppressWarnings({"unchecked"})
179    protected XContentBuilder toXContent(XContentBuilder builder, Params params, List list) throws IOException {
180        builder.startArray();
181        for (Object o : list) {
182            if (o instanceof Values) {
183                Values v = (Values) o;
184                v.toXContent(builder, ToXContent.EMPTY_PARAMS);
185            } else if (o instanceof Map) {
186                toXContent(builder, params, (Map<String, Object>) o);
187            } else if (o instanceof List) {
188                toXContent(builder, params, (List) o);
189            } else {
190                try {
191                    builder.value(o);
192                } catch (Exception e) {
193                    throw new IOException("unknown object class:" + o.getClass().getName());
194                }
195            }
196        }
197        builder.endArray();
198        return builder;
199    }
200
201    @Override
202    public boolean isEmpty() {
203        return optype() == null && index() == null && type() == null && id() == null && core.isEmpty();
204    }
205
206    @Override
207    public String toString() {
208        return "[" + optype() + "/" + index() + "/" + type() + "/" + id() + "]->" + core;
209    }
210
211    @Override
212    public int compareTo(IndexableObject o) {
213        if (o == null) {
214            return -1;
215        }
216        String s1 = optype() + '/' + index() + '/' + type() + '/' + id();
217        String s2 = o.optype() + '/' + o.index() + '/' + o.type() + '/' + o.id();
218        return s1.compareTo(s2);
219    }
220
221    @Override
222    public boolean equals(Object o) {
223        return o != null && o instanceof IndexableObject && hashCode() == o.hashCode();
224    }
225
226    @Override
227    public int hashCode() {
228        int hash = 3;
229        hash = 37 * hash + (optype() != null ? optype().hashCode() : 0);
230        hash = 37 * hash + (index() != null ? index().hashCode() : 0);
231        hash = 37 * hash + (type() != null ? type().hashCode() : 0);
232        hash = 37 * hash + (id() != null ? id().hashCode() : 0);
233        return hash;
234    }

236}
