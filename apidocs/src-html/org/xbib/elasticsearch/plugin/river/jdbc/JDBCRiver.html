001package org.xbib.elasticsearch.plugin.river.jdbc;
002
003import org.elasticsearch.client.Client;
004import org.elasticsearch.common.inject.Inject;
005import org.elasticsearch.common.logging.ESLogger;
006import org.elasticsearch.common.logging.ESLoggerFactory;
007import org.elasticsearch.common.settings.Settings;
008import org.elasticsearch.common.settings.loader.JsonSettingsLoader;
009import org.elasticsearch.common.util.concurrent.EsExecutors;
010import org.elasticsearch.common.xcontent.support.XContentMapValues;
011import org.elasticsearch.river.AbstractRiverComponent;
012import org.elasticsearch.river.RiverName;
013import org.elasticsearch.river.RiverSettings;
014import org.xbib.elasticsearch.action.river.execute.RunnableRiver;
015import org.xbib.elasticsearch.action.river.state.RiverState;
016import org.xbib.elasticsearch.action.river.state.StatefulRiver;
017import org.xbib.elasticsearch.plugin.feeder.Feeder;
018import org.xbib.elasticsearch.plugin.feeder.jdbc.JDBCFeeder;
019import org.xbib.elasticsearch.plugin.jdbc.RiverServiceLoader;
020import org.xbib.elasticsearch.river.jdbc.RiverFlow;
021
022import java.io.IOException;
023import java.util.Date;
024import java.util.Map;
025
026import static org.elasticsearch.common.settings.ImmutableSettings.settingsBuilder;
027import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
028
029/**
030 * JDBC river
031 */
032public class JDBCRiver extends AbstractRiverComponent implements RunnableRiver, StatefulRiver {
033
034    private final ESLogger logger = ESLoggerFactory.getLogger(JDBCRiver.class.getName());
035
036    private final Client client;
037
038    private final Feeder feeder;
039
040    private volatile Thread riverThread;
041
042    private volatile boolean closed;
043
044    @Inject
045    @SuppressWarnings({"unchecked"})
046    public JDBCRiver(RiverName riverName, RiverSettings riverSettings, Client client) {
047        super(riverName, riverSettings);
048        if (!riverSettings.settings().containsKey("jdbc")) {
049            throw new IllegalArgumentException("no 'jdbc' settings in river settings?");
050        }
051        this.client = client;
052        this.feeder = createFeeder(riverName.getType(), riverName.getName(), riverSettings);
053    }
054
055    private Feeder createFeeder(String riverType, String riverName, RiverSettings riverSettings) {
056        JDBCFeeder feeder = null;
057        try {
058            Map<String, Object> spec = (Map<String, Object>) riverSettings.settings().get("jdbc");
059            Map<String, String> loadedSettings = new JsonSettingsLoader().load(jsonBuilder().map(spec).string());
060            Settings mySettings = settingsBuilder().put(loadedSettings).build();
061            String strategy = XContentMapValues.nodeStringValue(spec.get("strategy"), "simple");
062            RiverFlow riverFlow = RiverServiceLoader.findRiverFlow(strategy);
063            logger.debug("found river flow class {} for strategy {}", riverFlow.getClass().getName(), strategy);
064            feeder = riverFlow.getFeeder();
065            logger.debug("spec = {} settings = {}", spec, mySettings.getAsMap());
066            feeder.setName(riverName)
067                    .setType(riverType)
068                    .setSpec(spec).setSettings(mySettings);
069        } catch (IOException e) {
070            logger.error(e.getMessage(), e);
071        }
072        return feeder;
073    }
074
075    @Override
076    public void start() {
077        feeder.setClient(client);
078        feeder.setRiverState(new RiverState()
079                        .setEnabled(true)
080                        .setStarted(new Date())
081                        .setName(riverName.getName())
082                        .setType(riverName.getType())
083                        .setCoordinates("_river", riverName.getName(), "_custom")
084        );
085        this.riverThread = EsExecutors.daemonThreadFactory(settings.globalSettings(),
086                "river(" + riverName().getType() + "/" + riverName().getName() + ")")
087                .newThread(feeder);
088        feeder.schedule(riverThread);
089    }
090
091    @Override
092    public void close() {
093        if (closed) {
094            return;
095        }
096        closed = true;
097        logger.info("closing river({}/{})", riverName.getType(), riverName.getName());
098        try {
099            feeder.getRiverState().setEnabled(false).save(client);
100            feeder.shutdown();
101        } catch (IOException e) {
102            logger.error(e.getMessage(), e);
103        }
104        if (riverThread != null) {
105            riverThread.interrupt();
106        }
107    }
108
109    @Override
110    public RiverState getRiverState() {
111        return feeder != null ? feeder.getRiverState() : null;
112    }
113
114    @Override
115    public void run() {
116        feeder.run();
117    }
118}
