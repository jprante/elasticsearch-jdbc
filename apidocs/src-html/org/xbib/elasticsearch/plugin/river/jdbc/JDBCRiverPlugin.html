019package org.xbib.elasticsearch.plugin.river.jdbc;
020
021import org.elasticsearch.common.inject.Inject;
022import org.elasticsearch.plugins.AbstractPlugin;
023import org.elasticsearch.rest.RestModule;
024import org.xbib.elasticsearch.rest.action.RestJDBCRiverInduceAction;
025import org.elasticsearch.river.RiversModule;
026import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
027import org.xbib.elasticsearch.river.jdbc.JDBCRiverModule;
028import org.xbib.elasticsearch.rest.action.RestJDBCRiverInduceAction;
029import org.xbib.elasticsearch.river.jdbc.JDBCRiver;
030import org.xbib.elasticsearch.river.jdbc.JDBCRiverModule;
031
032/**
033 * JDBC River plugin. This plugin is the enrty point for the
034 * Elasticsearch river service. The onModule() methods are used
035 * by reflection. The JDBC River Module - a simple one which just
036 * binds a JDBCRiver instance to a River interface - and the
037 * REST move is configured. Also, a concurrent bulk move
038 * is registered.
039 *
040 * @author JÃ¶rg Prante <joergprante@gmail.com>
041 */
042public class JDBCRiverPlugin extends AbstractPlugin {
043
044    @Inject
045    public JDBCRiverPlugin() {
046    }
047
048    @Override
049    public String name() {
050        return JDBCRiver.NAME;
051    }
052
053    @Override
054    public String description() {
055        return "JDBC River";
056    }
057
058    /**
059     * Register the JDBC river to Elasticsearch node
060     *
061     * @param module
062     */
063    public void onModule(RiversModule module) {
064        module.registerRiver(JDBCRiver.TYPE, JDBCRiverModule.class);
065    }
066
067    /**
068     * Register the REST move to Elasticsearch node
069     *
070     * @param module
071     */
072    public void onModule(RestModule module) {
073        module.addRestAction(RestJDBCRiverInduceAction.class);
074    }
075
076}
