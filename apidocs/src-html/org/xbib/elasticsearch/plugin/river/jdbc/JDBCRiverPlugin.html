001package org.xbib.elasticsearch.plugin.river.jdbc;
002
003import org.elasticsearch.common.inject.Inject;
004import org.elasticsearch.plugins.AbstractPlugin;
005import org.elasticsearch.river.RiversModule;
006
007public class JDBCRiverPlugin extends AbstractPlugin {
008
009    @Inject
010    public JDBCRiverPlugin() {
011    }
012
013    @Override
014    public String name() {
015        return "jdbc-"
016                + Build.getInstance().getVersion() + "-"
017                + Build.getInstance().getShortHash();
018    }
019
020    @Override
021    public String description() {
022        return "JDBC plugin";
023    }
024
025    /**
026     * Register the river
027     *
028     * @param module the rivers module
029     */
030    public void onModule(RiversModule module) {
031        module.registerRiver("jdbc", JDBCRiverModule.class);
032    }
033
034}
