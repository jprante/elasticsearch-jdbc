001package org.xbib.elasticsearch.action.river.state;
002
003import org.elasticsearch.action.support.nodes.NodeOperationResponse;
004import org.elasticsearch.cluster.node.DiscoveryNode;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.xbib.elasticsearch.plugin.support.RiverState;
008
009import java.io.IOException;
010
011public class NodeRiverStateResponse extends NodeOperationResponse {
012
013    private RiverState state;
014
015    NodeRiverStateResponse() {
016    }
017
018    public NodeRiverStateResponse(DiscoveryNode node) {
019        super(node);
020    }
021
022    public NodeRiverStateResponse setState(RiverState state) {
023        this.state = state;
024        return this;
025    }
026
027    public RiverState getState() {
028        return state;
029    }
030
031    @Override
032    public void readFrom(StreamInput in) throws IOException {
033        super.readFrom(in);
034        boolean b = in.readBoolean();
035        if (b) {
036            state = new RiverState();
037            state.readFrom(in);
038        } else {
039            state = null;
040        }
041    }
042
043    @Override
044    public void writeTo(StreamOutput out) throws IOException {
045        super.writeTo(out);
046        if (state != null) {
047            out.writeBoolean(true);
048            state.writeTo(out);
049        } else {
050            out.writeBoolean(false);
051        }
052    }
053
054}
