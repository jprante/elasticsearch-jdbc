001package org.xbib.elasticsearch.action.river.state;
002
003import org.elasticsearch.action.support.nodes.NodesOperationResponse;
004import org.elasticsearch.common.io.stream.StreamInput;
005import org.elasticsearch.common.io.stream.StreamOutput;
006import org.xbib.common.xcontent.ToXContent;
007import org.xbib.common.xcontent.XContentBuilder;
008import org.xbib.elasticsearch.plugin.support.RiverState;
009
010import java.io.IOException;
011import java.util.Collection;
012import java.util.Set;
013import java.util.TreeSet;
014
015public class RiverStateResponse extends NodesOperationResponse implements ToXContent {
016
017    private Set<RiverState> state;
018
019    public RiverStateResponse() {
020        state = new TreeSet<>();
021    }
022
023    public RiverStateResponse addState(RiverState state) {
024        this.state.add(state);
025        return this;
026    }
027
028    public Collection<RiverState> getStates() {
029        return state;
030    }
031
032    @Override
033    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
034        builder.field("state", state);
035        return builder;
036    }
037
038    @Override
039    public void readFrom(StreamInput in) throws IOException {
040        super.readFrom(in);
041        state = new TreeSet<>();
042        int len = in.readInt();
043        for (int i = 0; i < len; i++) {
044            RiverState rs = new RiverState();
045            rs.readFrom(in);
046            state.add(rs);
047        }
048    }
049
050    @Override
051    public void writeTo(StreamOutput out) throws IOException {
052        super.writeTo(out);
053        out.writeInt(state.size());
054        for (RiverState rs : state) {
055            rs.writeTo(out);
056        }
057    }
058
059}
