001package org.xbib.elasticsearch.action.river.state;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.admin.cluster.node.info.NodeInfo;
005import org.elasticsearch.action.support.nodes.TransportNodesOperationAction;
006import org.elasticsearch.cluster.ClusterName;
007import org.elasticsearch.cluster.ClusterService;
008import org.elasticsearch.common.inject.Inject;
009import org.elasticsearch.common.inject.Injector;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.node.service.NodeService;
012import org.elasticsearch.river.River;
013import org.elasticsearch.river.RiverName;
014import org.elasticsearch.threadpool.ThreadPool;
015import org.elasticsearch.transport.TransportService;
016import org.xbib.elasticsearch.plugin.river.StatefulRiver;
017import org.xbib.elasticsearch.plugin.support.RiverHelper;
018
019import java.util.Map;
020import java.util.concurrent.atomic.AtomicReferenceArray;
021
022public class TransportRiverStateAction extends TransportNodesOperationAction<RiverStateRequest, RiverStateResponse, NodeRiverStateRequest, NodeRiverStateResponse> {
023
024    private final NodeService nodeService;
025    private final Injector injector;
026
027    @Inject
028    public TransportRiverStateAction(Settings settings, ClusterName clusterName, ThreadPool threadPool,
029                                     ClusterService clusterService, TransportService transportService,
030                                     NodeService nodeService, Injector injector) {
031        super(settings, clusterName, threadPool, clusterService, transportService);
032        this.nodeService = nodeService;
033        this.injector = injector;
034    }
035
036    @Override
037    protected String transportAction() {
038        return RiverStateAction.NAME;
039    }
040
041    @Override
042    protected String executor() {
043        return ThreadPool.Names.GENERIC;
044    }
045
046    @Override
047    protected NodeRiverStateResponse nodeOperation(NodeRiverStateRequest request) throws ElasticsearchException {
048        NodeInfo nodeInfo = nodeService.info(false, true, false, true, false, false, true, false, true);
049        String riverType = request.getRiverType();
050        String riverName = request.getRiverName();
051        for (Map.Entry<RiverName, River> entry : RiverHelper.rivers(injector).entrySet()) {
052            RiverName name = entry.getKey();
053            if ((riverName == null || name.getName().equals(riverName))
054                    && (riverType == null || name.getType().equals(riverType))
055                    && entry.getValue() instanceof StatefulRiver) {
056                StatefulRiver river = (StatefulRiver) entry.getValue();
057                return new NodeRiverStateResponse(nodeInfo.getNode()).setState(river.getRiverState());
058            }
059        }
060        return new NodeRiverStateResponse(nodeInfo.getNode());
061    }
062
063    @Override
064    protected RiverStateRequest newRequest() {
065        return new RiverStateRequest();
066    }
067
068    @Override
069    protected RiverStateResponse newResponse(RiverStateRequest request, AtomicReferenceArray shardsResponses) {
070        RiverStateResponse riverStateResponse = new RiverStateResponse();
071        for (int i = 0; i < shardsResponses.length(); i++) {
072            Object shardsResponse = shardsResponses.get(i);
073            if (shardsResponse instanceof NodeRiverStateResponse) {
074                NodeRiverStateResponse nodeRiverStateResponse = (NodeRiverStateResponse) shardsResponse;
075                if (nodeRiverStateResponse.getState() != null) {
076                    riverStateResponse.addState(nodeRiverStateResponse.getState());
077                }
078            }
079        }
080        return riverStateResponse;
081    }
082
083    @Override
084    protected NodeRiverStateRequest newNodeRequest() {
085        return new NodeRiverStateRequest();
086    }
087
088    @Override
089    protected NodeRiverStateRequest newNodeRequest(String nodeId, RiverStateRequest request) {
090        return new NodeRiverStateRequest(nodeId, request);
091    }
092
093    @Override
094    protected NodeRiverStateResponse newNodeResponse() {
095        return new NodeRiverStateResponse();
096    }
097
098    @Override
099    protected boolean accumulateExceptions() {
100        return false;
101    }
102
103}
