001package org.xbib.elasticsearch.action.river.jdbc.state.get;
002
003import org.elasticsearch.action.ActionRequestValidationException;
004import org.elasticsearch.action.support.master.MasterNodeReadOperationRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007
008import java.io.IOException;
009
010public class GetRiverStateRequest extends MasterNodeReadOperationRequest<GetRiverStateRequest> {
011
012    private String riverName;
013
014    private String riverType;
015
016    public GetRiverStateRequest setRiverName(String riverName) {
017        this.riverName = riverName;
018        return this;
019    }
020
021    public String getRiverName() {
022        return riverName;
023    }
024
025    public GetRiverStateRequest setRiverType(String riverType) {
026        this.riverType = riverType;
027        return this;
028    }
029
030    public String getRiverType() {
031        return riverType;
032    }
033
034    @Override
035    public ActionRequestValidationException validate() {
036        return null;
037    }
038
039    @Override
040    public void readFrom(StreamInput in) throws IOException {
041        super.readFrom(in);
042        this.riverName = in.readString();
043        this.riverType = in.readString();
044    }
045
046    @Override
047    public void writeTo(StreamOutput out) throws IOException {
048        super.writeTo(out);
049        out.writeString(riverName);
050        out.writeString(riverType);
051    }
052}
