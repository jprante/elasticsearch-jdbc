001package org.xbib.elasticsearch.action.river.jdbc.state.get;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.ActionListener;
005import org.elasticsearch.action.support.master.TransportMasterNodeReadOperationAction;
006import org.elasticsearch.cluster.ClusterService;
007import org.elasticsearch.cluster.ClusterState;
008import org.elasticsearch.cluster.metadata.MetaData;
009import org.elasticsearch.common.collect.ImmutableList;
010import org.elasticsearch.common.inject.Inject;
011import org.elasticsearch.common.inject.Injector;
012import org.elasticsearch.common.settings.Settings;
013import org.elasticsearch.river.River;
014import org.elasticsearch.river.RiverName;
015import org.elasticsearch.threadpool.ThreadPool;
016import org.elasticsearch.transport.TransportService;
017import org.xbib.elasticsearch.action.river.jdbc.state.RiverState;
018import org.xbib.elasticsearch.action.river.jdbc.state.RiverStatesMetaData;
019import org.xbib.elasticsearch.action.river.jdbc.state.StatefulRiver;
020import org.xbib.elasticsearch.support.river.RiverHelper;
021
022import java.util.Map;
023
024public class TransportGetRiverStateAction extends TransportMasterNodeReadOperationAction<GetRiverStateRequest, GetRiverStateResponse> {
025
026    private final Injector injector;
027
028    @Inject
029    public TransportGetRiverStateAction(Settings settings, ThreadPool threadPool,
030                                        ClusterService clusterService, TransportService transportService,
031                                        Injector injector) {
032        super(settings, GetRiverStateAction.NAME, transportService, clusterService, threadPool);
033        this.injector = injector;
034    }
035
036    @Override
037    protected String executor() {
038        return ThreadPool.Names.MANAGEMENT;
039    }
040
041    @Override
042    protected GetRiverStateRequest newRequest() {
043        return new GetRiverStateRequest();
044    }
045
046    @Override
047    protected GetRiverStateResponse newResponse() {
048        return new GetRiverStateResponse();
049    }
050
051    @Override
052    protected void masterOperation(GetRiverStateRequest request, ClusterState state, ActionListener<GetRiverStateResponse> listener) throws ElasticsearchException {
053        MetaData metaData = state.metaData();
054        RiverStatesMetaData riverStates = metaData.custom(RiverStatesMetaData.TYPE);
055        if (request.getRiverName() == null || request.getRiverType() == null) {
056            listener.onResponse(new GetRiverStateResponse(request, riverStates.rivers()));
057        } else {
058            String riverName = request.getRiverName();
059            String riverType = request.getRiverType();
060            ImmutableList.Builder<RiverState> builder = ImmutableList.builder();
061            for (Map.Entry<RiverName, River> entry : RiverHelper.rivers(injector).entrySet()) {
062                RiverName name = entry.getKey();
063                if (("*".equals(riverName) || name.getName().equals(riverName)) && ("*".equals(riverType) || name.getType().equals(riverType)) && entry.getValue() instanceof StatefulRiver) {
064                    StatefulRiver river = (StatefulRiver) entry.getValue();
065                    builder.add(river.getRiverState());
066                }
067            }
068            listener.onResponse(new GetRiverStateResponse(request, builder.build()));
069        }
070    }
071
072}
