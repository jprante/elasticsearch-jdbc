001package org.xbib.elasticsearch.action.river.jdbc.execute;
002
003import org.elasticsearch.ElasticsearchException;
004import org.elasticsearch.action.admin.cluster.node.info.NodeInfo;
005import org.elasticsearch.action.support.nodes.TransportNodesOperationAction;
006import org.elasticsearch.cluster.ClusterName;
007import org.elasticsearch.cluster.ClusterService;
008import org.elasticsearch.common.inject.Inject;
009import org.elasticsearch.common.inject.Injector;
010import org.elasticsearch.common.settings.Settings;
011import org.elasticsearch.node.service.NodeService;
012import org.elasticsearch.river.River;
013import org.elasticsearch.river.RiverName;
014import org.elasticsearch.threadpool.ThreadPool;
015import org.elasticsearch.transport.TransportService;
016import org.xbib.elasticsearch.support.river.RiverHelper;
017
018import java.util.Map;
019import java.util.concurrent.atomic.AtomicReferenceArray;
020
021public class TransportRiverExecuteAction extends TransportNodesOperationAction<RiverExecuteRequest, RiverExecuteResponse, NodeRiverExecuteRequest, NodeRiverExecuteResponse> {
022
023    private final NodeService nodeService;
024
025    private final Injector injector;
026
027    @Inject
028    public TransportRiverExecuteAction(Settings settings, ClusterName clusterName, ThreadPool threadPool,
029                                       ClusterService clusterService, TransportService transportService,
030                                       NodeService nodeService, Injector injector) {
031        super(settings, RiverExecuteAction.NAME, clusterName, threadPool, clusterService, transportService);
032        this.nodeService = nodeService;
033        this.injector = injector;
034    }
035
036    @Override
037    protected String executor() {
038        return ThreadPool.Names.GENERIC;
039    }
040
041    @Override
042    protected NodeRiverExecuteResponse nodeOperation(NodeRiverExecuteRequest request) throws ElasticsearchException {
043        NodeInfo nodeInfo = nodeService.info(false, true, false, true, false, false, true, false, true);
044        String riverType = request.getRiverType();
045        String riverName = request.getRiverName();
046        for (Map.Entry<RiverName, River> entry : RiverHelper.rivers(injector).entrySet()) {
047            RiverName name = entry.getKey();
048            if ((riverName == null || name.getName().equals(riverName))
049                    && (riverType == null || name.getType().equals(riverType))
050                    && entry.getValue() instanceof RunnableRiver) {
051                RunnableRiver river = (RunnableRiver) entry.getValue();
052                river.run();
053                return new NodeRiverExecuteResponse(nodeInfo.getNode()).setExecuted(true);
054            }
055        }
056        return new NodeRiverExecuteResponse(nodeInfo.getNode()).setExecuted(false);
057    }
058
059    @Override
060    protected RiverExecuteRequest newRequest() {
061        return new RiverExecuteRequest();
062    }
063
064    @Override
065    protected RiverExecuteResponse newResponse(RiverExecuteRequest request, AtomicReferenceArray nodesResponses) {
066        boolean[] b = new boolean[nodesResponses.length()];
067        for (int i = 0; i < nodesResponses.length(); i++) {
068            Object nodesResponse = nodesResponses.get(i);
069            if (nodesResponse instanceof NodeRiverExecuteResponse) {
070                NodeRiverExecuteResponse nodeRiverExecuteResponse = (NodeRiverExecuteResponse) nodesResponse;
071                b[i] = nodeRiverExecuteResponse.isExecuted();
072            }
073        }
074        return new RiverExecuteResponse().setExecuted(b);
075    }
076
077    @Override
078    protected NodeRiverExecuteRequest newNodeRequest() {
079        return new NodeRiverExecuteRequest();
080    }
081
082    @Override
083    protected NodeRiverExecuteRequest newNodeRequest(String nodeId, RiverExecuteRequest request) {
084        return new NodeRiverExecuteRequest(nodeId, request);
085    }
086
087    @Override
088    protected NodeRiverExecuteResponse newNodeResponse() {
089        return new NodeRiverExecuteResponse();
090    }
091
092    @Override
093    protected boolean accumulateExceptions() {
094        return false;
095    }
096
097}
