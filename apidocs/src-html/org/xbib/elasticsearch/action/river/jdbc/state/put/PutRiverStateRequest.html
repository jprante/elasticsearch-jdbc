001package org.xbib.elasticsearch.action.river.jdbc.state.put;
002
003import org.elasticsearch.action.ActionRequestValidationException;
004import org.elasticsearch.action.support.master.AcknowledgedRequest;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.xbib.elasticsearch.action.river.jdbc.state.RiverState;
008
009import java.io.IOException;
010
011import static org.elasticsearch.action.ValidateActions.addValidationError;
012
013public class PutRiverStateRequest extends AcknowledgedRequest<PutRiverStateRequest> {
014
015    private String riverName;
016
017    private String riverType;
018
019    private RiverState riverState;
020
021    public PutRiverStateRequest setRiverName(String riverName) {
022        this.riverName = riverName;
023        return this;
024    }
025
026    public String getRiverName() {
027        return riverName;
028    }
029
030    public PutRiverStateRequest setRiverType(String riverType) {
031        this.riverType = riverType;
032        return this;
033    }
034
035    public String getRiverType() {
036        return riverType;
037    }
038
039
040    public PutRiverStateRequest setRiverState(RiverState riverState) {
041        this.riverState = riverState;
042        return this;
043    }
044
045    public RiverState getRiverState() {
046        return riverState;
047    }
048
049    @Override
050    public ActionRequestValidationException validate() {
051        ActionRequestValidationException validationException = null;
052        if (riverName == null) {
053            validationException = addValidationError("name is missing", null);
054        }
055        if (riverType == null) {
056            validationException = addValidationError("type is missing", validationException);
057        }
058        if (riverState == null) {
059            validationException = addValidationError("river state is missing", validationException);
060        }
061        return validationException;
062    }
063
064    @Override
065    public void readFrom(StreamInput in) throws IOException {
066        super.readFrom(in);
067        readTimeout(in);
068        this.riverName = in.readString();
069        this.riverType = in.readString();
070        this.riverState = new RiverState();
071        riverState.readFrom(in);
072    }
073
074    @Override
075    public void writeTo(StreamOutput out) throws IOException {
076        super.writeTo(out);
077        writeTimeout(out);
078        out.writeString(riverName);
079        out.writeString(riverType);
080        riverState.writeTo(out);
081    }
082}
