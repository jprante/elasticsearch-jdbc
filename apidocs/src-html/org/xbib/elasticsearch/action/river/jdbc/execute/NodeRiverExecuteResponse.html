001package org.xbib.elasticsearch.action.river.jdbc.execute;
002
003import org.elasticsearch.action.support.nodes.NodeOperationResponse;
004import org.elasticsearch.cluster.node.DiscoveryNode;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007
008import java.io.IOException;
009
010public class NodeRiverExecuteResponse extends NodeOperationResponse {
011
012    private boolean executed;
013
014    NodeRiverExecuteResponse() {
015    }
016
017    public NodeRiverExecuteResponse(DiscoveryNode node) {
018        super(node);
019    }
020
021    public NodeRiverExecuteResponse setExecuted(boolean b) {
022        this.executed = b;
023        return this;
024    }
025
026    public boolean isExecuted() {
027        return executed;
028    }
029
030    @Override
031    public void readFrom(StreamInput in) throws IOException {
032        super.readFrom(in);
033        executed = in.readBoolean();
034    }
035
036    @Override
037    public void writeTo(StreamOutput out) throws IOException {
038        super.writeTo(out);
039        out.writeBoolean(executed);
040    }
041
042}
