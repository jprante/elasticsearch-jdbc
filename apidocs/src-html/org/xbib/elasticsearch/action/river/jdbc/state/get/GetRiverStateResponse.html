001package org.xbib.elasticsearch.action.river.jdbc.state.get;
002
003import org.elasticsearch.action.ActionResponse;
004import org.elasticsearch.common.collect.ImmutableList;
005import org.elasticsearch.common.io.stream.StreamInput;
006import org.elasticsearch.common.io.stream.StreamOutput;
007import org.elasticsearch.common.xcontent.ToXContent;
008import org.elasticsearch.common.xcontent.XContentBuilder;
009import org.xbib.elasticsearch.action.river.jdbc.state.RiverState;
010
011import java.io.IOException;
012
013public class GetRiverStateResponse extends ActionResponse implements ToXContent {
014
015    private GetRiverStateRequest getRiverStateRequest;
016
017    private ImmutableList<RiverState> states;
018
019    public GetRiverStateResponse() {
020        states = ImmutableList.of();
021    }
022
023    public GetRiverStateResponse(GetRiverStateRequest request, ImmutableList<RiverState> riverStates) {
024        getRiverStateRequest = request;
025        states = riverStates;
026    }
027
028    public RiverState getState() {
029        if (states == null || states.isEmpty()) {
030            return new RiverState(getRiverStateRequest.getRiverName(), getRiverStateRequest.getRiverType());
031        } else {
032            return states.get(0);
033        }
034    }
035
036    public ImmutableList<RiverState> getStates() {
037        return states;
038    }
039
040    @Override
041    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
042        builder.field("state", states);
043        return builder;
044    }
045
046    @Override
047    public void readFrom(StreamInput in) throws IOException {
048        super.readFrom(in);
049        getRiverStateRequest = new GetRiverStateRequest();
050        getRiverStateRequest.readFrom(in);
051        int len = in.readInt();
052        ImmutableList.Builder<RiverState> builder = ImmutableList.builder();
053        for (int i = 0; i < len; i++) {
054            RiverState rs = new RiverState();
055            rs.readFrom(in);
056            builder.add(rs);
057        }
058        states = builder.build();
059    }
060
061    @Override
062    public void writeTo(StreamOutput out) throws IOException {
063        super.writeTo(out);
064        getRiverStateRequest.writeTo(out);
065        out.writeInt(states.size());
066        for (RiverState rs : states) {
067            rs.writeTo(out);
068        }
069    }
070
071}
