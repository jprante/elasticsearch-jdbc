110    public NodeClient newClient(Client client) {
111        this.client = client;
112        this.state = new State();
113        BulkProcessor.Listener listener = new BulkProcessor.Listener() {
114            @Override
115            public void beforeBulk(long executionId, BulkRequest request) {
116                long l = concurrentRequestCounter.getAndIncrement();
117                int n = request.numberOfActions();
118                state.getSubmitted().inc(n);
119                state.getCurrentIngestNumDocs().inc(n);
120                state.getTotalIngestSizeInBytes().inc(request.estimatedSizeInBytes());
121                logger.debug("before bulk [{}] [actions={}] [bytes={}] [concurrent requests={}]",
122                            executionId,
123                            request.numberOfActions(),
124                            request.estimatedSizeInBytes(),
125                            l);
126            }
127
128            @Override
129            public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
130                long l = concurrentRequestCounter.decrementAndGet();
131                state.getSucceeded().inc(response.getItems().length);
132                state.getFailed().inc(0);
133                state.getTotalIngest().inc(response.getTookInMillis());
134                int n = 0;
135                for (BulkItemResponse itemResponse : response.getItems()) {
136                    if (itemResponse.isFailed()) {
137                        n++;
138                        state.getSucceeded().dec(1);
139                        state.getFailed().inc(1);
140                    }
141                }
142                logger.debug("after bulk [{}] [succeeded={}] [failed={}] [{}ms] [concurrent requests={}]",
143                            executionId,
144                            state.getSucceeded().count(),
145                            state.getFailed().count(),
146                            response.getTook().millis(),
147                            l);
148                if (n > 0) {
149                    logger.error("bulk [{}] failed with {} failed items, failure message = {}",
150                            executionId, n, response.buildFailureMessage());
151                } else {
152                    state.getCurrentIngestNumDocs().dec(response.getItems().length);
153                }
154            }
155
156            @Override
157            public void afterBulk(long executionId, BulkRequest request, Throwable failure) {
158                concurrentRequestCounter.decrementAndGet();
159                throwable = failure;
160                closed = true;
161                logger.error("after bulk [" + executionId + "] error", failure);
162            }
163        };
164        BulkProcessor.Builder builder = BulkProcessor.builder(client, listener)
165                .setBulkActions(maxActionsPerBulkRequest)  // off-by-one
166                .setConcurrentRequests(maxConcurrentBulkRequests)
167                .setFlushInterval(flushInterval);
168        if (maxVolume != null) {
169            builder.setBulkSize(maxVolume);
170        }
171        this.bulkProcessor = builder.build();
172        try {
173            waitForCluster(ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
174            closed = false;
175        } catch (IOException e) {
176            logger.error(e.getMessage(), e);
177            closed = true;
178        }
179        return this;
180    }

212    @Override
213    public NodeClient index(String index, String type, String id, String source) {
214        if (closed) {
215            throw new ElasticsearchIllegalStateException("client is closed");
216        }
217        try {
218            if (state != null) {
219                state.getCurrentIngest().inc();
220            }
221            bulkProcessor.add(new IndexRequest(index).type(type).id(id).create(false).source(source));
222        } catch (Exception e) {
223            throwable = e;
224            closed = true;
225            logger.error("bulk add of index request failed: " + e.getMessage(), e);
226        } finally {
227            if (state != null) {
228                state.getCurrentIngest().dec();
229            }
230        }
231        return this;
232    }

321        if (state == null) {
322            return this;

326            ClientHelper.disableRefresh(client, index);

372    @Override
373    public synchronized void shutdown() {
374        try {
375            if (bulkProcessor != null) {
376                logger.debug("closing bulk processor...");
377                bulkProcessor.close();
378            }
379            if (state != null && state.indices() != null && !state.indices().isEmpty()) {
380                logger.debug("stopping bulk mode for indices {}...", state.indices());
381                for (String index : ImmutableSet.copyOf(state.indices())) {
382                    stopBulk(index);
383                }
384            }
385            logger.debug("shutting down...");
386            client.close();
387            logger.debug("shutting down completed");
388        } catch (Exception e) {
389            logger.error(e.getMessage(), e);
390        }
391    }

393    @Override
394    public NodeClient newIndex(String index) {
395        if (closed) {
396            throw new ElasticsearchIllegalStateException("client is closed");
397        }
398        if (client == null) {
399            logger.warn("no client");
400            return this;
401        }
402        if (index == null) {
403            logger.warn("no index name given to create index");
404            return this;
405        }
406        CreateIndexRequest request = new CreateIndexRequest(index);
407        if (getSettings() != null) {
408            request.settings(getSettings());
409        }
410        if (getMappings() != null) {
411            for (Map.Entry<String, String> me : getMappings().entrySet()) {
412                request.mapping(me.getKey(), me.getValue());
413            }
414        }
415        logger.info("creating index {} with settings = {}, mappings = {}",
416                index, getSettings() != null ? getSettings().getAsMap() : "", getMappings());
417        try {
418            client.admin().indices().create(request).actionGet();
419        } catch (Exception e) {
420            logger.error(e.getMessage(), e);
421        }
422        return this;
423    }

438        try {

440        } catch (Exception e) {

442        }
