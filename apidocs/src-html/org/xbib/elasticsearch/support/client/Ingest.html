001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.action.admin.cluster.health.ClusterHealthStatus;
004import org.elasticsearch.action.delete.DeleteRequest;
005import org.elasticsearch.action.index.IndexRequest;
006import org.elasticsearch.client.Client;
007import org.elasticsearch.common.settings.ImmutableSettings;
008import org.elasticsearch.common.settings.Settings;
009import org.elasticsearch.common.unit.ByteSizeValue;
010import org.elasticsearch.common.unit.TimeValue;
011
012import java.io.IOException;
013import java.io.InputStream;
014import java.net.URI;
015import java.util.Map;
016
017/**
018 * Interface for providing convenient administrative methods for ingesting data into Elasticsearch.
019 */
020public interface Ingest extends Feed {
021
022    Ingest newClient(Client client);
023
024    Ingest newClient(URI uri);
025
026    /**
027     * Set the maximum number of actions per bulk request
028     *
029     * @param maxActions maximum number of bulk actions
030     * @return this ingest
031     */
032    Ingest maxActionsPerBulkRequest(int maxActions);
033
034    /**
035     * Set the maximum concurent bulk requests
036     *
037     * @param maxConcurentBulkRequests maximum number of concurrent ingest requests
038     * @return this Ingest
039     */
040    Ingest maxConcurrentBulkRequests(int maxConcurentBulkRequests);
041
042    /**
043     * Set the maximum volume for bulk request before flush
044     *
045     * @param maxVolume maximum volume
046     * @return this ingest
047     */
048    Ingest maxVolumePerBulkRequest(ByteSizeValue maxVolume);
049
050    /**
051     * Set the flush interval for automatic flushing outstanding ingest requests
052     *
053     * @param flushInterval the flush interval, default is 30 seconds
054     * @return this ingest
055     */
056    Ingest flushIngestInterval(TimeValue flushInterval);
057
058    /**
059     * Set request timeout. Default is 60s.
060     *
061     * @param timeout timeout
062     * @return this ingest
063     */
064    Ingest maxRequestWait(TimeValue timeout);
065
066    /**
067     * The number of shards for index creation
068     *
069     * @param shards the number of shards
070     * @return this
071     */
072    Ingest shards(int shards);
073
074    /**
075     * The number of replica for index creation
076     *
077     * @param replica the number of replica
078     * @return this
079     */
080    Ingest replica(int replica);
081
082    void setSettings(Settings settings);
083
084    ImmutableSettings.Builder getSettingsBuilder();
085
086    Settings getSettings();
087
088    /**
089     * Create a key/value in the settings
090     *
091     * @param key   the key
092     * @param value the value
093     */
094    void addSetting(String key, String value);
095
096    /**
097     * Create a key/value in the settings
098     *
099     * @param key   the key
100     * @param value the value
101     */
102    void addSetting(String key, Boolean value);
103
104    /**
105     * Create a key/value in the settings
106     *
107     * @param key   the key
108     * @param value the value
109     */
110    void addSetting(String key, Integer value);
111
112    /**
113     * Create a key/value in the settings
114     *
115     * @param in the input stream with settings
116     */
117    void addSetting(InputStream in) throws IOException;
118
119    void addMapping(String type, InputStream in) throws IOException;
120
121    void addMapping(String type, String mapping);
122
123    Map<String, String> getMappings();
124
125    Ingest putMapping(String index);
126
127    Ingest deleteMapping(String index, String type);
128
129    /**
130     * Create a new index
131     *
132     * @return this ingest
133     */
134    Ingest newIndex(String index);
135
136    /**
137     * Delete index
138     *
139     * @return this ingest
140     */
141    Ingest deleteIndex(String index);
142
143    /**
144     * Start bulk mode
145     *
146     * @return this ingest
147     */
148    Ingest startBulk(String index) throws IOException;
149
150    /**
151     * Stops bulk mode. Enables refresh.
152     *
153     * @return this Ingest
154     */
155    Ingest stopBulk(String index) throws IOException;
156
157    /**
158     * Bulked index request. Each request will be added to a queue for bulking requests.
159     * Submitting request will be done when bulk limits are exceeded.
160     *
161     * @param indexRequest the index request to add
162     * @return this ingest
163     */
164    Ingest bulkIndex(IndexRequest indexRequest);
165
166    /**
167     * Bulked delete request. Each request will be added to a queue for bulking requests.
168     * Submitting request will be done when bulk limits are exceeded.
169     *
170     * @param deleteRequest the delete request to add
171     * @return this ingest
172     */
173    Ingest bulkDelete(DeleteRequest deleteRequest);
174
175    /**
176     * Flush ingest, move all pending documents to the bulk indexer
177     *
178     * @return this
179     */
180    Ingest flushIngest();
181
182    /**
183     * Wait for all outstanding responses
184     *
185     * @param maxWait maximum wait time
186     * @return this ingest
187     * @throws InterruptedException
188     */
189    Ingest waitForResponses(TimeValue maxWait) throws InterruptedException;
190
191    /**
192     * Flush the index
193     */
194    Ingest flush(String index);
195
196    /**
197     * Refresh the index.
198     *
199     * @return this ingest
200     */
201    Ingest refresh(String index);
202
203    /**
204     * Add replica level.
205     *
206     * @param level the replica level
207     * @return number of shards after updating replica level
208     */
209    int updateReplicaLevel(String index, int level) throws IOException;
210
211    /**
212     * Wait for cluster being healthy.
213     *
214     * @throws java.io.IOException
215     */
216    Ingest waitForCluster(ClusterHealthStatus status, TimeValue timeValue) throws IOException;
217
218    /**
219     * Wait for index recovery (after replica change)
220     *
221     * @return number of shards found
222     */
223    int waitForRecovery(String index) throws IOException;
224
225    State getState();
226
227    boolean hasThrowable();
228
229    /**
230     * Return last throwable if exists.
231     *
232     * @return last throwable
233     */
234    Throwable getThrowable();
235
236    /**
237     * Shutdown the ingesting
238     */
239    void shutdown();
240}
