001package org.xbib.elasticsearch.support.client;
002
003import org.elasticsearch.client.Client;
004
005/**
006 * Minimal API for feed
007 */
008public interface Feed {
009
010    Client client();
011
012    /**
013     * Index document
014     *
015     * @param index  the index
016     * @param type   the type
017     * @param id     the id
018     * @param source the source
019     * @return this
020     */
021    Feed index(String index, String type, String id, String source);
022
023    /**
024     * Delete document
025     *
026     * @param index the index
027     * @param type  the type
028     * @param id    the id
029     * @return this
030     */
031    Feed delete(String index, String type, String id);
032
033}
