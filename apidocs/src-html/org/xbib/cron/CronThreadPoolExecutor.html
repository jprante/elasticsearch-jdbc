001package org.xbib.cron;
002
003import java.util.Date;
004import java.util.concurrent.CancellationException;
005import java.util.concurrent.RejectedExecutionException;
006import java.util.concurrent.RejectedExecutionHandler;
007import java.util.concurrent.ScheduledThreadPoolExecutor;
008import java.util.concurrent.ThreadFactory;
009import java.util.concurrent.TimeUnit;
010
011/**
012 * Scheduled thread-pool executor implementation that leverages a CronExpression
013 * to calculate future execution times for scheduled tasks.
014 */
015public class CronThreadPoolExecutor extends ScheduledThreadPoolExecutor implements CronExecutorService {
016    /**
017     * Constructs a new CronThreadPoolExecutor.
018     *
019     * @param corePoolSize the pool size
020     */
021    public CronThreadPoolExecutor(int corePoolSize) {
022        super(corePoolSize);
023    }
024
025    /**
026     * Constructs a new CronThreadPoolExecutor.
027     *
028     * @param corePoolSize  the pool size
029     * @param threadFactory the thread factory
030     */
031    public CronThreadPoolExecutor(int corePoolSize, ThreadFactory threadFactory) {
032        super(corePoolSize, threadFactory);
033    }
034
035    /**
036     * Constructs a new CronThreadPoolExecutor.
037     *
038     * @param corePoolSize the pool size
039     * @param handler      the handler for rejected executions
040     */
041    public CronThreadPoolExecutor(int corePoolSize, RejectedExecutionHandler handler) {
042        super(corePoolSize, handler);
043    }
044
045    /**
046     * Constructs a new CronThreadPoolExecutor.
047     *
048     * @param corePoolSize  the pool size
049     * @param handler       the handler for rejecting executions
050     * @param threadFactory the thread factory
051     */
052    public CronThreadPoolExecutor(int corePoolSize, ThreadFactory threadFactory, RejectedExecutionHandler handler) {
053        super(corePoolSize, threadFactory, handler);
054    }
055
056    @Override
057    public void schedule(final Runnable task, final CronExpression expression) {
058        if (task == null) {
059            throw new NullPointerException();
060        }
061        this.setCorePoolSize(this.getCorePoolSize() + 1);
062        Runnable scheduleTask = new Runnable() {
063            /**
064             * @see Runnable#run()
065             */
066            @Override
067            public void run() {
068                Date now = new Date();
069                Date time = expression.getNextValidTimeAfter(now);
070                try {
071                    while (time != null) {
072                        CronThreadPoolExecutor.this.schedule(task, time.getTime() - now.getTime(), TimeUnit.MILLISECONDS);
073                        while (now.before(time)) {
074                            Thread.sleep(time.getTime() - now.getTime());
075                            now = new Date();
076                        }
077                        time = expression.getNextValidTimeAfter(now);
078                    }
079                } catch (RejectedExecutionException e) {
080                    //
081                } catch (CancellationException e) {
082                    //
083                } catch (InterruptedException e) {
084                    //
085                    Thread.currentThread().interrupt();
086                }
087            }
088        };
089        this.execute(scheduleTask);
090    }
091}
