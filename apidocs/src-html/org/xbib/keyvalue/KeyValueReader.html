032package org.xbib.keyvalue;
033
034import java.io.BufferedReader;
035import java.io.IOException;
036import java.io.Reader;
037import java.util.List;
038
039public class KeyValueReader extends BufferedReader
040        implements KeyValueStreamListener<String, String> {
041
042    private KeyValueStreamListener<String, String> listener;
043
044    private boolean atStart;
045
046    private boolean atEnd;
047
048    private final char delimiter;
049
050    public KeyValueReader(Reader reader) throws IOException {
051        this(reader, '=');
052    }
053    
054    public KeyValueReader(Reader reader, char delimiter) throws IOException {
055        super(reader);
056        this.atStart = true;
057        this.atEnd = false;
058        this.delimiter = delimiter;
059    }
060
061    public KeyValueReader addListener(KeyValueStreamListener<String, String> listener) {
062        this.listener = listener;
063        return this;
064    }
065
066    @Override
067    public int read() throws IOException {
068        if (atStart) {
069            atStart = false;
070            begin();
071        }
072        return super.read();
073    }
074
075    @Override
076    public int read(char[] buf, int off, int len) throws IOException {
077        if (atStart) {
078            atStart = false;
079            begin();
080        }
081        return super.read(buf, off, len);
082    }
083
084    @Override
085    public String readLine() throws IOException {
086        if (atStart) {
087            atStart = false;
088            begin();
089        }
090        String line = super.readLine();
091        if (line != null) {
092            int pos = line.indexOf(delimiter);
093            if (pos > 0) {
094                keyValue(line.substring(0, pos), line.substring(pos+1));
095            }
096        }
097        return line;
098    }
099
100    @Override
101    public void close() throws IOException {
102        if (!atEnd) {
103            atEnd = true;
104            end();
105        }
106        super.close();
107    }
108
109    @Override
110    public KeyValueReader begin() throws IOException {
111        if (listener != null) {
112            listener.begin();
113        }
114        return this;
115    }
116
117    @Override
118    public KeyValueReader keyValue(String key, String value) throws IOException {
119        if (listener != null) {
120            listener.keyValue(key, value);
121        }
122        return this;
123    }
124
125    @Override
126    public KeyValueStreamListener<String, String> keys(List<String> keys) throws IOException {
127        if (listener != null) {
128            listener.keys(keys);
129        }
130        return this;
131    }
132
133    @Override
134    public KeyValueStreamListener<String, String> values(List<String> values) throws IOException {
135        if (listener != null) {
136            listener.values(values);
137        }
138        return this;
139    }
140
141    @Override
142    public KeyValueReader end() throws IOException {
143        if (listener != null) {
144            listener.end();
145        }
146        return this;
147    }
148
149}
