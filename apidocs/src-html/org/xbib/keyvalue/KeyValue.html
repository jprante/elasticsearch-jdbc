032package org.xbib.keyvalue;
033
034public class KeyValue<K, V> {
035
036    private final K key;
037
038    private final V value;
039
040    public KeyValue(K key, V value) {
041        this.key = key;
042        this.value = value;
043    }
044
045    public K key() {
046        return key;
047    }
048
049    public V value() {
050        return value;
051    }
052
053    public String toString() {
054        return String.valueOf(key) + "=" + value;
055    }
056}
